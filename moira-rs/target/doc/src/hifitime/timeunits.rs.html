55/// A trait to automatically convert some primitives to a duration
56///
57/// ```
58/// #[cfg(feature = "std")]
59/// {
60/// use hifitime::prelude::*;
61/// use std::str::FromStr;
62///
63/// assert_eq!(Duration::from_str("1 d").unwrap(), 1.days());
64/// assert_eq!(Duration::from_str("10.598 days").unwrap(), 10.598.days());
65/// assert_eq!(Duration::from_str("10.598 min").unwrap(), 10.598.minutes());
66/// assert_eq!(Duration::from_str("10.598 us").unwrap(), 10.598.microseconds());
67/// assert_eq!(Duration::from_str("10.598 seconds").unwrap(), 10.598.seconds());
68/// assert_eq!(Duration::from_str("10.598 nanosecond").unwrap(), 10.598.nanoseconds());
69/// }
70/// ```
71pub trait TimeUnits: Copy + Mul<Unit, Output = Duration> {
72    fn centuries(self) -> Duration {
73        self * Unit::Century
74    }
75    fn weeks(self) -> Duration {
76        self * Unit::Week
77    }
78    fn days(self) -> Duration {
79        self * Unit::Day
80    }
81    fn hours(self) -> Duration {
82        self * Unit::Hour
83    }
84    fn minutes(self) -> Duration {
85        self * Unit::Minute
86    }
87    fn seconds(self) -> Duration {
88        self * Unit::Second
89    }
90    fn milliseconds(self) -> Duration {
91        self * Unit::Millisecond
92    }
93    fn microseconds(self) -> Duration {
94        self * Unit::Microsecond
95    }
96    fn nanoseconds(self) -> Duration {
97        self * Unit::Nanosecond
98    }
99}

101/// A trait to automatically convert some primitives to an approximate frequency as a duration.
102///
103/// **Note on Precision:** The conversion is rounded to the closest nanosecond because `Duration`
104/// has nanosecond precision. Frequencies greater than 1 GHz (i.e., periods less than 1 nanosecond)
105/// cannot be accurately represented. Such high frequencies will result in a `Duration` of zero
106/// or an inaccurate, truncated nanosecond value.
107///
108/// ```
109/// use hifitime::prelude::*;
110///
111/// assert_eq!(1.Hz(), 1.seconds());
112/// assert_eq!(10.Hz(), 0.1.seconds());
113/// assert_eq!(100.Hz(), 0.01.seconds());
114/// assert_eq!(1.MHz(), 1.microseconds());
115/// assert_eq!(250.MHz(), 4.nanoseconds());
116/// assert_eq!(1.GHz(), 1.nanoseconds());
117/// // LIMITATIONS
118/// assert_eq!(240.MHz(), 4.nanoseconds()); // 240 MHz is actually 4.1666.. nanoseconds, not 4 exactly!
119/// assert_eq!(10.GHz(), 0.nanoseconds()); // As 10 GHz corresponds to 0.1 ns, it's truncated to 0 ns.
120/// ```
121#[allow(non_snake_case)]
122pub trait Frequencies: Copy + Mul<Freq, Output = Duration> {
123    fn GHz(self) -> Duration {
124        self * Freq::GigaHertz
125    }
126    fn MHz(self) -> Duration {
127        self * Freq::MegaHertz
128    }
129    fn kHz(self) -> Duration {
130        self * Freq::KiloHertz
131    }
132    fn Hz(self) -> Duration {
133        self * Freq::Hertz
134    }
135}

170    pub fn in_seconds(&self) -> f64 {
171        match self {
172            Unit::Century => DAYS_PER_CENTURY * SECONDS_PER_DAY,
173            Unit::Week => DAYS_PER_WEEK * SECONDS_PER_DAY,
174            Unit::Day => SECONDS_PER_DAY,
175            Unit::Hour => SECONDS_PER_HOUR,
176            Unit::Minute => SECONDS_PER_MINUTE,
177            Unit::Second => 1.0,
178            Unit::Millisecond => 1e-3,
179            Unit::Microsecond => 1e-6,
180            Unit::Nanosecond => 1e-9,
181        }
182    }
183
184    #[must_use]
185    pub fn from_seconds(&self) -> f64 {
186        1.0 / self.in_seconds()
187    }
188
189    #[cfg(feature = "python")]

191        *self + other
192    }
193

196        *self - other
197    }
198
199    #[cfg(feature = "python")]
200    fn __mul__(&self, other: f64) -> Duration {
201        *self * other
202    }

206/// 0: Second; 1: Nanosecond; 2: Microsecond; 3: Millisecond; 4: Minute; 5: Hour; 6: Day; 7: Century

214            Unit::Hour => 5,

217            Unit::Century => 8,
218            Unit::Second => 0,

220    }
221}

246impl Mul<i64> for Unit {
247    type Output = Duration;
248
249    /// Converts the input values to i128 and creates a duration from that
250    /// This method will necessarily ignore durations below nanoseconds
251    fn mul(self, q: i64) -> Duration {
252        let factor = match self {
253            Unit::Century => NANOSECONDS_PER_CENTURY as i64,
254            Unit::Week => NANOSECONDS_PER_DAY as i64 * DAYS_PER_WEEK_I64,
255            Unit::Day => NANOSECONDS_PER_DAY as i64,
256            Unit::Hour => NANOSECONDS_PER_HOUR as i64,
257            Unit::Minute => NANOSECONDS_PER_MINUTE as i64,
258            Unit::Second => NANOSECONDS_PER_SECOND as i64,
259            Unit::Millisecond => NANOSECONDS_PER_MILLISECOND as i64,
260            Unit::Microsecond => NANOSECONDS_PER_MICROSECOND as i64,
261            Unit::Nanosecond => 1,
262        };
263
264        match q.checked_mul(factor) {
265            Some(total_ns) => {
266                if total_ns.abs() < i64::MAX {
267                    Duration::from_truncated_nanoseconds(total_ns)
268                } else {
269                    Duration::from_total_nanoseconds(i128::from(total_ns))
270                }
271            }
272            None => {
273                // Does not fit on an i64, let's do this again on an 128.
274                let q = i128::from(q);
275                match q.checked_mul(factor.into()) {
276                    Some(total_ns) => Duration::from_total_nanoseconds(total_ns),
277                    None => {
278                        if q.is_negative() {
279                            Duration::MIN
280                        } else {
281                            Duration::MAX
282                        }
283                    }
284                }
285            }
286        }
287    }
288}

290impl Mul<f64> for Unit {
291    type Output = Duration;
292
293    /// Creates a duration from that f64
294    ///
295    /// ## Limitations
296    /// 1. If the input value times the unit does not fit on a Duration, then Duration::MAX or Duration::MIN will be returned depending on whether the value would have overflowed or underflowed (respectively).
297    /// 2. Floating point operations may round differently on different processors. It's advised to use integer initialization of Durations whenever possible.
298    fn mul(self, q: f64) -> Duration {
299        self.const_multiply(q)
300    }
301}
302
303impl Unit {
304    /// `const`-compatible copy of [Self::mul].
305    pub(crate) const fn const_multiply(self, q: f64) -> Duration {
306        let factor = match self {
307            Unit::Century => NANOSECONDS_PER_CENTURY as f64,
308            Unit::Week => NANOSECONDS_PER_DAY as f64 * DAYS_PER_WEEK,
309            Unit::Day => NANOSECONDS_PER_DAY as f64,
310            Unit::Hour => NANOSECONDS_PER_HOUR as f64,
311            Unit::Minute => NANOSECONDS_PER_MINUTE as f64,
312            Unit::Second => NANOSECONDS_PER_SECOND as f64,
313            Unit::Millisecond => NANOSECONDS_PER_MILLISECOND as f64,
314            Unit::Microsecond => NANOSECONDS_PER_MICROSECOND as f64,
315            Unit::Nanosecond => 1.0,
316        };
317
318        // Bound checking to prevent overflows
319        if q >= f64::MAX / factor {
320            Duration::MAX
321        } else if q <= f64::MIN / factor {
322            Duration::MIN
323        } else {
324            let total_ns = q * factor;
325
326            // The following manual `abs()` implementation was added because
327            // `f64::abs()` was not a `const` function in Rust 1.82, which is
328            // used for the MSRV workflow at the time of writing.
329            let absolute_nanoseconds = if total_ns >= 0.0 { total_ns } else { -total_ns };
330
331            if absolute_nanoseconds < (i64::MAX as f64) {
332                Duration::from_truncated_nanoseconds(total_ns as i64)
333            } else {
334                Duration::from_total_nanoseconds(total_ns as i128)
335            }
336        }
337    }
338}

340#[test]
341fn test_unit_conversion() {
342    for unit_u8 in 0..u8::MAX {
343        let unit = Unit::from(unit_u8);
344        let unit_u8_back: u8 = unit.into();
345        // If the u8 is greater than 9, it isn't valid and necessarily encoded as Second.
346        if unit_u8 < 9 {
347            assert_eq!(unit_u8_back, unit_u8, "got {unit_u8_back} want {unit_u8}");
348        } else {
349            assert_eq!(unit, Unit::Second);
350        }
351    }
