21const MAX_TOKENS: usize = 16;

23/// Format allows formatting an Epoch with some custom arrangement of the Epoch items.
24/// This provides almost all of the options from the 1989 C standard.
25///
26/// Construct a format with `Format::from_str` (no allocation needed) where the string
27/// contains a succession of tokens (each starting with `%`) and up to two separators between tokens.
28///
29/// Then this format can then be provided to the `Formatter` for formatting. This is also no-std.
30///
31/// # Supported tokens
32///
33/// Any token may be followed by `?` to make it optional. For integer values, an optional token will only be printed
34/// if its value is non-zero. For time scales, only non-UTC time scale will be printed.
35///
36/// ## C89 standard tokens
37///
38/// | Token | Explanation | Example | Notes
39/// | :-- | :-- | :-- | :-- |
40/// | `%Y` | Proleptic Gregorian year, zero-padded to 4 digits | `2022` | (1) |
41/// | `%y` | Proleptic Gregorian year after 2000, on two digits | `23` | N/A |
42/// | `%m` | Month number, zero-padded to 2 digits | `03` for March | N/A |
43/// | `%b` | Month name in short form | `Mar` for March | N/A |
44/// | `%B` | Month name in long form | `March` | N/A |
45/// | `%d` | Day number, zero-padded to 2 digits | `07` for the 7th day of the month | N/A |
46/// | `%j` | Day of year, zero-padded to 3 digits | `059` for 29 February 2000 | N/A |
47/// | `%A` | Weekday name in long form | `Monday` | N/A |
48/// | `%a` | Weekday name in short form | `Mon` for Monday | N/A |
49/// | `%H` | Hour number, zero-padded to 2 digits | `02` for the 2nd hour of the day | N/A |
50/// | `%M` | Minute number, zero-padded to 2 digits | `39` for the 39th minutes of the hour | N/A |
51/// | `%S` | Seconds, zero-padded to 2 digits | `27` for the 27th second of the minute | N/A |
52/// | `%f` | Sub-seconds, zero-padded to 9 digits | `000000007` for the 7th nanosecond past the second | (2) |
53/// | `%w` | Weekday in decimal form with C89 standard | `01` for Dynamical barycentric time | (3) |
54/// | `%z` | Offset timezone if the formatter is provided with an epoch. | `+15:00` For GMT +15 hours and zero minutes | N/A |
55///
56/// * (1): Hifitime supports years from -34668 to 34668. If your epoch is larger than +/- 9999 years, the formatting of the years _will_ show all five digits of the year.
57/// * (2): Hifitime supports exactly nanosecond precision, and this is not lost when formatting.
58///
59/// ## Hifitime specific tokens
60///
61/// These are chosen to not conflict with strptime/strfime of the C89 standard.
62/// | Token | Explanation | Example | Notes
63/// | :-- | :-- | :-- | :-- |
64/// | `%T` | Time scale used to represent this date | `TDB` for Dynamical barycentric time | (3) |
65/// | `%J` | Full day of year as a double | `59.62325231481524` for 29 February 2000 14:57:29 UTC | N/A |
66///
67/// * (3): Hifitime supports many time scales and these should not be lost when formatting. **This is a novelty compared to other time management libraries** as most do not have any concept of time scales.
68///
69///
70/// # Example
71/// ```
72/// use hifitime::prelude::*;
73/// use hifitime::efmt;
74/// use hifitime::efmt::consts;
75/// use core::str::FromStr;
76///
77/// let bday = Epoch::from_gregorian_utc(2000, 2, 29, 14, 57, 29, 37);
78///
79/// let fmt = efmt::Format::from_str("%Y-%m-%d").unwrap();
80/// assert_eq!(fmt, consts::ISO8601_DATE);
81///
82/// let fmtd_bday = Formatter::new(bday, fmt);
83/// assert_eq!(format!("{fmtd_bday}"), format!("2000-02-29"));
84///
85/// let fmt = Format::from_str("%Y-%m-%dT%H:%M:%S.%f %T").unwrap();
86/// assert_eq!(fmt, consts::ISO8601);
87///
88/// let fmtd_bday = Formatter::new(bday, fmt);
89/// // ISO with the timescale is the default format
90/// assert_eq!(format!("{fmtd_bday}"), format!("{bday}"));
91///
92/// let fmt = Format::from_str("%Y-%j").unwrap();
93/// assert_eq!(fmt, consts::ISO8601_ORDINAL);
94///
95/// let fmt_iso_ord = Formatter::new(bday, consts::ISO8601_ORDINAL);
96/// assert_eq!(format!("{fmt_iso_ord}"), "2000-060");
97///
98/// let fmt = Format::from_str("%A, %d %B %Y %H:%M:%S").unwrap();
99/// assert_eq!(fmt, consts::RFC2822_LONG);
100///
101/// let fmt = Formatter::new(bday, consts::RFC2822_LONG);
102/// assert_eq!(
103///     format!("{fmt}"),
104///     format!("Tuesday, 29 February 2000 14:57:29")
105/// );
106///
107/// let fmt = Format::from_str("%a, %d %b %Y %H:%M:%S").unwrap();
108/// assert_eq!(fmt, consts::RFC2822);
109///
110/// let fmt = Formatter::new(bday, consts::RFC2822);
111/// assert_eq!(format!("{fmt}"), format!("Tue, 29 Feb 2000 14:57:29"));
112/// ```
113#[cfg_attr(kani, derive(kani::Arbitrary))]
114#[derive(Copy, Clone, Default, PartialEq)]
115pub struct Format {
116    pub(crate) items: [Option<Item>; MAX_TOKENS],
117    pub(crate) num_items: usize,
118}
119
120impl Format {
121    pub(crate) fn need_gregorian(&self) -> bool {
122        for item in self.items.iter().take(self.num_items) {
123            match item.as_ref().unwrap().token {
124                Token::Year
125                | Token::YearShort
126                | Token::Month
127                | Token::MonthName
128                | Token::MonthNameShort
129                | Token::Day
130                | Token::Hour
131                | Token::Minute
132                | Token::Second
133                | Token::Subsecond
134                | Token::OffsetHours
135                | Token::OffsetMinutes => return true,
136                Token::Timescale
137                | Token::DayOfYearInteger
138                | Token::DayOfYear
139                | Token::Weekday
140                | Token::WeekdayShort
141                | Token::WeekdayDecimal => {
142                    // These tokens don't need the gregorian, but other tokens in the list of tokens might.
143                    // Hence, we don't return anything here and continue the loop.
144                }
145            }
146        }
147        false
148    }
149
150    pub fn parse(&self, s_in: &str) -> Result<Epoch, HifitimeError> {
151        // All of the integers in a date: year, month, day, hour, minute, second, subsecond, offset hours, offset minutes
152        let mut decomposed = [0_i32; MAX_TOKENS];
153        // The parsed time scale, defaults to UTC
154        let mut ts = TimeScale::UTC;
155        // The offset sign, defaults to positive.
156        let mut offset_sign = 1;
157        let mut day_of_year: Option<f64> = None;
158        let mut weekday: Option<Weekday> = None;
159
160        // Previous index of interest in the string
161        let mut prev_idx = 0;
162        let mut cur_item_idx = 0;
163        let mut cur_item = match self.items[cur_item_idx] {
164            Some(item) => item,
165            None => {
166                return Err(HifitimeError::Parse {
167                    source: ParsingError::NothingToParse,
168                    details: "format string contains no tokens",
169                })
170            }
171        };
172        let mut cur_token = cur_item.token;
173        let mut prev_item = cur_item;
174        let mut prev_token;
175
176        let s = s_in.trim();
177
178        for (idx, char) in s.char_indices() {
179            // We should parse if:
180            // 1. we're at the end of the string
181            // 2. Or we've hit a non-numeric char and the token is fully numeric
182            // 3. Or, token is not numeric (e.g. month name) and the current char is the separator
183            // 4. And, if the length of the current substring is longer than 1 and the char is not the optional separator of the previous token.
184            if idx == s.len() - 1
185                || ((cur_token.is_numeric() && !char.is_numeric())
186                    || (!cur_token.is_numeric() && (cur_item.sep_char_is(char))))
187            {
188                // If we've found the second separator of the previous token, let's simply increment the start index of the next substring.
189                if idx == prev_idx
190                    && (prev_item.second_sep_char.is_none() || prev_item.second_sep_char_is(char))
191                {
192                    prev_idx += 1;
193                    continue;
194                }
195
196                if cur_token == Token::Timescale {
197                    // Then we match the timescale directly.
198                    if idx != s.len() - 1 {
199                        // We have some remaining characters, so let's parse those in the only formats we know.
200                        ts = TimeScale::from_str(s[idx..].trim()).with_context(|_| ParseSnafu {
201                            details: "when parsing from format string",
202                        })?;
203                    }
204                    break;
205                } else if char == 'Z' {
206                    // This is a single character to represent UTC
207                    // UTC is the default time scale, so we don't need to do anything.
208                    break;
209                }
210                prev_item = cur_item;
211                prev_token = cur_token;
212
213                let end_idx = if idx != s.len() - 1 || !char.is_numeric() {
214                    // Only advance the token if we aren't at the end of the string
215                    if cur_item.sep_char_is_not(char)
216                        && (cur_item.second_sep_char.is_none()
217                            || (cur_item.second_sep_char_is_not(char)))
218                    {
219                        return Err(HifitimeError::Parse {
220                            source: ParsingError::UnexpectedCharacter {
221                                found: char,
222                                option1: cur_item.sep_char,
223                                option2: cur_item.second_sep_char,
224                            },
225                            details: "when parsing from format string",
226                        });
227                    }
228
229                    // Advance the token, unless we're at the end of the tokens.
230                    if cur_item_idx == self.num_items {
231                        break;
232                    }
233                    cur_item_idx += 1;
234                    match self.items[cur_item_idx] {
235                        Some(item) => {
236                            cur_item = item;
237                            cur_token = cur_item.token;
238                        }
239                        None => break,
240                    }
241
242                    idx
243                } else {
244                    idx + 1
245                };
246
247                let sub_str = &s[prev_idx..end_idx];
248
249                match prev_token {
250                    Token::YearShort => {
251                        decomposed[0] =
252                            sub_str.parse::<i32>().map_err(|_| HifitimeError::Parse {
253                                source: ParsingError::ValueError,
254                                details: "could not parse year as i32",
255                            })? + 2000;
256                    }
257                    Token::DayOfYear => {
258                        // We must parse this as a floating point value.
259                        match lexical_core::parse(sub_str.as_bytes()) {
260                            Ok(val) => day_of_year = Some(val),
261                            Err(_) => {
262                                return Err(HifitimeError::Parse {
263                                    source: ParsingError::ValueError,
264                                    details: "could not parse day of year as f64",
265                                })
266                            }
267                        }
268                    }
269                    Token::Weekday | Token::WeekdayShort => {
270                        // Set the weekday
271                        match Weekday::from_str(sub_str) {
272                            Ok(day) => weekday = Some(day),
273                            Err(source) => {
274                                return Err(HifitimeError::Parse {
275                                    source,
276                                    details: "could not parse weekday",
277                                })
278                            }
279                        }
280                    }
281                    Token::WeekdayDecimal => {
282                        todo!()
283                    }
284                    Token::MonthName | Token::MonthNameShort => {
285                        match MonthName::from_str(sub_str) {
286                            Ok(month) => {
287                                decomposed[1] = ((month as u8) + 1) as i32;
288                            }
289                            Err(_) => {
290                                return Err(HifitimeError::Parse {
291                                    source: ParsingError::ValueError,
292                                    details: "could not parse month name",
293                                })
294                            }
295                        }
296                    }
297                    _ => {
298                        match lexical_core::parse(sub_str.as_bytes()) {
299                            Ok(val) => {
300                                // Check that this valid is OK for the token we're reading it as.
301                                prev_token.value_ok(val)?;
302                                match prev_token.gregorian_position() {
303                                    Some(pos) => {
304                                        // If these are the subseconds, we must convert them to nanoseconds
305                                        if prev_token == Token::Subsecond {
306                                            if end_idx - prev_idx != 9 {
307                                                decomposed[pos] = val
308                                                    * 10_i32.pow((9 - (end_idx - prev_idx)) as u32);
309                                            } else {
310                                                decomposed[pos] = val;
311                                            }
312                                        } else {
313                                            decomposed[pos] = val
314                                        }
315                                    }
316                                    None => match prev_token {
317                                        Token::DayOfYearInteger => day_of_year = Some(val as f64),
318                                        Token::Weekday => todo!(),
319                                        Token::WeekdayShort => todo!(),
320                                        Token::WeekdayDecimal => todo!(),
321                                        Token::MonthName => todo!(),
322                                        Token::MonthNameShort => todo!(),
323                                        _ => unreachable!(),
324                                    },
325                                }
326                            }
327                            Err(err) => {
328                                return Err(HifitimeError::Parse {
329                                    source: ParsingError::Lexical { err },
330                                    details: "could not parse numerical",
331                                });
332                            }
333                        }
334                    }
335                }
336
337                prev_idx = idx + 1;
338                // If we are about to parse an hours offset, we need to set the sign now.
339                if cur_token == Token::OffsetHours {
340                    if &s[idx..idx + 1] == "-" {
341                        offset_sign = -1;
342                    }
343                    prev_idx += 1;
344                }
345            }
346        }
347
348        let tz = if offset_sign > 0 {
349            // We oppose the sign in the string to undo the offset
350            -(i64::from(decomposed[7]) * Unit::Hour + i64::from(decomposed[8]) * Unit::Minute)
351        } else {
352            i64::from(decomposed[7]) * Unit::Hour + i64::from(decomposed[8]) * Unit::Minute
353        };
354
355        let epoch = match day_of_year {
356            Some(days) => {
357                // Parse the elapsed time in the given day
358                let elapsed = (decomposed[3] as i64) * Unit::Hour
359                    + (decomposed[4] as i64) * Unit::Minute
360                    + (decomposed[5] as i64) * Unit::Second
361                    + (decomposed[6] as i64) * Unit::Nanosecond;
362                Epoch::from_day_of_year(decomposed[0], days, ts) + elapsed
363            }
364            None => Epoch::maybe_from_gregorian(
365                decomposed[0],
366                decomposed[1].try_into().unwrap(),
367                decomposed[2].try_into().unwrap(),
368                decomposed[3].try_into().unwrap(),
369                decomposed[4].try_into().unwrap(),
370                decomposed[5].try_into().unwrap(),
371                decomposed[6].try_into().unwrap(),
372                ts,
373            )?,
374        };
375
376        if let Some(weekday) = weekday {
377            // Check that the weekday is correct
378            if weekday != epoch.weekday() {
379                return Err(HifitimeError::Parse {
380                    source: ParsingError::WeekdayMismatch {
381                        found: weekday,
382                        expected: epoch.weekday(),
383                    },
384                    details: "weekday and day number do not match",
385                });
386            }
387        }
388
389        Ok(epoch + tz)
390    }
391}

393impl fmt::Debug for Format {
394    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
395        write!(f, "EpochFormat:`")?;
396        for maybe_item in self.items.iter().take(self.num_items) {
397            let item = maybe_item.as_ref().unwrap();
398            write!(f, "{:?}", item.token)?;
399            if let Some(char) = item.sep_char {
400                write!(f, "{char}")?;
401            }
402            if let Some(char) = item.second_sep_char {
403                write!(f, "{char}")?;
404            }
405            if item.optional {
406                write!(f, "?")?;
407            }
408        }
409        write!(f, "`")?;
410        Ok(())
411    }
412}

414impl FromStr for Format {
415    type Err = ParsingError;
416    fn from_str(s: &str) -> Result<Self, Self::Err> {
417        let mut me = Format::default();
418        for token in s.split('%') {
419            if me.num_items == MAX_TOKENS && token.chars().next().is_some() {
420                return Err(ParsingError::UnknownFormat);
421            }
422            match token.chars().next() {
423                Some(char) => match char {
424                    'Y' => {
425                        me.items[me.num_items] = Some(Item::new(
426                            Token::Year,
427                            token.chars().nth(1),
428                            token.chars().nth(2),
429                        ));
430                        me.num_items += 1;
431                    }
432                    'y' => {
433                        me.items[me.num_items] = Some(Item::new(
434                            Token::YearShort,
435                            token.chars().nth(1),
436                            token.chars().nth(2),
437                        ));
438                        me.num_items += 1;
439                    }
440                    'm' => {
441                        me.items[me.num_items] = Some(Item::new(
442                            Token::Month,
443                            token.chars().nth(1),
444                            token.chars().nth(2),
445                        ));
446                        me.num_items += 1;
447                    }
448                    'b' => {
449                        me.items[me.num_items] = Some(Item::new(
450                            Token::MonthNameShort,
451                            token.chars().nth(1),
452                            token.chars().nth(2),
453                        ));
454                        me.num_items += 1;
455                    }
456                    'B' => {
457                        me.items[me.num_items] = Some(Item::new(
458                            Token::MonthName,
459                            token.chars().nth(1),
460                            token.chars().nth(2),
461                        ));
462                        me.num_items += 1;
463                    }
464                    'd' => {
465                        me.items[me.num_items] = Some(Item::new(
466                            Token::Day,
467                            token.chars().nth(1),
468                            token.chars().nth(2),
469                        ));
470                        me.num_items += 1;
471                    }
472                    'j' => {
473                        me.items[me.num_items] = Some(Item::new(
474                            Token::DayOfYearInteger,
475                            token.chars().nth(1),
476                            token.chars().nth(2),
477                        ));
478                        me.num_items += 1;
479                    }
480                    'J' => {
481                        me.items[me.num_items] = Some(Item::new(
482                            Token::DayOfYear,
483                            token.chars().nth(1),
484                            token.chars().nth(2),
485                        ));
486                        me.num_items += 1;
487                    }
488                    'A' => {
489                        me.items[me.num_items] = Some(Item::new(
490                            Token::Weekday,
491                            token.chars().nth(1),
492                            token.chars().nth(2),
493                        ));
494                        me.num_items += 1;
495                    }
496                    'a' => {
497                        me.items[me.num_items] = Some(Item::new(
498                            Token::WeekdayShort,
499                            token.chars().nth(1),
500                            token.chars().nth(2),
501                        ));
502                        me.num_items += 1;
503                    }
504                    'H' => {
505                        me.items[me.num_items] = Some(Item::new(
506                            Token::Hour,
507                            token.chars().nth(1),
508                            token.chars().nth(2),
509                        ));
510                        me.num_items += 1;
511                    }
512                    'M' => {
513                        me.items[me.num_items] = Some(Item::new(
514                            Token::Minute,
515                            token.chars().nth(1),
516                            token.chars().nth(2),
517                        ));
518                        me.num_items += 1;
519                    }
520                    'S' => {
521                        me.items[me.num_items] = Some(Item::new(
522                            Token::Second,
523                            token.chars().nth(1),
524                            token.chars().nth(2),
525                        ));
526                        me.num_items += 1;
527                    }
528                    'f' => {
529                        me.items[me.num_items] = Some(Item::new(
530                            Token::Subsecond,
531                            token.chars().nth(1),
532                            token.chars().nth(2),
533                        ));
534                        me.num_items += 1;
535                    }
536                    'T' => {
537                        me.items[me.num_items] = Some(Item::new(
538                            Token::Timescale,
539                            token.chars().nth(1),
540                            token.chars().nth(2),
541                        ));
542                        me.num_items += 1;
543                    }
544                    'w' => {
545                        me.items[me.num_items] = Some(Item::new(
546                            Token::WeekdayDecimal,
547                            token.chars().nth(1),
548                            token.chars().nth(2),
549                        ));
550                        me.num_items += 1;
551                    }
552                    'z' => {
553                        me.items[me.num_items] = Some(Item::new(
554                            Token::OffsetHours,
555                            token.chars().nth(1),
556                            token.chars().nth(2),
557                        ));
558                        me.num_items += 1;
559                    }
560                    _ => return Err(ParsingError::UnknownToken { token: char }),
561                },
562                None => continue, // We're probably just at the start of the string
563            }
564        }
565
566        Ok(me)
567    }
568}

570#[test]
571fn epoch_format_from_str() {
572    let fmt = Format::from_str("%Y-%m-%d").unwrap();
573    assert_eq!(fmt, crate::efmt::consts::ISO8601_DATE);
574
575    let fmt = Format::from_str("%Y-%m-%dT%H:%M:%S.%f %T").unwrap();
576    assert_eq!(fmt, crate::efmt::consts::ISO8601);
577
578    let fmt = Format::from_str("%Y-%m-%dT%H:%M:%S.%f? %T?").unwrap();
579    assert_eq!(fmt, crate::efmt::consts::ISO8601_FLEX);
580
581    let fmt = Format::from_str("%Y-%j").unwrap();
582    assert_eq!(fmt, crate::efmt::consts::ISO8601_ORDINAL);
583
584    let fmt = Format::from_str("%A, %d %B %Y %H:%M:%S").unwrap();
585    assert_eq!(fmt, crate::efmt::consts::RFC2822_LONG);
586
587    let fmt = Format::from_str("%a, %d %b %Y %H:%M:%S").unwrap();
588    assert_eq!(fmt, crate::efmt::consts::RFC2822);
589}

591#[cfg(feature = "std")]
592#[test]
593fn gh_248_regression() {
594    /*
595    Update on 2023-12-30 to match the Python behavior:
596
597    >>> from datetime import datetime
598    >>> dt, fmt = "2023-117T12:55:26", "%Y-%jT%H:%M:%S"
599    >>> datetime.strptime(dt, fmt)
600    datetime.datetime(2023, 4, 27, 12, 55, 26)
601     */
602
603    let e = Epoch::from_format_str("2023-117T12:55:26", "%Y-%jT%H:%M:%S").unwrap();
604
605    assert_eq!(format!("{e}"), "2023-04-27T12:55:26 UTC");
606}
