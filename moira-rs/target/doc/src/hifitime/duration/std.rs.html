17impl From<Duration> for std::time::Duration {
18    /// Converts a Duration into a std::time::Duration
19    ///
20    /// # Limitations
21    /// 1. If the Duration is negative, this will return a std::time::Duration::ZERO.
22    /// 2. If the Duration is Duration::MAX, this will return the equivalent of std::time::Duration::from_secs(103407943680000)
23    fn from(hf_duration: Duration) -> Self {
24        use crate::NANOSECONDS_PER_SECOND;
25        if hf_duration.signum() == -1 {
26            std::time::Duration::ZERO
27        } else {
28            let nanos = hf_duration.total_nanoseconds();
29            let unsigned_nanos = u128::try_from(nanos).unwrap_or(0);
30
31            let secs: u64 = (unsigned_nanos / NANOSECONDS_PER_SECOND as u128)
32                .try_into()
33                .unwrap_or(u64::MAX);
34            let subsec_nanos = (unsigned_nanos % NANOSECONDS_PER_SECOND as u128) as u32;
35
36            std::time::Duration::new(secs, subsec_nanos)
37        }
38    }
39}
40
41impl From<std::time::Duration> for Duration {
42    /// Converts a std::time::Duration into a Duration
43    ///
44    /// # Limitations
45    /// 1. If the std::time::Duration is larger than Duration::MAX, this will return Duration::MAX
46    fn from(std_duration: std::time::Duration) -> Self {
47        Duration::from_total_nanoseconds(std_duration.as_nanos().try_into().unwrap_or(i128::MAX))
48    }
49}
