14/// Julian date for the J1900 epoch, as per NAIF SPICE.
15pub const JD_J1900: f64 = 2_415_020.0;
16/// Julian date for the J2000 epoch, as per NAIF SPICE.
17pub const JD_J2000: f64 = 2_451_545.0;
18/// Julian days between 01 Jan 1900 at midnight and the Modified Julian Day at 17 November 1858.
19pub const MJD_J1900: f64 = 15_020.0;
20/// Julian days between 01 Jan 2000 at **noon** and the Modified Julian Day at 17 November 1858.
21pub const MJD_J2000: f64 = 51_544.5;
22/// The Ephemeris Time epoch, in seconds
23pub const ET_EPOCH_S: i64 = 3_155_716_800;
24/// Modified Julian Date in seconds as defined [here](http://tycho.usno.navy.mil/mjd.html). MJD epoch is Modified Julian Day at 17 November 1858 at midnight.
25pub const MJD_OFFSET: f64 = 2_400_000.5;
26/// `DAYS_PER_YEAR` corresponds to the number of days per year in the Julian calendar.
27pub const DAYS_PER_YEAR: f64 = 365.25;
28/// `DAYS_PER_YEAR_NLD` corresponds to the number of days per year **without leap days**.
29pub const DAYS_PER_YEAR_NLD: f64 = 365.0;
30/// `DAYS_PER_CENTURY` corresponds to the number of days per century in the Julian calendar.
31pub const DAYS_PER_CENTURY: f64 = 36525.0;
32pub const DAYS_PER_CENTURY_I64: i64 = 36525;
33pub const DAYS_PER_WEEK: f64 = 7.0;
34pub const DAYS_PER_WEEK_I64: i64 = 7;
35/// `SECONDS_PER_MINUTE` defines the number of seconds per minute.
36pub const SECONDS_PER_MINUTE: f64 = 60.0;
37/// `SECONDS_PER_HOUR` defines the number of seconds per hour.
38pub const SECONDS_PER_HOUR: f64 = 3_600.0;
39/// `SECONDS_PER_DAY` defines the number of seconds per day.
40pub const SECONDS_PER_DAY: f64 = 86_400.0;
41pub const SECONDS_PER_DAY_I64: i64 = 86_400;
42/// `SECONDS_PER_CENTURY` defines the number of seconds per century.
43pub const SECONDS_PER_CENTURY: f64 = SECONDS_PER_DAY * DAYS_PER_CENTURY;
44/// `SECONDS_PER_YEAR` corresponds to the number of seconds per julian year from [NAIF SPICE](https://naif.jpl.nasa.gov/pub/naif/toolkit_docs/C/cspice/jyear_c.html).
45pub const SECONDS_PER_YEAR: f64 = 31_557_600.0;
46pub const SECONDS_PER_YEAR_I64: i64 = 31_557_600;
47/// `SECONDS_PER_TROPICAL_YEAR` corresponds to the number of seconds per tropical year from [NAIF SPICE](https://naif.jpl.nasa.gov/pub/naif/toolkit_docs/C/cspice/tyear_c.html).
48pub const SECONDS_PER_TROPICAL_YEAR: f64 = 31_556_925.974_7;
49/// `SECONDS_PER_SIDEREAL_YEAR` corresponds to the number of seconds per sidereal year from [NIST](https://www.nist.gov/pml/special-publication-811/nist-guide-si-appendix-b-conversion-factors/nist-guide-si-appendix-b9#TIME).
50pub const SECONDS_PER_SIDEREAL_YEAR: f64 = 31_558_150.0;
