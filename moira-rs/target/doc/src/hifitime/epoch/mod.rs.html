66/// NAIF leap second kernel data for M_0 used to calculate the mean anomaly of the heliocentric orbit of the Earth-Moon barycenter.
67pub const NAIF_M0: f64 = 6.239996;
68/// NAIF leap second kernel data for M_1 used to calculate the mean anomaly of the heliocentric orbit of the Earth-Moon barycenter.
69pub const NAIF_M1: f64 = 1.99096871e-7;
70/// NAIF leap second kernel data for EB used to calculate the eccentric anomaly of the heliocentric orbit of the Earth-Moon barycenter.
71pub const NAIF_EB: f64 = 1.671e-2;
72/// NAIF leap second kernel data used to calculate the difference between ET and TAI.
73pub const NAIF_K: f64 = 1.657e-3;

119    /// Get the accumulated number of leap seconds up to this Epoch from the provided LeapSecondProvider.
120    /// Returns None if the epoch is before 1960, year at which UTC was defined.
121    ///
122    /// # Why does this function return an `Option` when the other returns a value
123    /// This is to match the `iauDat` function of SOFA (src/dat.c). That function will return a warning and give up if the start date is before 1960.
124    pub fn leap_seconds_with<L: LeapSecondProvider>(
125        &self,
126        iers_only: bool,
127        provider: L,
128    ) -> Option<f64> {
129        for leap_second in provider.rev() {
130            if self.to_tai_duration().to_seconds() >= leap_second.timestamp_tai_s
131                && (!iers_only || leap_second.announced_by_iers)
132            {
133                return Some(leap_second.delta_at);
134            }
135        }
136        None
137    }

139    /// Creates an epoch from given duration expressed in given timescale, i.e. since the given time scale's reference epoch.
140    ///
141    /// For example, if the duration is 1 day and the time scale is Ephemeris Time, then this will create an epoch of 2000-01-02 at midnight ET. If the duration is 1 day and the time scale is TAI, this will create an epoch of 1900-01-02 at noon, because the TAI reference epoch in Hifitime is chosen to be the J1900 epoch.
142    /// In case of ET, TDB Timescales, a duration since J2000 is expected.
143    #[must_use]
144    pub const fn from_duration(duration: Duration, ts: TimeScale) -> Self {
145        Self {
146            duration,
147            time_scale: ts,
148        }
149    }
150
151    fn delta_et_tai(seconds: f64) -> f64 {
152        // Calculate M, the mean anomaly.4
153        let m = NAIF_M0 + seconds * NAIF_M1;
154        // Calculate eccentric anomaly
155        let e = m + NAIF_EB * m.sin();
156
157        (TT_OFFSET_MS * Unit::Millisecond).to_seconds() + NAIF_K * e.sin()
158    }
159
160    fn inner_g(seconds: f64) -> f64 {
161        use core::f64::consts::TAU;
162        let g = TAU / 360.0 * 357.528 + 1.990_910_018_065_731e-7 * seconds;
163        // Return gamma
164        1.658e-3 * (g + 1.67e-2 * g.sin()).sin()
165    }
166}

170    #[must_use]
171    /// Converts self to another time scale
172    ///
173    /// As per the [Rust naming convention](https://rust-lang.github.io/api-guidelines/naming.html#ad-hoc-conversions-follow-as_-to_-into_-conventions-c-conv),
174    /// this borrows an Epoch and returns an owned Epoch.
175    ///
176    /// :type ts: TimeScale
177    /// :rtype: Epoch
178    pub fn to_time_scale(&self, ts: TimeScale) -> Self {
179        if ts == self.time_scale {
180            // Do nothing, just return a copy
181            *self
182        } else {
183            // Now we need to convert from the current time scale into the desired time scale.
184            // Let's first compute this epoch from its current time scale into TAI.
185            let prime_epoch_offset = match self.time_scale {
186                TimeScale::TAI => self.duration,
187                TimeScale::TT => self.duration - TT_OFFSET_MS.milliseconds(),
188                TimeScale::ET => {
189                    // Run a Newton Raphston to convert find the correct value of the
190                    let mut seconds_j2000 = self.duration.to_seconds();
191                    for _ in 0..5 {
192                        seconds_j2000 += -NAIF_K
193                            * (NAIF_M0
194                                + NAIF_M1 * seconds_j2000
195                                + NAIF_EB * (NAIF_M0 + NAIF_M1 * seconds_j2000).sin())
196                            .sin();
197                    }
198
199                    // At this point, we have a good estimate of the number of seconds of this epoch.
200                    // Reverse the algorithm:
201                    let delta_et_tai = Self::delta_et_tai(
202                        seconds_j2000 - (TT_OFFSET_MS * Unit::Millisecond).to_seconds(),
203                    );
204
205                    // Match SPICE by changing the UTC definition.
206                    self.duration - delta_et_tai.seconds() + self.time_scale.prime_epoch_offset()
207                }
208                TimeScale::TDB => {
209                    let gamma = Self::inner_g(self.duration.to_seconds());
210
211                    let delta_tdb_tai = gamma * Unit::Second + TT_OFFSET_MS * Unit::Millisecond;
212
213                    // Offset back to J1900.
214                    self.duration - delta_tdb_tai + self.time_scale.prime_epoch_offset()
215                }
216                TimeScale::UTC => {
217                    // Assume this is TAI
218                    let mut tai_assumption = *self;
219                    tai_assumption.time_scale = TimeScale::TAI;
220                    self.duration + tai_assumption.leap_seconds(true).unwrap_or(0.0).seconds()
221                }
222                TimeScale::GPST => self.duration + GPST_REF_EPOCH.to_tai_duration(),
223                TimeScale::GST => self.duration + GST_REF_EPOCH.to_tai_duration(),
224                TimeScale::BDT => self.duration + BDT_REF_EPOCH.to_tai_duration(),
225                TimeScale::QZSST => self.duration + QZSST_REF_EPOCH.to_tai_duration(),
226            };
227
228            // Convert to the desired time scale from the TAI duration
229            let ts_ref_offset = match ts {
230                TimeScale::TAI => prime_epoch_offset,
231                TimeScale::TT => prime_epoch_offset + TT_OFFSET_MS.milliseconds(),
232                TimeScale::ET => {
233                    // Run a Newton Raphston to convert find the correct value of the ... ?!
234
235                    let mut seconds = (prime_epoch_offset - ts.prime_epoch_offset()).to_seconds();
236                    for _ in 0..5 {
237                        seconds -= -NAIF_K
238                            * (NAIF_M0
239                                + NAIF_M1 * seconds
240                                + NAIF_EB * (NAIF_M0 + NAIF_M1 * seconds).sin())
241                            .sin();
242                    }
243
244                    // At this point, we have a good estimate of the number of seconds of this epoch.
245                    // Reverse the algorithm:
246                    let delta_et_tai = Self::delta_et_tai(
247                        seconds + (TT_OFFSET_MS * Unit::Millisecond).to_seconds(),
248                    );
249
250                    // Match SPICE by changing the UTC definition.
251                    prime_epoch_offset + delta_et_tai.seconds() - ts.prime_epoch_offset()
252                }
253                TimeScale::TDB => {
254                    // Iterate to convert find the correct value of the
255                    let mut seconds = (prime_epoch_offset - ts.prime_epoch_offset()).to_seconds();
256                    let mut delta = 1e8; // Arbitrary large number, greater than first step of Newton Raphson.
257                    for _ in 0..5 {
258                        let next = seconds - Self::inner_g(seconds);
259                        let new_delta = (next - seconds).abs();
260                        if (new_delta - delta).abs() < 1e-9 {
261                            break;
262                        }
263                        seconds = next; // Loop
264                        delta = new_delta;
265                    }
266
267                    // At this point, we have a good estimate of the number of seconds of this epoch.
268                    // Reverse the algorithm:
269                    let gamma =
270                        Self::inner_g(seconds + (TT_OFFSET_MS * Unit::Millisecond).to_seconds());
271                    let delta_tdb_tai = gamma.seconds() + TT_OFFSET_MS.milliseconds();
272
273                    prime_epoch_offset + delta_tdb_tai - ts.prime_epoch_offset()
274                }
275                TimeScale::UTC => {
276                    // Assume it's TAI
277                    let epoch = Self {
278                        duration: prime_epoch_offset,
279                        time_scale: TimeScale::TAI,
280                    };
281                    // TAI = UTC + leap_seconds <=> UTC = TAI - leap_seconds
282                    prime_epoch_offset - epoch.leap_seconds(true).unwrap_or(0.0).seconds()
283                }
284                TimeScale::GPST => prime_epoch_offset - GPST_REF_EPOCH.to_tai_duration(),
285                TimeScale::GST => prime_epoch_offset - GST_REF_EPOCH.to_tai_duration(),
286                TimeScale::BDT => prime_epoch_offset - BDT_REF_EPOCH.to_tai_duration(),
287                TimeScale::QZSST => prime_epoch_offset - QZSST_REF_EPOCH.to_tai_duration(),
288            };
289
290            Self {
291                duration: ts_ref_offset,
292                time_scale: ts,
293            }
294        }
295    }

337    /// Attempts to return the number of nanoseconds since the reference epoch of the provided time scale.
338    /// This will return an overflow error if more than one century has past since the reference epoch in the provided time scale.
339    /// If this is _not_ an issue, you should use `epoch.to_duration_in_time_scale().to_parts()` to retrieve both the centuries and the nanoseconds
340    /// in that century.
341    ///
342    /// :type time_scale: TimeScale
343    /// :rtype: int
344    #[allow(clippy::wrong_self_convention)]
345    fn to_nanoseconds_in_time_scale(&self, time_scale: TimeScale) -> Result<u64, HifitimeError> {
346        let (centuries, nanoseconds) = self.to_duration_in_time_scale(time_scale).to_parts();
347        if centuries != 0 {
348            Err(HifitimeError::Duration {
349                source: DurationError::Overflow,
350            })
351        } else {
352            Ok(nanoseconds)
353        }
354    }

754    #[must_use]
755    /// Returns the Dynamics Barycentric Time (TDB) as a high precision Duration since J2000
756    ///
757    /// ## Algorithm
758    /// Given the embedded sine functions in the equation to compute the difference between TDB and TAI from the number of TDB seconds
759    /// past J2000, one cannot solve the revert the operation analytically. Instead, we iterate until the value no longer changes.
760    ///
761    /// 1. Assume that the TAI duration is in fact the TDB seconds from J2000.
762    /// 2. Offset to J2000 because `Epoch` stores everything in the J1900 but the TDB duration is in J2000.
763    /// 3. Compute the offset `g` due to the TDB computation with the current value of the TDB seconds (defined in step 1).
764    /// 4. Subtract that offset to the latest TDB seconds and store this as a new candidate for the true TDB seconds value.
765    /// 5. Compute the difference between this candidate and the previous one. If the difference is less than one nanosecond, stop iteration.
766    /// 6. Set the new candidate as the TDB seconds since J2000 and loop until step 5 breaks the loop, or we've done five iterations.
767    /// 7. At this stage, we have a good approximation of the TDB seconds since J2000.
768    /// 8. Reverse the algorithm given that approximation: compute the `g` offset, compute the difference between TDB and TAI, add the TT offset (32.184 s), and offset by the difference between J1900 and J2000.
769    ///
770    /// :rtype: Duration
771    pub fn to_tdb_duration(&self) -> Duration {
772        self.to_time_scale(TimeScale::TDB).duration
773    }

947#[cfg(not(kani))]
948impl FromStr for Epoch {
949    type Err = HifitimeError;
950
951    /// Attempts to convert a string to an Epoch.
952    ///
953    /// Format identifiers:
954    ///  + JD: Julian days
955    ///  + MJD: Modified Julian days
956    ///  + SEC: Seconds past a given epoch (e.g. SEC 17.2 TAI is 17.2 seconds past TAI Epoch)
957    /// # Example
958    /// ```
959    /// use hifitime::Epoch;
960    /// use core::str::FromStr;
961    ///
962    /// assert!(Epoch::from_str("JD 2452312.500372511 TDB").is_ok());
963    /// assert!(Epoch::from_str("JD 2452312.500372511 ET").is_ok());
964    /// assert!(Epoch::from_str("JD 2452312.500372511 TAI").is_ok());
965    /// assert!(Epoch::from_str("MJD 51544.5 TAI").is_ok());
966    /// assert!(Epoch::from_str("SEC 0.5 TAI").is_ok());
967    /// assert!(Epoch::from_str("SEC 66312032.18493909 TDB").is_ok());
968    /// ```
969    fn from_str(s_in: &str) -> Result<Self, Self::Err> {
970        let s = s_in.trim();
971
972        if s.len() < 7 {
973            // We need at least seven characters for a valid epoch
974            Err(HifitimeError::Parse {
975                source: ParsingError::UnknownFormat,
976                details: "less than 7 characters",
977            })
978        } else {
979            let format = if &s[..2] == "JD" {
980                "JD"
981            } else if &s[..3] == "MJD" {
982                "MJD"
983            } else if &s[..3] == "SEC" {
984                "SEC"
985            } else {
986                // Not a valid format, hopefully it's a Gregorian date.
987                return Self::from_gregorian_str(s_in);
988            };
989
990            // This is a valid numerical format.
991            // Parse the time scale from the last three characters (TS trims white spaces).
992            let ts = TimeScale::from_str(&s[s.len() - 3..]).with_context(|_| ParseSnafu {
993                details: "parsing from string",
994            })?;
995            // Iterate through the string to figure out where the numeric data starts and ends.
996            let start_idx = format.len();
997            let num_str = s[start_idx..s.len() - ts.formatted_len()].trim();
998            let value: f64 = match lexical_core::parse(num_str.as_bytes()) {
999                Ok(val) => val,
1000                Err(_) => {
1001                    return Err(HifitimeError::Parse {
1002                        source: ParsingError::ValueError,
1003                        details: "parsing as JD, MJD, or SEC",
1004                    })
1005                }
1006            };
1007
1008            match format {
1009                "JD" => match ts {
1010                    TimeScale::ET => Ok(Self::from_jde_et(value)),
1011                    TimeScale::TAI => Ok(Self::from_jde_tai(value)),
1012                    TimeScale::TDB => Ok(Self::from_jde_tdb(value)),
1013                    TimeScale::UTC => Ok(Self::from_jde_utc(value)),
1014                    _ => Err(HifitimeError::Parse {
1015                        source: ParsingError::UnsupportedTimeSystem,
1016                        details: "for Julian Date",
1017                    }),
1018                },
1019                "MJD" => match ts {
1020                    TimeScale::TAI => Ok(Self::from_mjd_tai(value)),
1021                    TimeScale::UTC | TimeScale::GPST | TimeScale::BDT | TimeScale::GST => {
1022                        Ok(Self::from_mjd_in_time_scale(value, ts))
1023                    }
1024                    _ => Err(HifitimeError::Parse {
1025                        source: ParsingError::UnsupportedTimeSystem,
1026                        details: "for Modified Julian Date",
1027                    }),
1028                },
1029                "SEC" => match ts {
1030                    TimeScale::TAI => Ok(Self::from_tai_seconds(value)),
1031                    TimeScale::ET => Ok(Self::from_et_seconds(value)),
1032                    TimeScale::TDB => Ok(Self::from_tdb_seconds(value)),
1033                    TimeScale::TT => Ok(Self::from_tt_seconds(value)),
1034                    ts => {
1035                        let secs = value * Unit::Second;
1036                        Ok(Self::from_duration(secs, ts))
1037                    }
1038                },
1039                _ => Err(HifitimeError::Parse {
1040                    source: ParsingError::UnknownFormat,
1041                    details: "suffix not understood",
1042                }),
1043            }
1044        }
1045    }
1046}

1048fn div_rem_f64(me: f64, rhs: f64) -> (i32, f64) {
1049    ((div_euclid_f64(me, rhs) as i32), rem_euclid_f64(me, rhs))
1050}
1051
1052fn div_euclid_f64(lhs: f64, rhs: f64) -> f64 {
1053    let q = (lhs / rhs).trunc();
1054    if lhs % rhs < 0.0 {
1055        if rhs > 0.0 {
1056            q - 1.0
1057        } else {
1058            q + 1.0
1059        }
1060    } else {
1061        q
1062    }
1063}
1064
1065fn rem_euclid_f64(lhs: f64, rhs: f64) -> f64 {
1066    let r = lhs % rhs;
1067    if r < 0.0 {
1068        r + rhs.abs()
1069    } else {
1070        r
1071    }
1072}
