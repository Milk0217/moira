11#[cfg(feature = "python")]
12use pyo3::prelude::*;
13
14use crate::{Duration, Epoch};

16#[cfg_attr(feature = "python", pymethods)]
17impl Epoch {
18    /// Returns a copy of self where the time is set to the provided hours, minutes, seconds
19    /// Invalid number of hours, minutes, and seconds will overflow into their higher unit.
20    /// Warning: this does _not_ set the subdivisions of second to zero.
21    /// :type hours: int
22    /// :type minutes: int
23    /// :type seconds: int
24    /// :rtype: Epoch
25    pub fn with_hms(&self, hours: u64, minutes: u64, seconds: u64) -> Self {
26        let (sign, days, _, _, _, milliseconds, microseconds, nanoseconds) =
27            self.duration.decompose();
28        Self::from_duration(
29            Duration::compose(
30                sign,
31                days,
32                hours,
33                minutes,
34                seconds,
35                milliseconds,
36                microseconds,
37                nanoseconds,
38            ),
39            self.time_scale,
40        )
41    }
42
43    /// Returns a copy of self where the hours, minutes, seconds is set to the time of the provided epoch but the
44    /// sub-second parts are kept from the current epoch.
45    ///
46    /// :type other: Epoch
47    /// :rtype: Epoch
48    /// ```
49    /// use hifitime::prelude::*;
50    ///
51    /// let epoch = Epoch::from_gregorian_utc(2022, 12, 01, 10, 11, 12, 13);
52    /// let other_utc = Epoch::from_gregorian_utc(2024, 12, 01, 20, 21, 22, 23);
53    /// let other = other_utc.to_time_scale(TimeScale::TDB);
54    ///
55    /// assert_eq!(
56    ///     epoch.with_hms_from(other),
57    ///     Epoch::from_gregorian_utc(2022, 12, 01, 20, 21, 22, 13)
58    /// );
59    /// ```
60    pub fn with_hms_from(&self, other: Self) -> Self {
61        let (sign, days, _, _, _, milliseconds, microseconds, nanoseconds) =
62            self.duration.decompose();
63        // Shadow other with the provided other epoch but in the correct time scale.
64        let other = other.to_time_scale(self.time_scale);
65        Self::from_duration(
66            Duration::compose(
67                sign,
68                days,
69                other.hours(),
70                other.minutes(),
71                other.seconds(),
72                milliseconds,
73                microseconds,
74                nanoseconds,
75            ),
76            self.time_scale,
77        )
78    }
79
80    /// Returns a copy of self where all of the time components (hours, minutes, seconds, and sub-seconds) are set to the time of the provided epoch.
81    ///
82    /// :type other: Epoch
83    /// :rtype: Epoch
84    /// ```
85    /// use hifitime::prelude::*;
86    ///
87    /// let epoch = Epoch::from_gregorian_utc(2022, 12, 01, 10, 11, 12, 13);
88    /// let other_utc = Epoch::from_gregorian_utc(2024, 12, 01, 20, 21, 22, 23);
89    /// // If the other Epoch is in another time scale, it does not matter, it will be converted to the correct time scale.
90    /// let other = other_utc.to_time_scale(TimeScale::TDB);
91    ///
92    /// assert_eq!(
93    ///     epoch.with_time_from(other),
94    ///     Epoch::from_gregorian_utc(2022, 12, 01, 20, 21, 22, 23)
95    /// );
96    /// ```
97    pub fn with_time_from(&self, other: Self) -> Self {
98        // Grab days from self
99        let (sign, days, _, _, _, _, _, _) = self.duration.decompose();
100
101        // Grab everything else from other
102        let (_, _, hours, minutes, seconds, milliseconds, microseconds, nanoseconds) =
103            other.to_duration_in_time_scale(self.time_scale).decompose();
104
105        Self::from_duration(
106            Duration::compose(
107                sign,
108                days,
109                hours,
110                minutes,
111                seconds,
112                milliseconds,
113                microseconds,
114                nanoseconds,
115            ),
116            self.time_scale,
117        )
118    }
119
120    /// Returns a copy of self where the time is set to the provided hours, minutes, seconds
121    /// Invalid number of hours, minutes, and seconds will overflow into their higher unit.
122    /// Warning: this will set the subdivisions of seconds to zero.
123    /// :type hours: int
124    /// :type minutes: int
125    /// :type seconds: int
126    /// :rtype: Epoch
127    pub fn with_hms_strict(&self, hours: u64, minutes: u64, seconds: u64) -> Self {
128        let (sign, days, _, _, _, _, _, _) = self.duration.decompose();
129        Self::from_duration(
130            Duration::compose(sign, days, hours, minutes, seconds, 0, 0, 0),
131            self.time_scale,
132        )
133    }
134
135    /// Returns a copy of self where the time is set to the time of the other epoch but the subseconds are set to zero.
136    ///
137    /// :type other: Epoch
138    /// :rtype: Epoch
139    /// ```
140    /// use hifitime::prelude::*;
141    ///
142    /// let epoch = Epoch::from_gregorian_utc(2022, 12, 01, 10, 11, 12, 13);
143    /// let other_utc = Epoch::from_gregorian_utc(2024, 12, 01, 20, 21, 22, 23);
144    /// let other = other_utc.to_time_scale(TimeScale::TDB);
145    ///
146    /// assert_eq!(
147    ///     epoch.with_hms_strict_from(other),
148    ///     Epoch::from_gregorian_utc(2022, 12, 01, 20, 21, 22, 0)
149    /// );
150    /// ```
151    pub fn with_hms_strict_from(&self, other: Self) -> Self {
152        let (sign, days, _, _, _, _, _, _) = self.duration.decompose();
153        let other = other.to_time_scale(self.time_scale);
154        Self::from_duration(
155            Duration::compose(
156                sign,
157                days,
158                other.hours(),
159                other.minutes(),
160                other.seconds(),
161                0,
162                0,
163                0,
164            ),
165            self.time_scale,
166        )
167    }
168}
