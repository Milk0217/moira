11use crate::{Duration, Epoch, HifitimeError};
12
13/// Converts the webtime Duration into a hifitime Duration.
14///
15/// Clippy thinks these are the same type, but they aren't.
16#[allow(clippy::unnecessary_fallible_conversions)]
17pub(crate) fn duration_since_unix_epoch() -> Result<Duration, HifitimeError> {
18    // map_err maps the duration_since error into a hifitime error, if the conversion to a SystemTime fails.
19    // Then we converts a valid SystemTime into a Hifitime duration, unless it fails via and_then
20    web_time::SystemTime::now()
21        .duration_since(web_time::SystemTime::UNIX_EPOCH)
22        .map_err(|_| HifitimeError::SystemTimeError)
23        .and_then(|d| d.try_into().map_err(|_| HifitimeError::SystemTimeError))
24}
25
26// This is in its separate impl far away from the Python feature because pyO3's classmethod does not work with cfg_attr
27#[cfg(feature = "std")]
28impl Epoch {
29    /// Initializes a new Epoch from `now`.
30    /// WARNING: This assumes that the system time returns the time in UTC (which is the case on Linux)
31    /// Uses [`std::time::SystemTime::now`](https://doc.rust-lang.org/std/time/struct.SystemTime.html#method.now)
32    /// or javascript interop under the hood
33    pub fn now() -> Result<Self, HifitimeError> {
34        let duration = duration_since_unix_epoch()?;
35        Ok(Self::from_unix_duration(duration))
36    }
37}
