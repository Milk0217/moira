41/// Computes the true anomaly from the given mean anomaly for an orbit.
42///
43/// The computation process varies depending on whether the orbit is elliptical (eccentricity less than or equal to 1)
44/// or hyperbolic (eccentricity greater than 1). In each case, the method uses an iterative algorithm to find a
45/// sufficiently accurate approximation of the true anomaly.
46///
47/// # Arguments
48///
49/// * `ma_radians` - The mean anomaly in radians.
50/// * `ecc` - The eccentricity of the orbit.
51///
52/// # Remarks
53///
54/// This function uses GTDS MathSpec Equations 3-180, 3-181, and 3-186 for the iterative computation process.
55///
56/// Source: GMAT source code (`compute_mean_to_true_anomaly`)
57pub fn compute_mean_to_true_anomaly_rad(ma_radians: f64, ecc: f64) -> PhysicsResult<f64> {
58    let rm = ma_radians;
59    if ecc <= 1.0 {
60        // Elliptical orbit
61        let mut e2 = rm + ecc * rm.sin(); // GTDS MathSpec Equation 3-182
62
63        let mut iter = 0;
64
65        loop {
66            iter += 1;
67            if iter > 1000 {
68                return Err(PhysicsError::AppliedMath {
69                    source: MathError::MaxIterationsReached {
70                        iter,
71                        action: "computing the true anomaly from the mean anomaly",
72                    },
73                });
74            }
75
76            // GTDS MathSpec Equation 3-180  Note: a little difference here is that it uses Cos(E) instead of Cos(E-0.5*f)
77            let normalized_anomaly = 1.0 - ecc * e2.cos();
78
79            if normalized_anomaly.abs() < MA_EPSILON {
80                return Err(PhysicsError::AppliedMath {
81                    source: MathError::DomainError {
82                        value: normalized_anomaly,
83                        msg: "normalized anomaly too small",
84                    },
85                });
86            }
87
88            // GTDS MathSpec Equation 3-181
89            let e1 = e2 - (e2 - ecc * e2.sin() - rm) / normalized_anomaly;
90
91            if (e2 - e1).abs() < MA_EPSILON {
92                break;
93            }
94
95            e2 = e1;
96        }
97
98        let mut e = e2;
99
100        if e < 0.0 {
101            e += TAU;
102        }
103
104        let c = (e - PI).abs();
105
106        let mut ta = if c >= 1.0e-08 {
107            let normalized_anomaly = 1.0 - ecc;
108
109            if (normalized_anomaly).abs() < MA_EPSILON {
110                return Err(PhysicsError::AppliedMath {
111                    source: MathError::DomainError {
112                        value: normalized_anomaly,
113                        msg: "normalized anomaly too small",
114                    },
115                });
116            }
117
118            let eccentricity_ratio = (1.0 + ecc) / normalized_anomaly; // temp2 = (1+ecc)/(1-ecc)
119
120            if eccentricity_ratio < 0.0 {
121                return Err(PhysicsError::AppliedMath {
122                    source: MathError::DomainError {
123                        value: eccentricity_ratio,
124                        msg: "eccentricity ratio too small",
125                    },
126                });
127            }
128
129            let f = eccentricity_ratio.sqrt();
130            let g = (e / 2.0).tan();
131            // tan(TA/2) = Sqrt[(1+ecc)/(1-ecc)] * tan(E/2)
132            2.0 * (f * g).atan()
133        } else {
134            e
135        };
136
137        if ta < 0.0 {
138            ta += TAU;
139        }
140        Ok(ta)
141    } else {
142        //---------------------------------------------------------
143        // hyperbolic orbit
144        //---------------------------------------------------------
145
146        // For hyperbolic orbit, anomaly is nolonger to be an angle so we cannot use mod of 2*PI to mean anomaly.
147        // We need to keep its original value for calculation.
148        //if (rm > PI)                       // incorrect
149        //   rm = rm - TWO_PI;               // incorrect
150
151        //f2 = ecc * Sinh(rm) - rm;          // incorrect
152        //f2 = rm / 2;                       // incorrect  // GTDS MathSpec Equation 3-186
153        let mut f2: f64 = 0.0; // This is the correct initial value for hyperbolic eccentric anomaly.
154        let mut iter = 0;
155
156        loop {
157            iter += 1;
158            if iter > 1000 {
159                return Err(PhysicsError::AppliedMath {
160                    source: MathError::MaxIterationsReached {
161                        iter,
162                        action: "computing the true anomaly from the mean anomaly",
163                    },
164                });
165            }
166
167            let normalizer = ecc * f2.cosh() - 1.0;
168
169            if normalizer.abs() < MA_EPSILON {
170                return Err(PhysicsError::AppliedMath {
171                    source: MathError::DomainError {
172                        value: normalizer,
173                        msg: "normalized anomaly too small (hyperbolic case)",
174                    },
175                });
176            }
177
178            let f1 = f2 - (ecc * f2.sinh() - f2 - rm) / normalizer; // GTDS MathSpec Equation 3-186
179            if (f2 - f1).abs() < MA_EPSILON {
180                break;
181            }
182            f2 = f1;
183        }
184
185        let f = f2;
186        let normalized_anomaly = ecc - 1.0;
187
188        if normalized_anomaly.abs() < MA_EPSILON {
189            return Err(PhysicsError::AppliedMath {
190                source: MathError::DomainError {
191                    value: normalized_anomaly,
192                    msg: "normalized anomaly too small (hyperbolic case)",
193                },
194            });
195        }
196
197        let eccentricity_ratio = (ecc + 1.0) / normalized_anomaly; // temp2 = (ecc+1)/(ecc-1)
198
199        if eccentricity_ratio < 0.0 {
200            return Err(PhysicsError::AppliedMath {
201                source: MathError::DomainError {
202                    value: eccentricity_ratio,
203                    msg: "eccentricity ratio too small (hyperbolic case)",
204                },
205            });
206        }
207
208        let e = eccentricity_ratio.sqrt();
209        let g = (f / 2.0).tanh();
210        let mut ta = 2.0 * (e * g).atan(); // tan(TA/2) = Sqrt[(ecc+1)/(ecc-1)] * Tanh(F/2)    where: F is hyperbolic centric anomaly
211
212        if ta < 0.0 {
213            ta += TAU;
214        }
215        Ok(ta)
216    }
217}

219/// Computes the mean anomaly from the true anomaly.
220///
221/// # Arguments
222///
223/// * `nu_rad` - True anomaly in radians.
224/// * `ecc` - Eccentricity.
225///
226/// # Returns
227///
228/// The mean anomaly in radians (normalized to `[0, 2*PI)`), or a `MathError` if the computation fails.
229pub fn true_anomaly_to_mean_anomaly_rad(nu_rad: f64, ecc: f64) -> Result<f64, MathError> {
230    // Step 1: Calculate Eccentric Anomaly E_rad
231    // The true_anomaly_to_eccentric_anomaly_rad function already checks for ecc < 0.0
232    let e_rad = true_anomaly_to_eccentric_anomaly_rad(nu_rad, ecc)?;
233
234    // Step 2: Calculate Mean Anomaly M_rad based on eccentricity
235    let m_rad = if ecc < 1.0 {
236        // Elliptical or circular orbit: M = E - e * sin(E)
237        e_rad - ecc * e_rad.sin()
238    } else {
239        // Hyperbolic orbit: M = e * sinh(E) - E
240        ecc * e_rad.sinh() - e_rad
241    };
242
243    // Step 3: Normalize M_rad to be within the range [0, 2*PI)
244    // For hyperbolic orbits, mean anomaly is not an angle in the same way,
245    // but Kepler's equation for hyperbolic orbits (M = e*sinh(H) - H) is often used with M directly.
246    // However, the problem asks to normalize it to [0, 2*PI), which implies it's being treated
247    // as an angle for some purpose (perhaps consistency or specific conventions in this library).
248    // If this normalization is not desired for hyperbolic cases in a broader context,
249    // this part might need reconsideration for those cases.
250    // For now, strictly following the requirement.
251    let mut m_normalized_rad = m_rad % TAU;
252    if m_normalized_rad < 0.0 {
253        m_normalized_rad += TAU;
254    }
255
256    // Step 4: Return Ok(m_normalized_rad)
257    Ok(m_normalized_rad)
258}

260/// Computes the eccentric anomaly from the true anomaly.
261///
262/// # Arguments
263///
264/// * `nu_rad` - True anomaly in radians.
265/// * `ecc` - Eccentricity.
266///
267/// # Returns
268///
269/// The eccentric anomaly in radians, or a `MathError` if the computation fails.
270pub fn true_anomaly_to_eccentric_anomaly_rad(nu_rad: f64, ecc: f64) -> Result<f64, MathError> {
271    if ecc < 0.0 {
272        return Err(MathError::DomainError {
273            value: ecc,
274            msg: "eccentricity cannot be negative",
275        });
276    }
277
278    if ecc < 1.0 {
279        // Elliptical or circular orbit
280        // E = atan2( sqrt(1 - e^2) * sin(nu), e + cos(nu) )
281        let e_num = (1.0 - ecc * ecc).sqrt() * nu_rad.sin();
282        let e_den = ecc + nu_rad.cos();
283        Ok(e_num.atan2(e_den))
284    } else {
285        // Hyperbolic orbit
286        // E = 2.0 * atanh( sqrt((e - 1.0) / (e + 1.0)) * tan(nu / 2.0) )
287
288        if (ecc + 1.0).abs() < f64::EPSILON {
289            return Err(MathError::DivisionByZero {
290                action: "computing hyperbolic eccentric anomaly, (e + 1.0) is zero",
291            });
292        }
293
294        let factor_sqrt = (ecc - 1.0) / (ecc + 1.0);
295        if factor_sqrt < 0.0 {
296            // This case should ideally not happen if ecc >= 1.0,
297            // but floating point inaccuracies might lead to it if ecc is very close to 1.0
298            // or if ecc was negative (which is checked above).
299            return Err(MathError::DomainError {
300                value: factor_sqrt,
301                msg: "argument for sqrt in hyperbolic case is negative",
302            });
303        }
304
305        let tan_nu_half = (nu_rad / 2.0).tan();
306
307        // The argument to atanh must be in (-1, 1).
308        // If tan_nu_half is very large (nu_rad/2.0 approaches PI/2),
309        // the argument might exceed this range.
310        // sqrt((e-1)/(e+1)) is always < 1 for e >= 1.
311        // So, the product can only be >= 1 or <= -1 if tan_nu_half is large enough.
312
313        let atanh_arg = factor_sqrt.sqrt() * tan_nu_half;
314
315        if atanh_arg >= 1.0 || atanh_arg <= -1.0 {
316            // Check for edge cases like nu_rad = PI for ecc = 1.0 (parabolic)
317            // For parabolic orbits (ecc = 1.0), the term sqrt((e-1)/(e+1)) becomes 0.
318            // So atanh_arg would be 0 unless tan_nu_half is infinite.
319            // If ecc is exactly 1.0, factor_sqrt is 0. So atanh_arg is 0 unless tan_nu_half is Inf/NaN.
320            // If nu_rad makes tan(nu_rad/2.0) infinite (e.g. nu_rad = PI),
321            // and ecc > 1, then atanh_arg would be Inf * a_number_lt_1 = Inf.
322            // The original problem states: "if nu_rad / 2.0 makes tan very large or if ecc is exactly 1.0.
323            // Specifically, if ecc == 1.0, this formula is problematic due to division by zero in the sqrt
324            // and atanh of infinity."
325            // The (e+1.0) division by zero is handled.
326            // If ecc == 1.0, then factor_sqrt is 0.0. atanh_arg is 0.0 * tan(nu_rad/2.0).
327            // If tan(nu_rad/2.0) is infinite (nu_rad = PI), then 0.0 * Inf = NaN.
328            // f64::atanh(NaN) is NaN.
329            // If atanh_arg is NaN, it will not satisfy >= 1.0 or <= -1.0.
330            // Let's test for NaN explicitly as well.
331            if atanh_arg.is_nan() {
332                return Err(MathError::DomainError {
333                    value: atanh_arg,
334                    msg: "atanh argument is NaN in hyperbolic eccentric anomaly calculation",
335                });
336            }
337            // If it's not NaN but out of [-1, 1] range
338            return Err(MathError::DomainError {
339                value: atanh_arg,
340                msg: "atanh argument out of domain (-1, 1) in hyperbolic eccentric anomaly calculation",
341            });
342        }
343
344        Ok(2.0 * atanh_arg.atanh())
345    }

527    #[test]
528    fn test_ta_to_ma_normalization() {
529        // 5. Normalization
530        let ecc = 0.1;
531        let nu = 1.8 * PI; // equivalent to -0.2 * PI
532
533        // For nu = 1.8*PI (324 deg), cos(nu) = cos(-36 deg) = 0.80901699
534        // sin(nu) = sin(-36 deg) = -0.58778525
535        // E = atan2(sqrt(1-0.01)*sin(nu), 0.1+cos(nu)) = atan2(sqrt(0.99)*(-0.58778525), 0.1+0.80901699)
536        // E = atan2(0.994987437*(-0.58778525), 0.90901699) = atan2(-0.58483208, 0.90901699)
