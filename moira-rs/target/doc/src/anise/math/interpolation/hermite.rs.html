65/// From the abscissas (xs), the ordinates (ys), and the first derivatives (ydots), build the Hermite interpolation of the function and evaluate it at the requested abscissa (x).
66///
67/// # Runtime verifications
68/// 1. Ensure that all provided arrays are of the same size.
69/// 2. Ensure that there are no more than 32 items to interpolate.
70/// 3. Ensure no division by zero errors (zero is set to f64::EPSILON, which is about 2e-16).
71pub fn hermite_eval(
72    xs: &[f64],
73    ys: &[f64],
74    ydots: &[f64],
75    x_eval: f64,
76) -> Result<(f64, f64), InterpolationError> {
77    if xs.len() != ys.len() || xs.len() != ydots.len() {
78        return Err(InterpolationError::CorruptedData {
79            what: "lengths of abscissas (xs), ordinates (ys), and first derivatives (ydots) differ",
80        });
81    } else if xs.is_empty() {
82        return Err(InterpolationError::CorruptedData {
83            what: "list of abscissas (xs) is empty",
84        });
85    } else if xs.len() > MAX_SAMPLES {
86        return Err(InterpolationError::CorruptedData {
87            what: "list of abscissas (xs) contains more items than MAX_SAMPLES (32)",
88        });
89    }
90
91    // At this point, we know that the lengths of items is correct, so we can directly address them without worry for overflowing the array.
92
93    let work: &mut [f64] = &mut [0.0; 4 * MAX_SAMPLES];
94    let n: usize = xs.len();
95
96    /*  Copy the input array into WORK.  After this, the first column */
97    /*  of WORK represents the first column of our triangular */
98    /*  interpolation table. */
99
100    for i in 0..n {
101        work[2 * i] = ys[i];
102        work[2 * i + 1] = ydots[i];
103    }
104
105    /*  Compute the second column of the interpolation table: this */
106    /*  consists of the N-1 values obtained by evaluating the */
107    /*  first-degree interpolants at X. We'll also evaluate the */
108    /*  derivatives of these interpolants at X and save the results in */
109    /*  the second column of WORK. Because the derivative computations */
110    /*  depend on the function computations from the previous column in */
111    /*  the interpolation table, and because the function interpolation */
112    /*  overwrites the previous column of interpolated function values, */
113    /*  we must evaluate the derivatives first. */
114
115    for i in 1..=n - 1 {
116        let c1 = xs[i] - x_eval;
117        let c2 = x_eval - xs[i - 1];
118        let denom = xs[i] - xs[i - 1];
119        if denom.abs() < f64::EPSILON {
120            return Err(InterpolationError::InterpMath {
121                source: MathError::DivisionByZero {
122                    action:
123                        "hermite data contains likely duplicate abcissa, remove duplicate states",
124                },
125            });
126        }
127
128        /*  The second column of WORK contains interpolated derivative */
129        /*  values. */
130
131        /*  The odd-indexed interpolated derivatives are simply the input */
132        /*  derivatives. */
133
134        let prev = 2 * i - 1;
135        let curr = 2 * i;
136        work[prev + 2 * n - 1] = work[prev];
137
138        /*  The even-indexed interpolated derivatives are the slopes of */
139        /*  the linear interpolating polynomials for adjacent input */
140        /*  abscissa/ordinate pairs. */
141
142        work[prev + 2 * n] = (work[curr] - work[prev - 1]) / denom;
143
144        /*  The first column of WORK contains interpolated function values. */
145        /*  The odd-indexed entries are the linear Taylor polynomials, */
146        /*  for each input abscissa value, evaluated at X. */
147
148        let temp = work[prev] * (x_eval - xs[i - 1]) + work[prev - 1];
149        work[prev] = (c1 * work[prev - 1] + c2 * work[curr]) / denom;
150        work[prev - 1] = temp;
151    }
152
153    /*  The last column entries were not computed by the preceding loop; */
154    /*  compute them now. */
155
156    work[4 * n - 2] = work[(2 * n) - 1];
157    work[2 * (n - 1)] += work[(2 * n) - 1] * (x_eval - xs[n - 1]);
158
159    /*  Compute columns 3 through 2*N of the table. */
160
161    for j in 2..=(2 * n) - 1 {
162        for i in 1..=(2 * n) - j {
163            /*  In the theoretical construction of the interpolation table, */
164            /*  there are 2*N abscissa values, since each input abcissa */
165            /*  value occurs with multiplicity two. In this theoretical */
166            /*  construction, the Jth column of the interpolation table */
167            /*  contains results of evaluating interpolants that span J+1 */
168            /*  consecutive abscissa values.  The indices XI and XIJ below */
169            /*  are used to pick the correct abscissa values out of the */
170            /*  physical XVALS array, in which the abscissa values are not */
171            /*  repeated. */
172
173            let xi = i.div_ceil(2);
174            let xij = (i + j).div_ceil(2);
175            let c1 = xs[xij - 1] - x_eval;
176            let c2 = x_eval - xs[xi - 1];
177            let denom = xs[xij - 1] - xs[xi - 1];
178            if denom.abs() < f64::EPSILON {
179                return Err(InterpolationError::InterpMath {
180                    source: MathError::DivisionByZero {
181                        action: "hermite data contains duplicate states",
182                    },
183                });
184            }
185
186            /*  Compute the interpolated derivative at X for the Ith */
187            /*  interpolant. This is the derivative with respect to X of */
188            /*  the expression for the interpolated function value, which */
189            /*  is the second expression below. This derivative computation */
190            /*  is done first because it relies on the interpolated */
191            /*  function values from the previous column of the */
192            /*  interpolation table. */
193
194            /*  The derivative expression here corresponds to equation */
195            /*  2.35 on page 64 in reference [2]. */
196
197            work[i + 2 * n - 1] =
198                (c1 * work[i + 2 * n - 1] + c2 * work[i + 2 * n] + (work[i] - work[i - 1])) / denom;
199
200            /*  Compute the interpolated function value at X for the Ith */
201            /*  interpolant. */
202
203            work[i - 1] = (c1 * work[i - 1] + c2 * work[i]) / denom;
204        }
205    }
206
207    /*  Our interpolated function value is sitting in WORK(1,1) at this */
208    /*  point. The interpolated derivative is located in WORK(1,2). */
209
210    let f = work[0];
211    let df = work[2 * n];
212    Ok((f, df))
213}

215#[test]
216fn hermite_spice_docs_example() {
217    let ts = [-1.0, 0.0, 3.0, 5.0];
218    let yvals = [6.0, 5.0, 2210.0, 78180.0];
219    let ydotvals = [3.0, 0.0, 5115.0, 109395.0];
220
221    // Check that we can interpolate the values exactly.
222    for (i, t) in ts.iter().enumerate() {
223        let (eval, deriv) = hermite_eval(&ts, &yvals, &ydotvals, *t).unwrap();
224        let eval_err = (eval - yvals[i]).abs();
225        assert!(eval_err < f64::EPSILON, "f(x) error is {eval_err:e}");
226
227        let deriv_err = (deriv - ydotvals[i]).abs();
228        assert!(deriv_err < f64::EPSILON, "f'(x) error is {deriv_err:e}");
229    }
230
231    // Check the interpolation from the SPICE documentation
232    let (x, vx) = hermite_eval(&ts, &yvals, &ydotvals, 2.0).unwrap();
233
234    assert!((x - 141.0).abs() < f64::EPSILON, "X error");
235    assert!((vx - 456.0).abs() < f64::EPSILON, "VX error");
236}
