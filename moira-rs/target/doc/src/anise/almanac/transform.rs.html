28impl Almanac {
29    /// Returns the Cartesian state needed to transform the `target_frame` to the `observer_frame`.
30    ///
31    /// # SPICE Compatibility
32    /// This function is the SPICE equivalent of spkezr: `spkezr(TARGET_ID, EPOCH_TDB_S, ORIENTATION_ID, ABERRATION, OBSERVER_ID)`
33    /// In ANISE, the TARGET_ID and ORIENTATION are provided in the first argument (TARGET_FRAME), as that frame includes BOTH
34    /// the target ID and the orientation of that target. The EPOCH_TDB_S is the epoch in the TDB time system, which is computed
35    /// in ANISE using Hifitime. THe ABERRATION is computed by providing the optional Aberration flag. Finally, the OBSERVER
36    /// argument is replaced by OBSERVER_FRAME: if the OBSERVER_FRAME argument has the same orientation as the TARGET_FRAME, then this call
37    /// will return exactly the same data as the spkerz SPICE call.
38    ///
39    /// # Note
40    /// The units will be those of the underlying ephemeris data (typically km and km/s)
41    pub fn transform(
42        &self,
43        target_frame: Frame,
44        observer_frame: Frame,
45        epoch: Epoch,
46        ab_corr: Option<Aberration>,
47    ) -> AlmanacResult<CartesianState> {
48        // Translate
49        let state = self
50            .translate(target_frame, observer_frame, epoch, ab_corr)
51            .context(EphemerisSnafu {
52                action: "transform from/to",
53            })?;
54        // Rotate
55        let dcm = self
56            .rotate(target_frame, observer_frame, epoch)
57            .context(OrientationSnafu {
58                action: "transform from/to",
59            })?;
60
61        (dcm * state)
62            .context(OrientationPhysicsSnafu {})
63            .context(OrientationSnafu {
64                action: "transform from/to",
65            })
66    }

68    /// Returns the provided state as seen from the observer frame, given the aberration.
69    pub fn transform_to(
70        &self,
71        mut state: CartesianState,
72        observer_frame: Frame,
73        ab_corr: Option<Aberration>,
74    ) -> AlmanacResult<CartesianState> {
75        // If the input and final rotations differ, rotate into J2000 first
76        state = if state.frame.orient_origin_match(observer_frame) {
77            state
78        } else {
79            self.rotate_to(state, state.frame.with_orient(J2000))
80                .context(OrientationSnafu {
81                    action: "transform state dcm",
82                })?
83        };
84
85        // Transform in the base frame (J2000) or the common frame
86        state = self
87            .translate_to(state, observer_frame, ab_corr)
88            .context(EphemerisSnafu {
89                action: "transform state",
90            })?;
91
92        // Rotate into the observer frame
93        self.rotate_to(state, observer_frame)
94            .context(OrientationSnafu {
95                action: "transform state",
96            })
97    }

99    /// Returns the Cartesian state of the object as seen from the provided observer frame (essentially `spkezr`).
100    ///
101    /// # Note
102    /// The units will be those of the underlying ephemeris data (typically km and km/s)
103    pub fn state_of(
104        &self,
105        object: NaifId,
106        observer: Frame,
107        epoch: Epoch,
108        ab_corr: Option<Aberration>,
109    ) -> AlmanacResult<CartesianState> {
110        self.transform(Frame::from_ephem_j2000(object), observer, epoch, ab_corr)
111    }

113    /// Alias fo SPICE's `spkezr` where the inputs must be the NAIF IDs of the objects and frames with the caveat that the aberration is moved to the last positional argument.
114    pub fn spk_ezr(
115        &self,
116        target: NaifId,
117        epoch: Epoch,
118        frame: NaifId,
119        observer: NaifId,
120        ab_corr: Option<Aberration>,
121    ) -> AlmanacResult<CartesianState> {
122        let tgt_j2000 = Frame::from_ephem_j2000(target);
123        let obs_j2000 = Frame::from_ephem_j2000(observer);
124
125        // Translate in J2000
126        let state = self
127            .translate(tgt_j2000, obs_j2000, epoch, ab_corr)
128            .context(EphemerisSnafu {
129                action: "transform from/to",
130            })?;
131
132        // Rotate into the desired frame
133        self.rotate_to(state, Frame::new(observer, frame))
134            .context(OrientationSnafu {
135                action: "spkerz from/to",
136            })
137    }

139    /// Translates a state with its origin (`to_frame`) and given its units (distance_unit, time_unit), returns that state with respect to the requested frame
140    ///
141    /// **WARNING:** This function only performs the translation and no rotation _whatsoever_. Use the `transform_state_to` function instead to include rotations.
142    #[allow(clippy::too_many_arguments)]
143    pub fn transform_state_to(
144        &self,
145        position: Vector3,
146        velocity: Vector3,
147        from_frame: Frame,
148        to_frame: Frame,
149        epoch: Epoch,
150        ab_corr: Option<Aberration>,
151        distance_unit: LengthUnit,
152        time_unit: TimeUnit,
153    ) -> AlmanacResult<CartesianState> {
154        let state = self
155            .translate_state_to(
156                position,
157                velocity,
158                from_frame,
159                to_frame,
160                epoch,
161                ab_corr,
162                distance_unit,
163                time_unit,
164            )
165            .context(EphemerisSnafu {
166                action: "transform provided state",
167            })?;
168
169        // Compute the frame rotation
170        let dcm = self
171            .rotate(from_frame, to_frame, epoch)
172            .context(OrientationSnafu {
173                action: "transform provided state dcm",
174            })?;
175
176        (dcm * state)
177            .context(OrientationPhysicsSnafu {})
178            .context(OrientationSnafu {
179                action: "transform provided state",
180            })
181    }

183    /// Returns the unitary 3D vector between two [Frame]s (solid bodies) at desired [Epoch]
184    pub fn unit_vector(
185        &self,
186        target_frame: Frame,
187        observer_frame: Frame,
188        epoch: Epoch,
189        ab_corr: Option<Aberration>,
190    ) -> AlmanacResult<Vector3> {
191        let state = self.transform(target_frame, observer_frame, epoch, ab_corr)?;
192        Ok(state.radius_km / state.rmag_km())
193    }

195    /// Returns the unitary 3D vector between desired [Frame] (solid body) and the Sun at desired [Epoch]
196    pub fn sun_unit_vector(
197        &self,
198        epoch: Epoch,
199        observer_frame: Frame,
200        ab_corr: Option<Aberration>,
201    ) -> AlmanacResult<Vector3> {
202        self.unit_vector(SUN_J2000, observer_frame, epoch, ab_corr)
