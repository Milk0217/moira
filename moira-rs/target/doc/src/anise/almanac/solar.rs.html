24impl Almanac {
25    /// Returns the angle (between 0 and 180 degrees) between the observer and the Sun, and the observer and the target body ID.
26    /// This computes the Sun Probe Earth angle (SPE) if the probe is in a loaded SPK, its ID is the "observer_id", and the target is set to its central body.
27    ///
28    /// # Geometry
29    /// If the SPE is greater than 90 degrees, then the celestial object below the probe is in sunlight.
30    ///
31    /// ## Sunrise at nadir
32    /// ```text
33    /// Sun
34    ///  |  \      
35    ///  |   \
36    ///  |    \
37    ///  Obs. -- Target
38    /// ```
39    /// ## Sun high at nadir
40    /// ```text
41    /// Sun
42    ///  \        
43    ///   \  __ θ > 90
44    ///    \     \
45    ///     Obs. ---------- Target
46    /// ```
47    ///
48    /// ## Sunset at nadir
49    /// ```text
50    ///          Sun
51    ///        /  
52    ///       /  __ θ < 90
53    ///      /    /
54    ///  Obs. -- Target
55    /// ```
56    ///
57    /// # Algorithm
58    /// 1. Compute the position of the Sun as seen from the observer
59    /// 2. Compute the position of the target as seen from the observer
60    /// 3. Return the arccosine of the dot product of the norms of these vectors.
61    ///
62    /// :type target_id: int
63    /// :type observer_id: int
64    /// :type epoch: Epoch
65    /// :type ab_corr: Aberration
66    /// :rtype: float
67    pub fn sun_angle_deg(
68        &self,
69        target_id: NaifId,
70        observer_id: NaifId,
71        epoch: Epoch,
72        ab_corr: Option<Aberration>,
73    ) -> Result<f64, EphemerisError> {
74        let obs_to_sun = self.translate(
75            SUN_J2000,
76            Frame::from_ephem_j2000(observer_id),
77            epoch,
78            ab_corr,
79        )?;
80        let obs_to_target = self.translate(
81            Frame::from_ephem_j2000(target_id),
82            Frame::from_ephem_j2000(observer_id),
83            epoch,
84            ab_corr,
85        )?;
86
87        Ok(obs_to_sun
88            .r_hat()
89            .dot(&obs_to_target.r_hat())
90            .acos()
91            .to_degrees())
92    }
93
94    /// Convenience function that calls `sun_angle_deg` with the provided frames instead of the ephemeris ID.
95    ///
96    /// :type target: Frame
97    /// :type observer: Frame
98    /// :type epoch: Epoch
99    /// :type ab_corr: Aberration
100    /// :rtype: float
101    pub fn sun_angle_deg_from_frame(
102        &self,
103        target: Frame,
104        observer: Frame,
105        epoch: Epoch,
106        ab_corr: Option<Aberration>,
107    ) -> Result<f64, EphemerisError> {
108        self.sun_angle_deg(target.ephemeris_id, observer.ephemeris_id, epoch, ab_corr)
109    }
110}

112#[cfg(test)]
113mod ut_solar {
114    use crate::{
115        constants::{
116            celestial_objects::EARTH,
117            frames::{EARTH_J2000, IAU_EARTH_FRAME, SUN_J2000},
118            usual_planetary_constants::MEAN_EARTH_ANGULAR_VELOCITY_DEG_S,
119        },
120        prelude::*,
121    };
122
123    /// Load a BSP of a spacecraft trajectory, compute the sun elevation at different points on the surface below that spacecraft
124    /// and ensure that it matches with the SPE calculation.
125    #[test]
126    fn verify_geometry() {
127        let ctx = Almanac::default()
128            .load("../data/de440s.bsp")
129            .and_then(|ctx| ctx.load("../data/gmat-hermite.bsp"))
130            .and_then(|ctx| ctx.load("../data/pck11.pca"))
131            .unwrap();
132
133        let epoch = Epoch::from_gregorian_hms(2000, 1, 1, 12, 0, 0, TimeScale::UTC);
134
135        let sc_id = -10000001;
136
137        let my_sc_j2k = Frame::from_ephem_j2000(sc_id);
138
139        // Grab the state in the J2000 frame
140        let state = ctx.transform(my_sc_j2k, EARTH_J2000, epoch, None).unwrap();
141
142        // We'll check at four different points in the orbit
143        for epoch in TimeSeries::inclusive(
144            epoch,
145            epoch + state.period().unwrap(),
146            0.05 * state.period().unwrap(),
147        ) {
148            let spe_deg = ctx
149                .sun_angle_deg(EARTH, sc_id, epoch, Aberration::NONE)
150                .unwrap();
151            assert_eq!(
152                spe_deg,
153                ctx.sun_angle_deg_from_frame(EARTH_J2000, my_sc_j2k, epoch, Aberration::NONE)
154                    .unwrap()
155            );
156
157            let iau_earth = ctx.frame_info(IAU_EARTH_FRAME).unwrap();
158
159            // Compute this state in the body fixed frame.
160            let state_bf = ctx
161                .transform(my_sc_j2k, IAU_EARTH_FRAME, epoch, None)
162                .unwrap();
163            // Build a local point on the surface below this spacecraft
164            let nadir_surface_point = Orbit::try_latlongalt(
165                state_bf.latitude_deg().unwrap(),
166                state_bf.longitude_deg(),
167                0.0,
168                MEAN_EARTH_ANGULAR_VELOCITY_DEG_S,
169                epoch,
170                iau_earth,
171            )
172            .unwrap();
173            // Fetch the state of the sun at this time.
174            let sun_state = ctx
175                .transform(SUN_J2000, IAU_EARTH_FRAME, epoch, None)
176                .unwrap();
177
178            // Compute the Sun elevation from that point.
179            let sun_elevation_deg = ctx
180                .azimuth_elevation_range_sez(sun_state, nadir_surface_point, None, None)
181                .unwrap()
182                .elevation_deg;
183
184            println!(
185                "{epoch}\tsun el = {sun_elevation_deg:.3} deg\tlat = {:.3} deg\tlong = {:.3} deg\t-->\tSPE = {spe_deg:.3} deg",
186                nadir_surface_point.latitude_deg().unwrap(),
187                nadir_surface_point.longitude_deg()
188            );
189
190            // Test: the sun elevation from the ground should be about 90 degrees _less_ than the SPE
191            // The "about" is because the SPE effectively assumes a spherical celestial object, but the frame
192            // from the Almanac accounts for the flattening when computing the exact location below the vehicle.
193            // Hence, there is a small difference (we accept up to 0.05 degrees of difference).
194            assert!((sun_elevation_deg + 90.0 - spe_deg).abs() < 5e-2)
195        }
196    }
