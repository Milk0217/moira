59/// An Almanac contains all of the loaded SPICE and ANISE data. It is the context for all computations.
60///
61/// :type path: str
62/// :rtype: Almanac
63#[derive(Clone, Default)]
64#[cfg_attr(feature = "python", pyclass)]
65#[cfg_attr(feature = "python", pyo3(module = "anise"))]
66pub struct Almanac {
67    /// NAIF SPK is kept unchanged
68    pub spk_data: IndexMap<String, SPK>,
69    /// NAIF BPC is kept unchanged
70    pub bpc_data: IndexMap<String, BPC>,
71    /// Dataset of planetary data
72    pub planetary_data: PlanetaryDataSet,
73    /// Dataset of spacecraft data
74    pub spacecraft_data: SpacecraftDataSet,
75    /// Dataset of euler parameters
76    pub euler_param_data: EulerParameterDataSet,
77    /// Dataset of locations
78    pub location_data: LocationDataSet,
79}
80
81impl fmt::Display for Almanac {
82    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> Result<(), fmt::Error> {
83        write!(
84            f,
85            "Almanac: #SPK = {}\t#BPC = {}",
86            self.num_loaded_spk(),
87            self.num_loaded_bpc()
88        )?;
89        if !self.planetary_data.lut.by_id.is_empty() {
90            write!(f, "\t{}", self.planetary_data)?;
91        }
92        if !self.spacecraft_data.lut.by_id.is_empty() {
93            write!(f, "\t{}", self.spacecraft_data)?;
94        }
95        if !self.euler_param_data.lut.by_id.is_empty() {
96            write!(f, "\t{}", self.euler_param_data)?;
97        }
98        Ok(())
99    }
100}

126    /// Loads the provides bytes as one of the data types supported in ANISE.
127    pub fn load_from_bytes(self, bytes: Bytes) -> AlmanacResult<Self> {
128        self._load_from_bytes(bytes, None)
129    }
130
131    fn _load_from_bytes(self, bytes: Bytes, path: Option<&str>) -> AlmanacResult<Self> {
132        // Check if they forgot to run git lfs
133        if let Some(lfs_header) = bytes.get(..8) {
134            if lfs_header == "version".as_bytes() {
135                return Err(AlmanacError::GenericError {
136                    err: "file is a git lfs pointer, run `git lfs pull`".to_string(),
137                });
138            }
139        }
140
141        // Load the header only
142        if let Some(file_record_bytes) = bytes.get(..FileRecord::SIZE) {
143            let file_record = FileRecord::read_from_bytes(file_record_bytes).unwrap();
144            if let Ok(fileid) = file_record.identification() {
145                return match fileid {
146                    "PCK" => {
147                        info!("Loading {} as DAF/PCK", path.unwrap_or("bytes"));
148                        let bpc = BPC::parse(bytes)
149                            .context(BPCSnafu {
150                                action: "parsing bytes",
151                            })
152                            .context(OrientationSnafu {
153                                action: "from generic loading",
154                            })?;
155                        Ok(self
156                            .with_bpc_as(bpc, path.map_or_else(|| None, |p| Some(p.to_string()))))
157                    }
158                    "SPK" => {
159                        info!("Loading {} as DAF/SPK", path.unwrap_or("bytes"));
160                        let spk = SPK::parse(bytes)
161                            .context(SPKSnafu {
162                                action: "parsing bytes",
163                            })
164                            .context(EphemerisSnafu {
165                                action: "from generic loading",
166                            })?;
167                        Ok(self
168                            .with_spk_as(spk, path.map_or_else(|| None, |p| Some(p.to_string()))))
169                    }
170                    fileid => Err(AlmanacError::GenericError {
171                        err: format!("DAF/{fileid} is not yet supported"),
172                    }),
173                };
174            }
175            // Fall through to try to load as an ANISE file
176        }
177
178        if let Ok(metadata) = Metadata::decode_header(&bytes) {
179            // Use `try_from` to validate the dataset type
180            let dataset_type =
181                DataSetType::try_from(metadata.dataset_type as u8).map_err(|err| {
182                    AlmanacError::GenericError {
183                        err: format!("Invalid dataset type: {err}"),
184                    }
185                })?;
186
187            // Now, we can load this depending on the kind of data that it is
188            match dataset_type {
189                DataSetType::NotApplicable => {
190                    // Not something that can be decoded
191                    Err(AlmanacError::GenericError {
192                        err: format!("Malformed dataset type in {}", path.unwrap_or("bytes")),
193                    })
194                }
195                DataSetType::SpacecraftData => {
196                    // Decode as spacecraft data
197                    let dataset = SpacecraftDataSet::try_from_bytes(bytes).context({
198                        TLDataSetSnafu {
199                            action: "loading as spacecraft data",
200                        }
201                    })?;
202                    info!(
203                        "Loading {} as ANISE spacecraft data",
204                        path.unwrap_or("bytes")
205                    );
206                    Ok(self.with_spacecraft_data(dataset))
207                }
208                DataSetType::PlanetaryData => {
209                    // Decode as planetary data
210                    let dataset = PlanetaryDataSet::try_from_bytes(bytes).context({
211                        TLDataSetSnafu {
212                            action: "loading as planetary data",
213                        }
214                    })?;
215                    info!("Loading {} as ANISE/PCA", path.unwrap_or("bytes"));
216                    Ok(self.with_planetary_data(dataset))
217                }
218                DataSetType::EulerParameterData => {
219                    // Decode as euler parameter data
220                    let dataset = EulerParameterDataSet::try_from_bytes(bytes).context({
221                        TLDataSetSnafu {
222                            action: "loading Euler parameters",
223                        }
224                    })?;
225                    info!("Loading {} as ANISE/EPA", path.unwrap_or("bytes"));
226                    Ok(self.with_euler_parameters(dataset))
227                }
228                DataSetType::LocationData => {
229                    let dataset = LocationDataSet::try_from_bytes(bytes).context({
230                        TLDataSetSnafu {
231                            action: "loading location data",
232                        }
233                    })?;
234                    info!("Loading {} as ANISE/LDA", path.unwrap_or("bytes"));
235                    Ok(self.with_location_data(dataset))
236                }
237            }
238        } else {
239            Err(AlmanacError::GenericError {
240                err: "file cannot be inspected or loaded directly in ANISE".to_string(),
241            })
242        }
243    }
244
245    /// Generic function that tries to load the provided path guessing to the file type.
246    pub fn load(self, path: &str) -> AlmanacResult<Self> {
247        // Load the data onto the heap
248        let bytes = file2heap!(path).context(LoadingSnafu {
249            path: path.to_string(),
250        })?;
251
252        self._load_from_bytes(bytes, Some(path))
253            .map_err(|e| match e {
254                AlmanacError::GenericError { err } => {
255                    // Add the path to the error
256                    AlmanacError::GenericError {
257                        err: format!("with {path}: {err}"),
258                    }
259                }
260                _ => e,
261            })
262    }

264    /// Pretty prints the description of this Almanac, showing everything by default. Default time scale is TDB.
265    /// If any parameter is set to true, then nothing other than that will be printed.
266    #[allow(clippy::too_many_arguments)]
267    pub fn describe(
268        &self,
269        spk: Option<bool>,
270        bpc: Option<bool>,
271        planetary: Option<bool>,
272        eulerparams: Option<bool>,
273        locations: Option<bool>,
274        time_scale: Option<TimeScale>,
275        round_time: Option<bool>,
276    ) {
277        let print_any = spk.unwrap_or(false)
278            || bpc.unwrap_or(false)
279            || planetary.unwrap_or(false)
280            || eulerparams.unwrap_or(false)
281            || locations.unwrap_or(false);
282
283        if spk.unwrap_or(!print_any) {
284            for (spk_no, (alias, spk)) in self.spk_data.iter().rev().enumerate() {
285                println!(
286                    "=== SPK #{spk_no}: `{alias}` ===\n{}",
287                    spk.describe_in(time_scale.unwrap_or(TimeScale::TDB), round_time)
288                );
289            }
290        }
291
292        if bpc.unwrap_or(!print_any) {
293            for (bpc_no, (alias, bpc)) in self.bpc_data.iter().rev().enumerate() {
294                println!(
295                    "=== BPC #{bpc_no}: `{alias}` ===\n{}",
296                    bpc.describe_in(time_scale.unwrap_or(TimeScale::TDB), round_time)
297                );
298            }
299        }
300
301        if planetary.unwrap_or(!print_any) {
302            println!("=== PLANETARY DATA ==\n{}", self.planetary_data.describe());
303        }
304
305        if eulerparams.unwrap_or(!print_any) {
306            println!(
307                "=== EULER PARAMETER DATA ==\n{}",
308                self.euler_param_data.describe()
309            );
310        }
311
312        if locations.unwrap_or(!print_any) {
313            println!("=== LOCATIONS DATA ==\n{}", self.location_data.describe());
314        }
315    }
