24/// A structure to set up an Almanac, with automatic downloading, local storage, checksum checking, and more.
25///
26/// # Behavior
27/// If the URI is a local path, relative or absolute, nothing will be fetched from a remote. Relative paths are relative to the execution folder (i.e. the current working directory).
28/// If the URI is a remote path, the MetaAlmanac will first check if the file exists locally. If it exists, it will check that the CRC32 checksum of this file matches that of the specs.
29/// If it does not match, the file will be downloaded again. If no CRC32 is provided but the file exists, then the MetaAlmanac will fetch the remote file and overwrite the existing file.
30/// The downloaded path will be stored in the "AppData" folder.
31///
32/// :type maybe_path: str, optional
33/// :rtype: MetaAlmanac
34#[derive(Clone, Debug, Serialize, Deserialize, PartialEq)]
35#[cfg_attr(feature = "python", pyclass)]
36#[cfg_attr(feature = "python", pyo3(module = "anise"))]
37pub struct MetaAlmanac {
38    pub files: Vec<MetaFile>,
39}
40
41impl MetaAlmanac {
42    /// Loads the provided path as a Dhall configuration file and processes each file.
43    pub fn new(path: &str) -> Result<Self, MetaAlmanacError> {
44        match serde_dhall::from_file(path).parse::<Self>() {
45            Err(e) => Err(MetaAlmanacError::ParseDhall {
46                path: path.to_string(),
47                err: format!("{e}"),
48            }),
49            Ok(me) => Ok(me),
50        }
51    }
52
53    /// Fetch all of the URIs and return a loaded Almanac
54    /// When downloading the data, ANISE will create a temporarily lock file to prevent race conditions
55    /// where multiple processes download the data at the same time. Set `autodelete` to true to delete
56    /// this lock file if a dead lock is detected after 10 seconds. Set this flag to false if you have
57    /// more than ten processes which may attempt to download files in parallel.
58    pub fn process(&mut self, autodelete: bool) -> AlmanacResult<Almanac> {
59        for (fno, file) in self.files.iter_mut().enumerate() {
60            file.process(autodelete).context(MetaSnafu {
61                fno,
62                file: file.clone(),
63            })?;
64        }
65        // At this stage, all of the files are local files, so we can load them as is.
66        let mut ctx = Almanac::default();
67        for uri in &self.files {
68            ctx = ctx.load(&uri.uri)?;
69        }
70        Ok(ctx)
71    }
72
73    /// Returns an Almanac loaded from the latest NAIF data via the `default` MetaAlmanac.
74    /// The MetaAlmanac will download the DE440s.bsp file, the PCK0008.PCA, the full Moon Principal Axis BPC (moon_pa_de440_200625) and the latest high precision Earth kernel from JPL.
75    ///
76    /// # File list
77    /// - <http://public-data.nyxspace.com/anise/de440s.bsp>
78    /// - <http://public-data.nyxspace.com/anise/v0.5/pck11.pca>
79    /// - <http://public-data.nyxspace.com/anise/moon_pa_de440_200625.bpc>
80    /// - <https://naif.jpl.nasa.gov/pub/naif/generic_kernels/pck/earth_latest_high_prec.bpc>
81    ///
82    /// # Reproducibility
83    ///
84    /// Note that the `earth_latest_high_prec.bpc` file is regularly updated daily (or so). As such,
85    /// if queried at some future time, the Earth rotation parameters may have changed between two queries.
86    pub fn latest() -> AlmanacResult<Almanac> {
87        Self::default().process(true)
88    }
89}

91impl FromStr for MetaAlmanac {
92    type Err = MetaAlmanacError;
93
94    fn from_str(s: &str) -> Result<Self, Self::Err> {
95        match serde_dhall::from_str(s).parse::<Self>() {
96            Err(e) => Err(MetaAlmanacError::ParseDhall {
97                path: s.to_string(),
98                err: format!("{e}"),
99            }),
100            Ok(me) => Ok(me),
101        }
102    }
103}

105// Methods shared between Rust and Python
106#[cfg_attr(feature = "python", pymethods)]
107#[allow(deprecated_in_future)]
108impl MetaAlmanac {
109    /// Dumps the configured Meta Almanac into a Dhall string.
110    ///
111    /// :rtype: str
112    pub fn dumps(&self) -> Result<String, MetaAlmanacError> {
113        // Define the Dhall type
114        let dhall_type: SimpleType =
115            serde_dhall::from_str("{ files : List { uri : Text, crc32 : Optional Natural } }")
116                .parse()
117                .unwrap();
118
119        serde_dhall::serialize(&self)
120            .type_annotation(&dhall_type)
121            .to_string()
122            .map_err(|e| MetaAlmanacError::ExportDhall {
123                err: format!("{e}"),
124            })
125    }
126}

128/// By default, the MetaAlmanac will download the DE440s.bsp file, the PCK0008.PCA, the full Moon Principal Axis BPC (moon_pa_de440_200625) and the latest high precision Earth kernel from JPL.
129///
130/// # File list
131/// - <http://public-data.nyxspace.com/anise/de440s.bsp>
132/// - <http://public-data.nyxspace.com/anise/v0.5/pck11.pca>
133/// - <http://public-data.nyxspace.com/anise/v0.5/moon_fk_de440.epa>
134/// - <http://public-data.nyxspace.com/anise/moon_pa_de440_200625.bpc>
135/// - <https://naif.jpl.nasa.gov/pub/naif/generic_kernels/pck/earth_latest_high_prec.bpc>
136///
137/// # Reproducibility
138///
139/// Note that the `earth_latest_high_prec.bpc` file is regularly updated daily (or so). As such,
140/// if queried at some future time, the Earth rotation parameters may have changed between two queries.
141///
142impl Default for MetaAlmanac {
143    fn default() -> Self {
144        let nyx_cloud_stor = Url::parse("http://public-data.nyxspace.com/anise/").unwrap();
145        let jpl_cloud_stor =
146            Url::parse("https://naif.jpl.nasa.gov/pub/naif/generic_kernels/").unwrap();
147
148        Self {
149            files: vec![
150                MetaFile {
151                    uri: nyx_cloud_stor.join("de440s.bsp").unwrap().to_string(),
152                    crc32: Some(0x7286750a),
153                },
154                MetaFile {
155                    uri: nyx_cloud_stor.join("v0.7/pck11.pca").unwrap().to_string(),
156                    crc32: Some(0x3503391),
157                },
158                MetaFile {
159                    uri: nyx_cloud_stor
160                        .join("v0.7/moon_fk_de440.epa")
161                        .unwrap()
162                        .to_string(),
163                    crc32: Some(0xabd5ff11),
164                },
165                MetaFile {
166                    uri: nyx_cloud_stor
167                        .join("moon_pa_de440_200625.bpc")
168                        .unwrap()
169                        .to_string(),
170                    crc32: Some(0xcde5ca7d),
171                },
172                MetaFile {
173                    uri: jpl_cloud_stor
174                        .join("pck/earth_latest_high_prec.bpc")
175                        .unwrap()
176                        .to_string(),
177                    crc32: None,
178                },
179            ],
180        }
181    }
