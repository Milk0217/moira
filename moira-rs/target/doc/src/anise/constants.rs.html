91/// References used in the constants.
92/// \[1\] Jay Lieske, ``Precession Matrix Based on IAU (1976)
93/// System of Astronomical Constants,'' Astron. Astrophys.
94/// 73, 282-284 (1979).

102/// 314.6-581, 24 June 1985.

105/// Ephemeris,'' JPL IOM 314.6-929, 26 February 1988.

119    /// Earth mean equator, dynamical equinox of J2000. The root reference frame for SPICE.
120    pub const J2000: NaifId = 1;
121    /// Earth mean equator, dynamical equinox of B1950.
122    /// The B1950 reference frame is obtained by precessing the J2000 frame backwards from Julian year 2000 to Besselian year 1950, using the 1976 IAU precession model.
123    /// The rotation from B1950 to J2000 is
124    /// \[ -z \]  \[ theta \]  \[ -zeta \]
125    ///         3            2            3
126    /// The values for z, theta, and zeta are computed from the formulas given in table 5 of \[5\].
127    /// z     =  1153.04066200330"
128    /// theta =  1002.26108439117"
129    /// zeta  =  1152.84248596724"
130    pub const B1950: NaifId = 2;
131    /// Fundamental Catalog (4). The FK4 reference frame is derived from the B1950 frame by applying the equinox offset determined by Fricke.
132    /// \[ 0.525" \]
133    ///             3
134    pub const FK4: NaifId = 3;
135
136    /// JPL Developmental Ephemeris (118). The DE-118 reference frame is nearly identical to the FK4 frame. It is also derived from the B1950 frame.
137    /// Only the offset is different
138    ///
139    ///  \[ 0.53155" \]
140    ///                3
141    ///
142    /// In \[2\], Standish uses two separate rotations,
143    ///
144    ///   \[ 0.00073" \]  P \[ 0.5316" \]
145    ///                 3                3
146    ///
147    /// (where P is the precession matrix used above to define the B1950 frame). The major effect of the second rotation is to correct for truncating the magnitude of the first rotation.
148    /// At his suggestion, we will use the untruncated value, and stick to a single rotation.
149    ///
150    ///
151    /// Most of the other DE historical reference frames are defined relative to either the DE-118 or B1950 frame.
152    /// The values below are taken from \[4\].
153    ///```text
154    ///    DE number  Offset from DE-118  Offset from B1950
155    ///    ---------  ------------------  -----------------
156    ///           96            +0.1209"           +0.4107"
157    ///          102            +0.3956"           +0.1359"
158    ///          108            +0.0541"           +0.4775"
159    ///          111            -0.0564"           +0.5880"
160    ///          114            -0.0213"           +0.5529"
161    ///          122            +0.0000"           +0.5316"
162    ///          125            -0.0438"           +0.5754"
163    ///          130            +0.0069"           +0.5247"
164    ///```
165    pub const DE118: NaifId = 4;
166    pub const DE096: NaifId = 5;
167    pub const DE102: NaifId = 6;
168    pub const DE108: NaifId = 7;
169    pub const DE111: NaifId = 8;
170    pub const DE114: NaifId = 9;
171    pub const DE122: NaifId = 10;
172    pub const DE125: NaifId = 11;
173    pub const DE130: NaifId = 12;
174    /// Galactic System II. The Galactic System II reference frame is defined by the following rotations:
175    ///       o            o              o
176    /// \[ 327  \]  \[ 62.6  \]  \[ 282.25  \]
177    ///           3            1             3
178    /// In the absence of better information, we assume the rotations are relative to the FK4 frame.
179    pub const GALACTIC: NaifId = 13;
180    pub const DE200: NaifId = 14;
181    pub const DE202: NaifId = 15;

183    /// This vector is the cross product of Earth mean north with Mars mean north.
184    pub const MARSIAU: NaifId = 16;
