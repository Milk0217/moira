34#[macro_export]
35macro_rules! parse_bytes_as {
36    ($type:ident, $input:expr, $order:expr) => {{
37        let (int_bytes, _) = $input.split_at(std::mem::size_of::<$type>());
38
39        match $order {
40            Endian::Little => $type::from_le_bytes(int_bytes.try_into().unwrap()),
41            Endian::Big => $type::from_be_bytes(int_bytes.try_into().unwrap()),
42        }
43    }};
44}
45
46#[derive(Debug, Copy, Clone, PartialEq, Eq, PartialOrd, Ord)]
47pub enum Endian {
48    Little,
49    Big,
50}
51
52impl Endian {
53    /// Returns the endianness of the platform we're running on for an f64.
54    /// This isn't const because f64 comparisons cannot be const yet
55    fn f64_native() -> Self {
56        let truth: f64 = 0.12345678;
57        if (f64::from_ne_bytes(truth.to_be_bytes()) - truth).abs() < f64::EPSILON {
58            Self::Big
59        } else {
60            Self::Little
61        }
62    }
63
64    /// Returns the endianness of the platform we're running on for an f64.
65    const fn u64_native() -> Self {
66        let truth: u32 = 0x12345678;
67        if u32::from_ne_bytes(truth.to_be_bytes()) == truth {
68            Self::Big
69        } else {
70            Self::Little
71        }
72    }
73}
