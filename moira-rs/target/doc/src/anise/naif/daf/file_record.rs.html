21#[derive(Debug, Snafu, PartialEq)]
22#[snafu(visibility(pub(crate)))]
23#[non_exhaustive]
24pub enum FileRecordError {
25    #[snafu(display("issue: endian of file does not match the endian order of the machine"))]
26    WrongEndian,
27    #[snafu(display("endian flag or internal filename is not a valid UTF8 string: {source:?}"))]
28    ParsingError {
29        source: Utf8Error,
30    },
31    #[snafu(display("endian flag is `{read}` but it should be either `BIG-IEEE` or `LTL-IEEE`"))]
32    InvalidEndian {
33        read: String,
34    },
35    UnsupportedIdentifier {
36        loci: String,
37    },
38    #[snafu(display("indicates this is not a SPICE DAF file"))]
39    NotDAF,
40    #[snafu(display("has no identifier"))]
41    NoIdentifier,
42    #[snafu(display("is empty (ensure file is valid, e.g. do you need to run git-lfs)"))]
43    EmptyRecord,
44}
45
46#[derive(Debug, Clone, FromBytes, KnownLayout, Immutable, IntoBytes, PartialEq)]
47#[repr(C)]
48pub struct FileRecord {
49    pub id_str: [u8; 8],
50    pub nd: u32,
51    pub ni: u32,
52    pub internal_filename: [u8; 60],
53    pub forward: u32,
54    pub backward: u32,
55    pub free_addr: u32,
56    pub endian_str: [u8; 8],
57    pub pre_null: [u8; 603],
58    pub ftp_str: [u8; 28],
59    pub pst_null: [u8; 297],
60}
61
62impl Default for FileRecord {
63    fn default() -> Self {
64        Self {
65            id_str: [0; 8],
66            nd: Default::default(),
67            ni: Default::default(),
68            internal_filename: [0; 60],
69            forward: Default::default(),
70            backward: Default::default(),
71            free_addr: Default::default(),
72            endian_str: [0; 8],
73            pre_null: [0; 603],
74            ftp_str: [0; 28],
75            pst_null: [0; 297],
76        }
77    }
78}
79
80impl NAIFRecord for FileRecord {}
81
82impl FileRecord {
83    pub fn ni(&self) -> usize {
84        self.ni as usize
85    }
86
87    pub fn nd(&self) -> usize {
88        self.nd as usize
89    }
90
91    pub fn fwrd_idx(&self) -> usize {
92        self.forward as usize
93    }
94
95    pub fn summary_size(&self) -> usize {
96        (self.nd + self.ni.div_ceil(2)) as usize
97    }
98
99    pub fn identification(&self) -> Result<&str, FileRecordError> {
100        let str_locidw =
101            core::str::from_utf8(&self.id_str).map_err(|_| FileRecordError::NoIdentifier)?;
102
103        if &str_locidw[0..3] != "DAF" || str_locidw.chars().nth(3) != Some('/') {
104            Err(FileRecordError::NotDAF)
105        } else {
106            let loci = str_locidw[4..].trim();
107            match loci {
108                "SPK" => Ok("SPK"),
109                "PCK" => Ok("PCK"),
110                _ => {
111                    error!("DAF of type `{}` is not yet supported", &str_locidw[4..]);
112                    Err(FileRecordError::UnsupportedIdentifier {
113                        loci: loci.to_string(),
114                    })
115                }
116            }
117        }
118    }
119
120    pub fn endianness(&self) -> Result<Endian, FileRecordError> {
121        let str_endianness = core::str::from_utf8(&self.endian_str).context(ParsingSnafu)?;
122
123        let file_endian = if str_endianness == "LTL-IEEE" {
124            Endian::Little
125        } else if str_endianness == "BIG-IEEE" {
126            Endian::Big
127        } else {
128            return Err(FileRecordError::InvalidEndian {
129                read: str_endianness.to_string(),
130            });
131        };
132        if file_endian != Endian::f64_native() || file_endian != Endian::u64_native() {
133            Err(FileRecordError::WrongEndian)
134        } else {
135            Ok(file_endian)
136        }
137    }
138
139    pub fn internal_filename(&self) -> Result<&str, FileRecordError> {
140        Ok(core::str::from_utf8(&self.internal_filename)
141            .context(ParsingSnafu)?
142            .trim())
143    }
144
145    /// Returns whether this record was just null bytes
146    pub fn is_empty(&self) -> bool {
147        self == &Self::default()
148    }
149}
