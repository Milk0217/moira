44pub(crate) const RCRD_LEN: usize = 1024;
45#[derive(Clone, Default, Debug, PartialEq)]
46pub struct GenericDAF<R: NAIFSummaryRecord, W: MutKind> {
47    pub bytes: W,
48    pub crc32_checksum: u32,
49    pub _daf_type: PhantomData<R>,
50}

52pub type DAF<R> = GenericDAF<R, Bytes>;
53pub type MutDAF<R> = GenericDAF<R, BytesMut>;

55pub trait MutKind: Deref<Target = [u8]> {}
56impl MutKind for Bytes {}
57impl MutKind for BytesMut {}

59impl<R: NAIFSummaryRecord, W: MutKind> GenericDAF<R, W> {
60    /// Compute the CRC32 of the underlying bytes
61    pub fn crc32(&self) -> u32 {
62        crc32fast::hash(&self.bytes)
63    }
64
65    /// Scrubs the data by computing the CRC32 of the bytes and making sure that it still matches the previously known hash
66    pub fn scrub(&self) -> Result<(), IntegrityError> {
67        if self.crc32() == self.crc32_checksum {
68            Ok(())
69        } else {
70            // Compiler will optimize the double computation away
71            Err(IntegrityError::ChecksumInvalid {
72                expected: self.crc32_checksum,
73                computed: self.crc32(),
74            })
75        }
76    }
77
78    /// Reads and parses the file record from the DAF bytes.
79    /// The file record is always the first 1024 bytes of the file.
80    pub fn file_record(&self) -> Result<FileRecord, DAFError> {
81        let file_record = FileRecord::read_from_bytes(
82            self.bytes
83                .get(..FileRecord::SIZE)
84                .ok_or_else(|| DecodingError::InaccessibleBytes {
85                    start: 0,
86                    end: FileRecord::SIZE,
87                    size: self.bytes.len(),
88                })
89                .context(DecodingDataSnafu {
90                    idx: 0_usize,
91                    kind: R::NAME,
92                })?,
93        )
94        .unwrap();
95        // Check that the endian-ness is compatible with this platform.
96        file_record
97            .endianness()
98            .context(FileRecordSnafu { kind: R::NAME })?;
99        Ok(file_record)
100    }
101
102    /// Reads and parses the name record from the DAF bytes.
103    /// The file record contains a pointer to the start of the name record.
104    pub fn name_record(&self) -> Result<NameRecord, DAFError> {
105        let rcrd_idx = self.file_record()?.fwrd_idx() * RCRD_LEN;
106        let rcrd_bytes = self
107            .bytes
108            .get(rcrd_idx..rcrd_idx + RCRD_LEN)
109            .ok_or_else(|| DecodingError::InaccessibleBytes {
110                start: rcrd_idx,
111                end: rcrd_idx + RCRD_LEN,
112                size: self.bytes.len(),
113            })
114            .context(DecodingNameSnafu { kind: R::NAME })?;
115        Ok(NameRecord::read_from_bytes(rcrd_bytes).unwrap())
116    }
117
118    /// Reads and parses the DAF summary record.
119    pub fn daf_summary(&self) -> Result<SummaryRecord, DAFError> {
120        let rcrd_idx = (self.file_record()?.fwrd_idx() - 1) * RCRD_LEN;
121        let rcrd_bytes = self
122            .bytes
123            .get(rcrd_idx..rcrd_idx + RCRD_LEN)
124            .ok_or_else(|| DecodingError::InaccessibleBytes {
125                start: rcrd_idx,
126                end: rcrd_idx + RCRD_LEN,
127                size: self.bytes.len(),
128            })
129            .context(DecodingSummarySnafu { kind: R::NAME })?;
130
131        SummaryRecord::read_from_bytes(&rcrd_bytes[..SummaryRecord::SIZE])
132            .or(Err(DecodingError::Casting))
133            .context(DecodingSummarySnafu { kind: R::NAME })
134    }
135
136    /// Parses and returns a slice of the data summaries.
137    /// The summaries are located in the same record as the DAF summary.
138    pub fn data_summaries(&self) -> Result<&[R], DAFError> {
139        if self.file_record()?.is_empty() {
140            return Err(DAFError::FileRecord {
141                kind: R::NAME,
142                source: FileRecordError::EmptyRecord,
143            });
144        }
145
146        // The file record's forward pointer points to the first summary record.
147        let rcrd_idx = (self.file_record()?.fwrd_idx() - 1) * RCRD_LEN;
148        let rcrd_bytes = match self
149            .bytes
150            .get(rcrd_idx..rcrd_idx + RCRD_LEN)
151            .ok_or_else(|| DecodingError::InaccessibleBytes {
152                start: rcrd_idx,
153                end: rcrd_idx + RCRD_LEN,
154                size: self.bytes.len(),
155            }) {
156            Ok(it) => it,
157            Err(source) => {
158                return Err(DAFError::DecodingSummary {
159                    kind: R::NAME,
160                    source,
161                })
162            }
163        };
164
165        // The summaries are located after the main DAF summary record within the same record.
166        Ok(
167            match Ref::<_, [R]>::from_bytes(&rcrd_bytes[SummaryRecord::SIZE..]) {
168                Ok(r) => Ref::into_ref(r),
169                Err(_) => &{
170                    R::default();
171                    [] as [R; 0]
172                },
173            },
174        )
175    }
176
177    /// Returns the summary given the name of the summary record
178    pub fn summary_from_name(&self, name: &str) -> Result<(&R, usize), DAFError> {
179        let idx = self
180            .name_record()?
181            .index_from_name::<R>(name, self.file_record()?.summary_size())?;
182
183        Ok((&self.data_summaries()?[idx], idx))
184    }
185
186    /// Returns the summary given the name of the summary record if that summary has data defined at the requested epoch
187    pub fn summary_from_name_at_epoch(
188        &self,
189        name: &str,
190        epoch: Epoch,
191    ) -> Result<(&R, usize), DAFError> {
192        let (summary, idx) = self.summary_from_name(name)?;
193
194        if epoch >= summary.start_epoch() - Unit::Nanosecond * 100
195            && epoch <= summary.end_epoch() + Unit::Nanosecond * 100
196        {
197            Ok((summary, idx))
198        } else {
199            error!("No summary {name} valid at epoch {epoch}");
200            Err(DAFError::SummaryNameAtEpochError {
201                kind: R::NAME,
202                name: name.to_string(),
203                epoch,
204            })
205        }
206    }
207
208    /// Returns the summary given the id of the summary record
209    pub fn summary_from_id(&self, id: i32) -> Result<(&R, usize), DAFError> {
210        for (idx, summary) in self.data_summaries()?.iter().enumerate() {
211            if summary.id() == id {
212                return Ok((summary, idx));
213            }
214        }
215
216        Err(DAFError::SummaryIdError { kind: R::NAME, id })
217    }
218
219    /// Returns the summary given the name of the summary record if that summary has data defined at the requested epoch
220    pub fn summary_from_id_at_epoch(&self, id: i32, epoch: Epoch) -> Result<(&R, usize), DAFError> {
221        // NOTE: We iterate through the whole summary because a specific NAIF ID may be repeated in the summary for different valid epochs
222        // so we can't just call `summary_from_id`.
223        for (idx, summary) in self.data_summaries()?.iter().enumerate() {
224            if summary.id() == id {
225                if epoch >= summary.start_epoch() - Unit::Nanosecond * 100
226                    && epoch <= summary.end_epoch() + Unit::Nanosecond * 100
227                {
228                    trace!("Found {id} in position {idx}: {summary:?}");
229                    return Ok((summary, idx));
230                } else {
231                    debug!(
232                        "Summary {id} not valid at {epoch:?} (only from {:?} to {:?}, offset of {} - {})",
233                        summary.start_epoch(),
234                        summary.end_epoch(),
235                        epoch - summary.start_epoch(),
236                        summary.end_epoch() - epoch
237                    );
238                }
239            }
240        }
241        Err(DAFError::InterpolationDataErrorFromId {
242            kind: R::NAME,
243            id,
244            epoch,
245        })
246    }

248    /// Provided a name that is in the summary, return its full data, if name is available.
249    pub fn data_from_name<'a, S: NAIFDataSet<'a>>(&'a self, name: &str) -> Result<S, DAFError> {
250        // O(N) search through the summaries
251        let name_rcrd = self.name_record()?;
252        for idx in 0..name_rcrd.num_entries(self.file_record()?.summary_size()) {
253            let this_name = name_rcrd.nth_name(idx, self.file_record()?.summary_size());
254
255            if name.trim() == this_name.trim() {
256                // Found it!
257                return self.nth_data(idx);
258            }
259        }
260        Err(DAFError::NameError {
261            kind: R::NAME,
262            name: name.to_string(),
263        })
264    }
265
266    /// Provided a name that is in the summary, return its full data, if name is available.
267    /// This function retrieves the data associated with the nth summary record.
268    pub fn nth_data<'a, S: NAIFDataSet<'a>>(&'a self, idx: usize) -> Result<S, DAFError> {
269        let this_summary = self
270            .data_summaries()?
271            .get(idx)
272            .ok_or(DAFError::InvalidIndex {
273                idx,
274                kind: S::DATASET_NAME,
275            })?;
276        // Grab the data in native endianness (TODO: How to support both big and little endian?)
277        trace!("{idx} -> {this_summary:?}");
278        if self.file_record()?.is_empty() {
279            return Err(DAFError::FileRecord {
280                kind: R::NAME,
281                source: FileRecordError::EmptyRecord,
282            });
283        }
284
285        let start = (this_summary.start_index() - 1) * DBL_SIZE;
286        let end = this_summary.end_index() * DBL_SIZE;
287        let data: &[f64] = Ref::into_ref(
288            Ref::<&[u8], [f64]>::from_bytes(
289                match self
290                    .bytes
291                    .get(start..end)
292                    .ok_or_else(|| DecodingError::InaccessibleBytes {
293                        start,
294                        end,
295                        size: self.bytes.len(),
296                    }) {
297                    Ok(it) => it,
298                    Err(source) => {
299                        return Err(DAFError::DecodingData {
300                            kind: R::NAME,
301                            idx,
302                            source,
303                        })
304                    }
305                },
306            )
307            .unwrap(),
308        );
309
310        // Convert it
311        S::from_f64_slice(data).context(DecodingDataSnafu { kind: R::NAME, idx })
312    }

314    pub fn comments(&self) -> Result<Option<String>, DAFError> {
315        // TODO: This can be cleaned up to avoid allocating a string. In my initial tests there were a bunch of additional spaces, so I canceled those changes.
316        let mut rslt = String::new();
317        // FWRD has the initial record of the summary. So we assume that all records between the second record and that one are comments
318        for rid in 1..self.file_record()?.fwrd_idx() {
319            match core::str::from_utf8(
320                match self
321                    .bytes
322                    .get(rid * RCRD_LEN..(rid + 1) * RCRD_LEN)
323                    .ok_or_else(|| DecodingError::InaccessibleBytes {
324                        start: rid * RCRD_LEN,
325                        end: (rid + 1) * RCRD_LEN,
326                        size: self.bytes.len(),
327                    }) {
328                    Ok(it) => it,
329                    Err(source) => {
330                        return Err(DAFError::DecodingComments {
331                            kind: R::NAME,
332                            source,
333                        })
334                    }
335                },
336            ) {
337                Ok(s) => rslt += s.replace('\u{0}', "\n").trim(),
338                Err(e) => {
339                    // At this point, we know that the bytes are accessible because the embedded `match`
340                    // did not fail, so we can perform a direct access.
341                    let valid_s = core::str::from_utf8(
342                        &self.bytes[rid * RCRD_LEN..(rid * RCRD_LEN + e.valid_up_to())],
343                    )
344                    .unwrap();
345                    rslt += valid_s.replace('\u{0}', "\n").trim()
346                }
347            }
348        }
349
350        if rslt.is_empty() {
351            Ok(None)
352        } else {
353            Ok(Some(rslt))
354        }
355    }

357    /// Writes the contents of this DAF file to a new location.
358    pub fn persist<P: AsRef<Path>>(&self, path: P) -> IoResult<()> {
359        let mut fs = File::create(path)?;
360
361        let mut file_rcrd = Vec::from(self.file_record().unwrap().as_bytes());
362        file_rcrd.extend(vec![
363            0x0;
364            (self.file_record().unwrap().fwrd_idx() - 1) * RCRD_LEN
365                - file_rcrd.len()
366        ]);
367        fs.write_all(&file_rcrd)?;
368
369        let mut daf_summary = Vec::from(self.daf_summary().unwrap().as_bytes());
370        // Extend with the data summaries
371        for data_summary in self.data_summaries().unwrap() {
372            daf_summary.extend(data_summary.as_bytes());
373        }
374        // And pad with NULL
375        daf_summary.extend(vec![0x0; RCRD_LEN - daf_summary.len()]);
376        fs.write_all(&daf_summary)?;
377
378        let mut name_rcrd = Vec::from(self.name_record().unwrap().as_bytes());
379        name_rcrd.extend(vec![0x0; RCRD_LEN - name_rcrd.len()]);
380        fs.write_all(&name_rcrd)?;
381
382        fs.write_all(&self.bytes[self.file_record().unwrap().fwrd_idx() * (2 * RCRD_LEN)..])
383    }

386impl<R: NAIFSummaryRecord, W: MutKind> Hash for GenericDAF<R, W> {
387    /// Hash will only hash the bytes, nothing else (since these are derived from the bytes anyway).
388    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
389        self.bytes.hash(state);
390    }
391}

393impl<R: NAIFSummaryRecord> DAF<R> {
394    /// Parse the provided bytes as a SPICE Double Array File.
395    ///
396    /// # DAF File Structure
397    /// A DAF is composed of three main parts:
398    /// 1.  **File Record:** The first record (1024 bytes) of the file. It contains metadata about the file, such as the endianness, the number of records, and pointers to the other sections.
399    /// 2.  **Comment Area:** An optional area for storing comments.
400    /// 3.  **Summary/Name Records and Data:** The remaining records contain the summary records, name records, and the actual data arrays. The file record contains pointers to the start of these sections.
401    ///
402    /// # Parsing Process
403    /// 1.  The entire file is read into a `Bytes` object.
404    /// 2.  The CRC32 checksum of the bytes is computed.
405    /// 3.  The `file_record` and `name_record` are parsed to ensure the file is a valid DAF.
406    pub fn parse<B: Deref<Target = [u8]>>(bytes: B) -> Result<Self, DAFError> {
407        let crc32_checksum = crc32fast::hash(&bytes);
408        let me = Self {
409            bytes: Bytes::copy_from_slice(&bytes),
410            crc32_checksum,
411            _daf_type: PhantomData,
412        };
413        // Check that the file record and name record can be parsed successfully.
414        // This validates that the file is a DAF and that the endianness is correct.
415        me.file_record()?;
416        me.name_record()?;
417        Ok(me)
418    }
419
420    /// Parse the DAF only if the CRC32 checksum of the data is valid
421    pub fn check_then_parse<B: Deref<Target = [u8]>>(
422        bytes: B,
423        expected: u32,
424    ) -> Result<Self, DAFError> {
425        let computed = crc32fast::hash(&bytes);
426        if computed != expected {
427            return Err(DAFError::DAFIntegrity {
428                source: IntegrityError::ChecksumInvalid { expected, computed },
429            });
430        }
431
432        Self::parse(bytes)
433    }
434
435    pub fn load(path: &str) -> Result<Self, DAFError> {
436        let bytes = file2heap!(path).context(IOSnafu {
437            action: format!("loading {path:?}"),
438        })?;
439
440        Self::parse(bytes)
441    }
442
443    /// Parse the provided static byte array as a SPICE Double Array File
444    pub fn from_static<B: Deref<Target = [u8]>>(bytes: &'static B) -> Result<Self, DAFError> {
445        Self::parse(Bytes::from_static(bytes))
446    }
447
448    /// Copies the underlying bytes of this DAF into a MutDAF, enabling modification of the DAF.
449    pub fn to_mutable(&self) -> MutDAF<R> {
450        MutDAF {
451            bytes: BytesMut::from_iter(&self.bytes),
452            crc32_checksum: self.crc32_checksum,
453            _daf_type: PhantomData,
454        }
455    }
456}

472    #[test]
473    fn crc32_errors() {
474        let mut traj = SPK::load("../data/gmat-hermite.bsp").unwrap();
475        let nominal_crc = traj.crc32();
476
477        assert_eq!(
478            SPK::check_then_parse(
479                file2heap!("../data/gmat-hermite.bsp").unwrap(),
480                nominal_crc + 1
481            ),
482            Err(DAFError::DAFIntegrity {
483                source: IntegrityError::ChecksumInvalid {
484                    expected: nominal_crc + 1,
485                    computed: nominal_crc
486                },
487            })
488        );
489
490        // Change the checksum of the traj and check that scrub fails
491        traj.crc32_checksum += 1;
492        assert_eq!(
493            traj.scrub(),
494            Err(IntegrityError::ChecksumInvalid {
495                expected: nominal_crc + 1,
496                computed: nominal_crc
497            })
498        );
499    }

528            // We cannot user assert_eq! because the NAIF Data Set do not (and should not) impl Debug
