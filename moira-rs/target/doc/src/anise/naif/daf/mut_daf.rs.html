28impl<R: NAIFSummaryRecord> MutDAF<R> {
29    /// Parse the provided bytes as a SPICE Double Array File
30    pub fn parse<B: Deref<Target = [u8]>>(bytes: B) -> Result<Self, DAFError> {
31        let crc32_checksum = crc32fast::hash(&bytes);
32        let mut buf = BytesMut::with_capacity(0);
33        buf.extend(bytes.iter());
34        let me = Self {
35            bytes: buf,
36            crc32_checksum,
37            _daf_type: PhantomData,
38        };
39        // Check that these calls will succeed.
40        me.file_record()?;
41        me.name_record()?;
42        Ok(me)
43    }
44
45    pub fn load(path: &str) -> Result<Self, DAFError> {
46        let bytes = file2heap!(path).context(IOSnafu {
47            action: format!("loading {path:?}"),
48        })?;
49
50        Self::parse(bytes)
51    }

53    /// Sets the name record of this mutable DAF file to the one provided as a parameter.
54    pub fn set_name_record(&mut self, new_name_record: NameRecord) -> Result<(), DAFError> {
55        let rcrd_idx = self.file_record()?.fwrd_idx() * RCRD_LEN;
56        let size = self.bytes.len();
57        let rcrd_bytes = self
58            .bytes
59            .get_mut(rcrd_idx..rcrd_idx + RCRD_LEN)
60            .ok_or_else(|| DecodingError::InaccessibleBytes {
61                start: rcrd_idx,
62                end: rcrd_idx + RCRD_LEN,
63                size,
64            })
65            .context(DecodingNameSnafu { kind: R::NAME })?;
66        rcrd_bytes.copy_from_slice(new_name_record.as_bytes());
67        Ok(())
68    }

70    /// Sets the data for the n-th segment of this DAF file.
71    pub fn set_nth_data<'a, S: NAIFDataSet<'a>>(
72        &mut self,
73        idx: usize,
74        new_data: S,
75        new_start_epoch: Epoch,
76        new_end_epoch: Epoch,
77    ) -> Result<(), DAFError> {
78        let summaries = self.data_summaries()?;
79        let this_summary = summaries
80            .get(idx)
81            .ok_or(DAFError::InvalidIndex { idx, kind: R::NAME })?;
82
83        if self.file_record()?.is_empty() {
84            return Err(DAFError::FileRecord {
85                kind: R::NAME,
86                source: FileRecordError::EmptyRecord,
87            });
88        }
89
90        let orig_index_start = this_summary.start_index() - 1;
91        let orig_index_end = this_summary.end_index();
92        let orig_data_start = orig_index_start * DBL_SIZE;
93        let orig_data_end = orig_index_end * DBL_SIZE;
94
95        let original_size = ((orig_data_end - orig_data_start) / DBL_SIZE) as isize;
96
97        let new_data_bytes = new_data
98            .to_f64_daf_vec()
99            .or(Err(DAFError::DataBuildError { kind: R::NAME }))?;
100
101        let new_size = new_data_bytes.len() as isize;
102
103        // Size change will be positive if the new data is _smaller_ than the previous one, and vice versa.
104        let size_change = original_size - new_size;
105
106        // Update the bytes
107        let mut new_bytes = self.bytes.to_vec();
108        new_bytes.splice(
109            orig_data_start..orig_data_end,
110            new_data_bytes.as_bytes().iter().cloned(),
111        );
112
113        let mut new_summaries: Vec<R> = summaries.to_vec();
114        for (sno, summary) in new_summaries.iter_mut().enumerate() {
115            if sno < idx {
116                continue;
117            } else if sno == idx {
118                // Only update the end index for the data we modified
119                summary.update_indexes(
120                    orig_index_start + 1,
121                    (orig_index_end as isize - size_change) as usize,
122                );
123                summary.update_epochs(new_start_epoch, new_end_epoch);
124            } else if !summary.is_empty() {
125                // Shift all of the indexes.
126                let prev_start = summary.start_index();
127                let prev_end = summary.end_index();
128                summary.update_indexes(
129                    (prev_start as isize - size_change) as usize,
130                    (prev_end as isize - size_change) as usize,
131                );
132            }
133        }
134
135        let summary_bytes: Vec<u8> = new_summaries.as_bytes().to_vec();
136
137        let rcrd_idx = (self.file_record()?.fwrd_idx() - 1) * RCRD_LEN;
138        // Note: we use copy_from_slice here because we have the guarantee that the summary bytes are the same length as the original version.
139        let orig_summary_bytes =
140            &mut new_bytes[rcrd_idx..rcrd_idx + RCRD_LEN][SummaryRecord::SIZE..];
141        orig_summary_bytes.copy_from_slice(&summary_bytes);
142
143        self.bytes = BytesMut::from_iter(new_bytes);
144
145        Ok(())
146    }

148    /// Deletes the data for the n-th segment of this DAF file.
149    pub fn delete_nth_data(&mut self, idx: usize) -> Result<(), DAFError> {
150        let summaries = self.data_summaries()?;
151        let this_summary = summaries
152            .get(idx)
153            .ok_or(DAFError::InvalidIndex { idx, kind: R::NAME })?;
154
155        if self.file_record()?.is_empty() {
156            return Err(DAFError::FileRecord {
157                kind: R::NAME,
158                source: FileRecordError::EmptyRecord,
159            });
160        }
161
162        let orig_index_start = this_summary.start_index() - 1;
163        let orig_index_end = this_summary.end_index();
164        let orig_data_start = orig_index_start * DBL_SIZE;
165        let orig_data_end = orig_index_end * DBL_SIZE;
166
167        let original_size = ((orig_data_end - orig_data_start) / DBL_SIZE) as isize;
168
169        // Size change will be positive if the new data is _smaller_ than the previous one, and vice versa.
170        let size_change = original_size;
171
172        // Update the bytes
173        let mut new_bytes = self.bytes.to_vec();
174        new_bytes.drain(orig_data_start..orig_data_end);
175
176        let mut new_summaries: Vec<R> = summaries.to_vec();
177        for (sno, summary) in new_summaries.iter_mut().enumerate() {
178            if sno < idx {
179                continue;
180            } else if sno == idx {
181                // We've removed this data, so clear the summary.
182                *summary = R::default()
183            } else if !summary.is_empty() {
184                // Shift all of the indexes.
185                let prev_start = summary.start_index();
186                let prev_end = summary.end_index();
187                summary.update_indexes(
188                    (prev_start as isize - size_change) as usize,
189                    (prev_end as isize - size_change) as usize,
190                );
191            }
192        }
193
194        // Remove empty entries from the summary all together
195        let cleaned_summaries: Vec<R> = new_summaries
196            .iter()
197            .filter(|summary| !summary.is_empty())
198            .cloned()
199            .collect();
200
201        let mut summary_bytes: Vec<u8> = cleaned_summaries.as_bytes().to_vec();
202        // We need to pad with zeros all of the summaries we've removed.
203        summary_bytes.extend(vec![0x0; 1000 - summary_bytes.len()]);
204
205        let rcrd_idx = (self.file_record()?.fwrd_idx() - 1) * RCRD_LEN;
206        // Note: we use copy_from_slice here because we have the guarantee that the summary bytes are the same length as the original version.
207        let orig_summary_bytes =
208            &mut new_bytes[rcrd_idx..rcrd_idx + RCRD_LEN][SummaryRecord::SIZE..];
209        orig_summary_bytes.copy_from_slice(&summary_bytes);
210
211        self.bytes = BytesMut::from_iter(new_bytes);
212
213        Ok(())
214    }
