29#[derive(Debug, PartialEq, Snafu)]
30#[snafu(visibility(pub))]
31pub enum AlmanacError {
32    #[snafu(display("{action} encountered an error with ephemeris computation {source}"))]
33    Ephemeris {
34        action: &'static str,
35        #[snafu(source(from(EphemerisError, Box::new)))]
36        source: Box<EphemerisError>,
37    },
38    #[snafu(display("{action} encountered an error with orientation computation {source}"))]
39    Orientation {
40        action: &'static str,
41        #[snafu(source(from(OrientationError, Box::new)))]
42        source: Box<OrientationError>,
43    },
44    #[snafu(display("{source} encountered when loading {path}"))]
45    Loading {
46        path: String,
47        source: InputOutputError,
48    },
49    #[snafu(display("{source} encountered when {action}"))]
50    TLDataSet {
51        action: &'static str,
52        source: DataSetError,
53    },
54    #[snafu(display("{err}"))]
55    GenericError { err: String },
56    #[cfg(feature = "metaload")]
57    #[snafu(display("processing file #{fno} ({file:?}) caused an error: {source}"))]
58    Meta {
59        fno: usize,
60        file: MetaFile,
61        source: MetaAlmanacError,
62    },
63}
64
65pub type AlmanacResult<T> = Result<T, AlmanacError>;

77#[derive(Copy, Clone, Debug, Snafu, PartialEq)]
78#[snafu(visibility(pub(crate)))]
79#[non_exhaustive]
80pub enum DecodingError {
81    #[snafu(display(
82        "could not decode {dataset} data -- need at least {need} doubles but found {got}"
83    ))]
84    TooFewDoubles {
85        dataset: &'static str,
86        got: usize,
87        need: usize,
88    },
89    #[snafu(display("bytes between indexes {start} and {end} could not be read, array contains {size} bytes (data malformed?)"))]
90    InaccessibleBytes {
91        start: usize,
92        end: usize,
93        size: usize,
94    },
95    #[snafu(display("integrity error during decoding: {source}"))]
96    Integrity {
97        #[snafu(backtrace)]
98        source: IntegrityError,
99    },
100    #[snafu(display("decoding DER failed: {err}"))]
101    DecodingDer { err: DerError },
102    #[snafu(display("somehow casting the data failed"))]
103    Casting,
104    #[snafu(display("could not load ANISE data version {got}, expected {exp}"))]
105    AniseVersion { got: Semver, exp: Semver },
106    #[snafu(display("data could not be parsed as {kind} despite ANISE version matching (should be loaded as another type?)"))]
107    Obscure { kind: &'static str },
108}

110#[derive(Copy, Clone, PartialEq, Debug, Snafu)]
111#[snafu(visibility(pub(crate)))]
112#[non_exhaustive]
113pub enum IntegrityError {
114    /// Data checksum differs from expected checksum
115    ChecksumInvalid { expected: u32, computed: u32 },
116    /// Data between two ephemerides expected to be identical mismatch (may happen on merger of files)
117    DataMismatchOnMerge,
118    /// Could not fetch spline data that was expected to be there
119    DataMissing,
120    /// The lookup table is broken somehow
121    LookupTable,
122    /// Raised if a transformation is requested but the frames have no common origin
123    DisjointRoots {
124        from_frame: FrameUid,
125        to_frame: FrameUid,
126    },
127    #[snafu(display(
128        "data for {variable} in {dataset} decoded as subnormal double (data malformed?)"
129    ))]
130    SubNormal {
131        dataset: &'static str,
132        variable: &'static str,
133    },
134    #[snafu(display("data for {variable}={value} in {dataset} is invalid {reason}"))]
135    InvalidValue {
136        dataset: &'static str,
137        variable: &'static str,
138        value: f64,
139        reason: &'static str,
140    },
141}

155#[derive(Copy, Clone, Debug, Snafu, PartialEq)]
156#[snafu(visibility(pub(crate)))]
157#[non_exhaustive]
158pub enum PhysicsError {
159    /// Somehow you've entered code that should not be reachable, please file a bug.
160    Unreachable,
161    #[snafu(display("epochs {epoch1} and {epoch2} differ while {action}"))]
162    EpochMismatch {
163        action: &'static str,
164        epoch1: Epoch,
165        epoch2: Epoch,
166    },
167    #[snafu(display("frames {frame1} and {frame2} differ while {action}"))]
168    FrameMismatch {
169        action: &'static str,
170        frame1: FrameUid,
171        frame2: FrameUid,
172    },
173    #[snafu(display(
174        "cannot {action} because rotations {from1}->{to1} and {from2}->{to2} are incompatible"
175    ))]
176    InvalidRotation {
177        action: &'static str,
178        from1: NaifId,
179        to1: NaifId,
180        from2: NaifId,
181        to2: NaifId,
182    },
183    #[snafu(display("cannot rotate state in frame {state_frame} with rotation {from}->{to}"))]
184    InvalidStateRotation {
185        from: NaifId,
186        to: NaifId,
187        state_frame: FrameUid,
188    },
189    #[snafu(display("{action} requires the time derivative of the DCM but it is not set"))]
190    DCMMissingDerivative { action: &'static str },
191    #[snafu(display("{action} requires the frame {frame} to have {data} defined"))]
192    MissingFrameData {
193        action: &'static str,
194        data: &'static str,
195        frame: FrameUid,
196    },
197    #[snafu(display("parabolic orbits are physically impossible and the eccentricity calculated to be within {limit:e} of 1.0"))]
198    ParabolicEccentricity { limit: f64 },
199    #[snafu(display("parabolic orbits are physically impossible and the semilatus rectum (semi-parameter) calculated to be {p_km} km"))]
200    ParabolicSemiParam { p_km: f64 },
201    #[snafu(display("hyperbolic true anomaly is physically impossible: {ta_deg} deg"))]
202    HyperbolicTrueAnomaly { ta_deg: f64 },
203    #[snafu(display("calculation requires hyperbolic orbit, but its eccentricity is {ecc}"))]
204    NotHyperbolic { ecc: f64 },
205    #[snafu(display("infinite value encountered when {action}"))]
206    InfiniteValue { action: &'static str },
207    #[snafu(display("{source}"))]
208    AppliedMath { source: MathError },
209    #[snafu(display("invalid radius: {action}"))]
210    RadiusError { action: &'static str },
211    #[snafu(display("invalid velocity: {action}"))]
212    VelocityError { action: &'static str },
213    #[snafu(display("invalid aberration: {action}"))]
214    AberrationError { action: &'static str },
215}
