21/// **Limitation:** no translation or rotation may have more than 8 nodes.
22pub const MAX_TREE_DEPTH: usize = 8;
23
24impl Almanac {
25    /// Returns the root of all of the loaded orientations (BPC or planetary), typically this should be J2000.
26    ///
27    /// # Algorithm
28    ///
29    /// 1. For each loaded BPC, iterated in reverse order (to mimic SPICE behavior)
30    /// 2. For each summary record in each BPC, follow the orientation branch all the way up until the end of this BPC or until the J2000.
31    pub fn try_find_orientation_root(&self) -> Result<NaifId, OrientationError> {
32        ensure!(
33            self.num_loaded_bpc() > 0 || !self.planetary_data.is_empty(),
34            NoOrientationsLoadedSnafu
35        );
36
37        // The common center is the absolute minimum of all centers due to the NAIF numbering.
38        let mut common_center = i32::MAX;
39
40        for bpc in self.bpc_data.values().rev() {
41            for summary in bpc.data_summaries().context(BPCSnafu {
42                action: "finding orientation root",
43            })? {
44                // This summary exists, so we need to follow the branch of centers up the tree.
45                if !summary.is_empty() && summary.inertial_frame_id.abs() < common_center.abs() {
46                    common_center = summary.inertial_frame_id;
47                    if common_center == J2000 {
48                        // there is nothing higher up
49                        return Ok(common_center);
50                    }
51                }
52            }
53        }
54
55        // If we reached this point, it means that we didn't find J2000 in the loaded BPCs, so let's iterate through the planetary data
56        if !self.planetary_data.is_empty() {
57            for id in self.planetary_data.lut.by_id.keys() {
58                if let Ok(pc) = self.planetary_data.get_by_id(*id) {
59                    if pc.parent_id < common_center {
60                        common_center = pc.parent_id;
61                        if common_center == J2000 {
62                            // there is nothing higher up
63                            return Ok(common_center);
64                        }
65                    }
66                }
67            }
68        }
69
70        if common_center == ECLIPJ2000 {
71            // Rotation from ecliptic J2000 to J2000 is embedded.
72            common_center = J2000;
73        }
74
75        Ok(common_center)
76    }

78    /// Try to construct the path from the source frame all the way to the root orientation of this context.
79    pub fn orientation_path_to_root(
80        &self,
81        source: Frame,
82        epoch: Epoch,
83    ) -> Result<(usize, [Option<NaifId>; MAX_TREE_DEPTH]), OrientationError> {
84        let common_center = self.try_find_orientation_root()?;
85        // Build a tree, set a fixed depth to avoid allocations
86        let mut of_path = [None; MAX_TREE_DEPTH];
87        let mut of_path_len = 0;
88
89        if common_center == source.orientation_id {
90            // We're querying the source, no need to check that this summary even exists.
91            return Ok((of_path_len, of_path));
92        }
93
94        // Grab the summary data, which we use to find the paths
95        // Let's see if this orientation is defined in the loaded BPC files
96        let mut inertial_frame_id = match self.bpc_summary_at_epoch(source.orientation_id, epoch) {
97            Ok((summary, _, _)) => summary.inertial_frame_id,
98            Err(_) => {
99                // Not available as a BPC, so let's see if there's planetary data for it.
100                match self.planetary_data.get_by_id(source.orientation_id) {
101                    Ok(planetary_data) => planetary_data.parent_id,
102                    Err(_) => {
103                        // Finally, let's see if it's in the loaded Euler Parameters.
104                        self.euler_param_data
105                            .get_by_id(source.orientation_id)
106                            .context(OrientationDataSetSnafu)?
107                            .to
108                    }
109                }
110            }
111        };
112
113        of_path[of_path_len] = Some(inertial_frame_id);
114        of_path_len += 1;
115
116        if inertial_frame_id == ECLIPJ2000 {
117            // Add the hop to J2000
118            inertial_frame_id = J2000;
119            of_path[of_path_len] = Some(inertial_frame_id);
120            of_path_len += 1;
121        }
122
123        if inertial_frame_id == common_center {
124            // Well that was quick!
125            return Ok((of_path_len, of_path));
126        }
127
128        for _ in 0..MAX_TREE_DEPTH - 1 {
129            inertial_frame_id = match self.bpc_summary_at_epoch(inertial_frame_id, epoch) {
130                Ok((summary, _, _)) => summary.inertial_frame_id,
131                Err(_) => {
132                    // Not available as a BPC, so let's see if there's planetary data for it.
133                    match self.planetary_data.get_by_id(inertial_frame_id) {
134                        Ok(planetary_data) => planetary_data.parent_id,
135                        Err(_) => {
136                            // Finally, let's see if it's in the loaded Euler Parameters.
137                            self.euler_param_data
138                                .get_by_id(inertial_frame_id)
139                                .context(OrientationDataSetSnafu)?
140                                .to
141                        }
142                    }
143                }
144            };
145
146            of_path[of_path_len] = Some(inertial_frame_id);
147            of_path_len += 1;
148            if inertial_frame_id == common_center {
149                // We're found the path!
150                return Ok((of_path_len, of_path));
151            }
152        }
153
154        Err(OrientationError::BPC {
155            action: "computing path to common node",
156            source: DAFError::MaxRecursionDepth,
157        })
158    }

160    /// Returns the orientation path between two frames and the common node. This may return a `DisjointRoots` error if the frames do not share a common root, which is considered a file integrity error.
161    pub fn common_orientation_path(
162        &self,
163        from_frame: Frame,
164        to_frame: Frame,
165        epoch: Epoch,
166    ) -> Result<(usize, [Option<NaifId>; MAX_TREE_DEPTH], NaifId), OrientationError> {
167        if from_frame == to_frame {
168            // Both frames match, return this frame's hash (i.e. no need to go higher up).
169            return Ok((0, [None; MAX_TREE_DEPTH], from_frame.orientation_id));
170        }
171
172        // Grab the paths
173        let (from_len, from_path) = self.orientation_path_to_root(from_frame, epoch)?;
174        let (to_len, to_path) = self.orientation_path_to_root(to_frame, epoch)?;
175
176        // Now that we have the paths, we can find the matching origin.
177
178        // If either path is of zero length, that means one of them is at the root of this ANISE file, so the common
179        // path is which brings the non zero-length path back to the file root.
180        if from_len == 0 && to_len == 0 {
181            Err(OrientationError::RotationOrigin {
182                from: from_frame.into(),
183                to: to_frame.into(),
184                epoch,
185            })
186        } else if from_len != 0 && to_len == 0 {
187            // One has an empty path but not the other, so the root is at the empty path
188            Ok((from_len, from_path, to_frame.orientation_id))
189        } else if to_len != 0 && from_len == 0 {
190            // One has an empty path but not the other, so the root is at the empty path
191            Ok((to_len, to_path, from_frame.orientation_id))
192        } else {
193            // Either are at the orientation root, so we'll step through the paths until we find the common root.
194            let mut common_path = to_path;
195            let mut items: usize = to_len;
196            let common_node = to_path[to_len - 1].unwrap();
197
198            for from_obj in from_path.iter().take(from_len).rev().skip(1) {
199                common_path[items] = Some(from_obj.unwrap());
200                items += 1;
201            }
202
203            Ok((items, common_path, common_node))
204        }
205    }
