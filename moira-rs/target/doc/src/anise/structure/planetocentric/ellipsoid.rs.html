25/// Only the tri-axial Ellipsoid shape model is currently supported by ANISE.
26/// This is directly inspired from SPICE PCK.
27/// > For each body, three radii are listed: The first number is
28/// > the largest equatorial radius (the length of the semi-axis
29/// > containing the prime meridian), the second number is the smaller
30/// > equatorial radius, and the third is the polar radius.
31///
32/// Example: Radii of the Earth.
33///
34///    BODY399_RADII     = ( 6378.1366   6378.1366   6356.7519 )
35///
36/// :type semi_major_equatorial_radius_km: float
37/// :type polar_radius_km: float, optional
38/// :type semi_minor_equatorial_radius_km: float, optional
39/// :rtype: Ellipsoid
40#[derive(Copy, Clone, Debug, Serialize, Deserialize, PartialEq)]
41#[cfg_attr(feature = "metaload", derive(StaticType))]
42#[cfg_attr(feature = "python", pyclass)]
43#[cfg_attr(feature = "python", pyo3(module = "anise.astro"))]
44pub struct Ellipsoid {
45    pub semi_major_equatorial_radius_km: f64,
46    pub semi_minor_equatorial_radius_km: f64,
47    pub polar_radius_km: f64,
48}
49
50impl Ellipsoid {
51    /// Builds an ellipsoid as if it were a sphere
52    pub fn from_sphere(radius_km: f64) -> Self {
53        Self {
54            semi_major_equatorial_radius_km: radius_km,
55            semi_minor_equatorial_radius_km: radius_km,
56            polar_radius_km: radius_km,
57        }
58    }
59
60    /// Builds an ellipsoid as if it were a spheroid, where only the polar axis has a different radius
61    pub fn from_spheroid(equatorial_radius_km: f64, polar_radius_km: f64) -> Self {
62        Self {
63            semi_major_equatorial_radius_km: equatorial_radius_km,
64            semi_minor_equatorial_radius_km: equatorial_radius_km,
65            polar_radius_km,
66        }
67    }
68}

72impl Ellipsoid {
73    /// Initializes a new [Ellipsoid] shape provided at least its semi major equatorial radius, optionally its semi minor equatorial radius, and optionally its polar radius.
74    /// All units are in kilometers. If the semi minor equatorial radius is not provided, a bi-axial spheroid will be created using the semi major equatorial radius as
75    /// the equatorial radius and using the provided polar axis radius. If only the semi major equatorial radius is provided, a perfect sphere will be built.
76    #[new]
77    #[pyo3(signature=(semi_major_equatorial_radius_km, polar_radius_km=None, semi_minor_equatorial_radius_km=None))]
78    fn py_new(
79        semi_major_equatorial_radius_km: f64,
80        polar_radius_km: Option<f64>,
81        semi_minor_equatorial_radius_km: Option<f64>,
82    ) -> Self {
83        match polar_radius_km {
84            Some(polar_radius_km) => match semi_minor_equatorial_radius_km {
85                Some(semi_minor_equatorial_radius_km) => Self {
86                    semi_major_equatorial_radius_km,
87                    semi_minor_equatorial_radius_km,
88                    polar_radius_km,
89                },
90                None => Self::from_spheroid(semi_major_equatorial_radius_km, polar_radius_km),
91            },
92            None => Self::from_sphere(semi_major_equatorial_radius_km),
93        }
94    }
95
96    fn __str__(&self) -> String {
97        format!("{self}")
98    }
99
100    fn __repr__(&self) -> String {
101        format!("{self} (@{self:p})")
102    }
103
104    fn __richcmp__(&self, other: &Self, op: CompareOp) -> Result<bool, PyErr> {
105        match op {
106            CompareOp::Eq => Ok(self == other),
107            CompareOp::Ne => Ok(self != other),
108            _ => Err(PyErr::new::<PyTypeError, _>(format!(
109                "{op:?} not available"
110            ))),
111        }
112    }
113
114    /// Allows for pickling the object
115    ///
116    /// :rtype: typing.Tuple
117    fn __getnewargs__(&self) -> Result<(f64, Option<f64>, Option<f64>), PyErr> {
118        Ok((
119            self.semi_major_equatorial_radius_km,
120            Some(self.polar_radius_km),
121            Some(self.semi_minor_equatorial_radius_km),
122        ))
123    }
124
125    /// :rtype: float
126    #[getter]
127    fn get_semi_major_equatorial_radius_km(&self) -> PyResult<f64> {
128        Ok(self.semi_major_equatorial_radius_km)
129    }
130    /// :type semi_major_equatorial_radius_km: float
131    #[setter]
132    fn set_semi_major_equatorial_radius_km(
133        &mut self,
134        semi_major_equatorial_radius_km: f64,
135    ) -> PyResult<()> {
136        self.semi_major_equatorial_radius_km = semi_major_equatorial_radius_km;
137        Ok(())
138    }
139    /// :rtype: float
140    #[getter]
141    fn get_polar_radius_km(&self) -> PyResult<f64> {
142        Ok(self.polar_radius_km)
143    }
144    /// :type polar_radius_km: float
145    #[setter]
146    fn set_polar_radius_km(&mut self, polar_radius_km: f64) -> PyResult<()> {
147        self.polar_radius_km = polar_radius_km;
148        Ok(())
149    }
150    /// :rtype: float
151    #[getter]
152    fn get_semi_minor_equatorial_radius_km(&self) -> PyResult<f64> {
153        Ok(self.semi_minor_equatorial_radius_km)
154    }
155    /// :type semi_minor_equatorial_radius_km: float
156    #[setter]
157    fn set_semi_minor_equatorial_radius_km(
158        &mut self,
159        semi_minor_equatorial_radius_km: f64,
160    ) -> PyResult<()> {
161        self.semi_minor_equatorial_radius_km = semi_minor_equatorial_radius_km;
162        Ok(())
163    }

166#[cfg_attr(feature = "python", pymethods)]
167impl Ellipsoid {
168    /// Returns the mean equatorial radius in kilometers
169    ///
170    /// :rtype: float
171    pub fn mean_equatorial_radius_km(&self) -> f64 {
172        (self.semi_major_equatorial_radius_km + self.semi_minor_equatorial_radius_km) / 2.0
173    }
174
175    /// Returns true if the polar radius is equal to the semi minor radius.
176    ///
177    /// :rtype: bool
178    pub fn is_sphere(&self) -> bool {
179        self.is_spheroid()
180            && (self.polar_radius_km - self.semi_minor_equatorial_radius_km).abs() < f64::EPSILON
181    }
182
183    /// Returns true if the semi major and minor radii are equal
184    ///
185    /// :rtype: bool
186    pub fn is_spheroid(&self) -> bool {
187        (self.semi_major_equatorial_radius_km - self.semi_minor_equatorial_radius_km).abs()
188            < f64::EPSILON
189    }
190
191    /// Returns the flattening ratio, computed from the mean equatorial radius and the polar radius
192    ///
193    /// :rtype: float
194    pub fn flattening(&self) -> f64 {
195        (self.mean_equatorial_radius_km() - self.polar_radius_km) / self.mean_equatorial_radius_km()
196    }
197}

199impl fmt::Display for Ellipsoid {
200    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> Result<(), fmt::Error> {
201        if self.is_sphere() {
202            write!(f, "radius = {} km", self.semi_major_equatorial_radius_km)
203        } else if self.is_spheroid() {
204            write!(
205                f,
206                "eq. radius = {} km, polar radius = {} km, f = {}",
207                self.semi_major_equatorial_radius_km,
208                self.polar_radius_km,
209                self.flattening()
210            )
211        } else {
212            write!(
213                f,
214                "major radius = {} km, minor radius = {} km, polar radius = {} km, f = {}",
215                self.semi_major_equatorial_radius_km,
216                self.semi_minor_equatorial_radius_km,
217                self.polar_radius_km,
218                self.flattening()
219            )
220        }
221    }
222}

224impl Encode for Ellipsoid {
225    fn encoded_len(&self) -> der::Result<der::Length> {
226        self.semi_major_equatorial_radius_km.encoded_len()?
227            + self.semi_minor_equatorial_radius_km.encoded_len()?
228            + self.polar_radius_km.encoded_len()?
229    }
230
231    fn encode(&self, encoder: &mut impl Writer) -> der::Result<()> {
232        self.semi_major_equatorial_radius_km.encode(encoder)?;
233        self.semi_minor_equatorial_radius_km.encode(encoder)?;
234        self.polar_radius_km.encode(encoder)
235    }
236}

238impl<'a> Decode<'a> for Ellipsoid {
239    fn decode<R: Reader<'a>>(decoder: &mut R) -> der::Result<Self> {
240        Ok(Self {
241            semi_major_equatorial_radius_km: decoder.decode()?,
242            semi_minor_equatorial_radius_km: decoder.decode()?,
243            polar_radius_km: decoder.decode()?,
244        })
245    }
