32/// Defines a state parameter event finder from the desired value of the scalar expression to compute, precision on timing and value, and the aberration.
33///
34/// :type scalar: ScalarExpr
35/// :type desired_value: float
36/// :type epoch_precision: Duration
37/// :type value_precision: float
38/// :type ab_corr: Aberration, optional
39#[cfg_attr(feature = "python", pyclass)]
40#[cfg_attr(feature = "python", pyo3(module = "anise.analysis"))]
41#[derive(Clone, Debug, Serialize, Deserialize, PartialEq)]
42pub struct Event {
43    /// The state parameter
44    pub scalar: ScalarExpr,
45    /// The desired self.desired_value, must be in the same units as the state parameter
46    pub desired_value: f64,
47    /// The duration precision after which the solver will report that it cannot find any more precise
48    pub epoch_precision: Duration,
49    /// The precision on the desired value. Avoid setting it too low (e.g. 1e-3 degrees) because it may
50    /// cause events to be skipped if the value is not found within the epoch precision.
51    pub value_precision: f64,
52    pub ab_corr: Option<Aberration>,
53}

121impl Event {
122    /// Compute the event finding function of this event provided an Orbit and Almanac.
123    ///
124    /// :type orbit: Orbit
125    /// :type almanac: Almanac
126    /// :rtype: float
127    pub fn eval(&self, orbit: Orbit, almanac: &Almanac) -> Result<f64, AnalysisError> {
128        let current_val = self.scalar.evaluate(orbit, self.ab_corr, almanac)?;
129
130        // Check if the scalar is an angle that needs special handling
131        let is_angle = match self.scalar {
132            ScalarExpr::Element(oe) => oe.is_angle(),
133            ScalarExpr::AngleBetween { a: _, b: _ }
134            | ScalarExpr::BetaAngle
135            | ScalarExpr::SunAngle { observer_id: _ }
136            | ScalarExpr::AzimuthFromLocation {
137                location_id: _,
138                obstructing_body: _,
139            }
140            | ScalarExpr::ElevationFromLocation {
141                location_id: _,
142                obstructing_body: _,
143            } => true,
144            _ => false,
145        };
146
147        if is_angle {
148            // Use the arctan function because it's smooth around zero, but convert back to degrees
149            // for the comparison.
150
151            let current_rad = current_val.to_radians();
152            let desired_rad = self.desired_value.to_radians();
153
154            // Convert the angles to points on a unit circle
155            let (cur_sin, cur_cos) = current_rad.sin_cos();
156            let (des_sin, des_cos) = desired_rad.sin_cos();
157
158            // Calculate the difference vector and find its angle with atan2.
159            // This will be zero only when the angles are identical.
160            let y = cur_sin * des_cos - cur_cos * des_sin; // sin(current - desired)
161            let x = cur_cos * des_cos + cur_sin * des_sin; // cos(current - desired)
162
163            Ok(y.atan2(x).to_degrees())
164        } else {
165            // For all non-angular scalars, use the original logic
166            Ok(current_val - self.desired_value)
167        }
168    }
169
170    /// Pretty print the evaluation of this event for the provided Orbit and Almanac
171    ///
172    /// :type orbit: Orbit
173    /// :type almanac: Almanac
174    /// :rtype: str
175    pub fn eval_string(&self, orbit: Orbit, almanac: &Almanac) -> Result<String, AnalysisError> {
176        let val = self.eval(orbit, almanac)?;
177
178        if self.desired_value.abs() > 1e3 {
179            Ok(format!(
180                "|{} - {:e}| = {val:e} on {}",
181                self.scalar, self.desired_value, orbit.epoch
182            ))
183        } else if self.desired_value > self.value_precision {
184            Ok(format!(
185                "|{} - {:.3}| = {val:.3} on {}",
186                self.scalar, self.desired_value, orbit.epoch
187            ))
188        } else {
189            Ok(format!("|{}| = {val:.3} on {}", self.scalar, orbit.epoch))
190        }
191    }
192}

353/// Enumerates the possible edges of an event in a trajectory.
354///
355/// `EventEdge` is used to describe the nature of a trajectory event, particularly in terms of its temporal dynamics relative to a specified condition or threshold. This enum helps in distinguishing whether the event is occurring at a rising edge, a falling edge, or if the edge is unclear due to insufficient data or ambiguous conditions.
356#[derive(Copy, Clone, Debug, PartialEq, Serialize, Deserialize)]
357#[cfg_attr(feature = "python", pyclass)]
358#[cfg_attr(feature = "python", pyo3(module = "anise.analysis"))]
359pub enum EventEdge {
360    /// Represents a rising edge of the event. This indicates that the event is transitioning from a lower to a higher evaluation of the event. For example, in the context of elevation, a rising edge would indicate an increase in elevation from a lower angle.
361    Rising,
362    /// Represents a falling edge of the event. This is the opposite of the rising edge, indicating a transition from a higher to a lower value of the event evaluator. For example, if tracking the elevation of an object, a falling edge would signify a
363    Falling,
364    /// Represents a local minimum of the event. This indicates that the previous and next values are both greater than the current value.
365    LocalMin,
366    /// Represents a local maximum of the event. This indicates that the previous and next values are both lower than the current value.
367    LocalMax,
368    /// If the edge cannot be clearly defined, it will be marked as unclear. This happens if the event is at a saddle point and the epoch precision is too large to find the exact slope.
369    Unclear,
370}

383/// Represents the details of an event occurring along a trajectory.
384///
385/// `EventDetails` encapsulates the state at which a particular event occurs in a trajectory, along with additional information about the nature of the event. This struct is particularly useful for understanding the dynamics of the event, such as whether it represents a rising or falling edge, or if the edge is unclear.
386///
387/// # Generics
388/// S: Interpolatable - A type that represents the state of the trajectory. This type must implement the `Interpolatable` trait, ensuring that it can be interpolated and manipulated according to the trajectory's requirements.
389#[derive(Clone, PartialEq)]
390#[cfg_attr(feature = "python", pyclass)]
391#[cfg_attr(feature = "python", pyo3(module = "anise.analysis", get_all))]
392pub struct EventDetails {
393    /// The state of the trajectory at the found event.
394    /// :rtype: Orbit
395    pub orbit: Orbit,
396    /// Indicates whether the event is a rising edge, falling edge, or unclear. This helps in understanding the direction of change at the event point.
397    /// :rtype: EventEdge
398    pub edge: EventEdge,
399    /// Numerical evaluation of the event condition, e.g. if seeking the apoapsis, this returns the near zero
400    /// :rtype: float
401    pub value: f64,
402    /// Numertical evaluation of the event condition one epoch step before the found event (used to compute the rising/falling edge).
403    /// :rtype: float
404    pub prev_value: Option<f64>,
405    /// Numertical evaluation of the event condition one epoch step after the found event (used to compute the rising/falling edge).
406    /// :rtype: float
407    pub next_value: Option<f64>,
408    /// Precision of the epoch for this value
409    /// :rtype: Duration
410    pub pm_duration: Duration,
411    /// Store the representation of this event as a string because we can't move or clone the event reference
412    /// :rtype: str
413    pub repr: String,
414}
415
416impl EventDetails {
417    /// Generates detailed information about an event at a specific epoch in a trajectory.
418    ///
419    /// This takes an `Epoch` as an input and returns a `Result<Self, EventError>`.
420    /// It is designed to determine the state of a trajectory at a given epoch, evaluate the specific event at that state, and ascertain the nature of the event (rising, falling, or unclear).
421    /// The initialization intelligently determines the edge type of the event by comparing the event's value at the current, previous, and next epochs.
422    /// It ensures robust event characterization in trajectories.
423    ///
424    /// # Returns
425    /// - `Ok(EventDetails<S>)` if the state at the given epoch can be determined and the event details are successfully evaluated.
426    /// - `Err(EventError)` if there is an error in retrieving the state at the specified epoch.
427    ///
428    pub fn new(
429        state: Orbit,
430        value: f64,
431        event: &Event,
432        prev_state: Option<Orbit>,
433        next_state: Option<Orbit>,
434        almanac: &Almanac,
435    ) -> Result<Self, AnalysisError> {
436        let prev_value = if let Some(state) = prev_state {
437            Some(event.eval(state, almanac)?)
438        } else {
439            None
440        };
441
442        let next_value = if let Some(state) = next_state {
443            Some(event.eval(state, almanac)?)
444        } else {
445            None
446        };
447
448        let edge = if let Some(prev_value) = prev_value {
449            if let Some(next_value) = next_value {
450                if prev_value > value {
451                    if value > next_value {
452                        EventEdge::Falling
453                    } else {
454                        EventEdge::LocalMin
455                    }
456                } else if prev_value < value {
457                    if value < next_value {
458                        EventEdge::Rising
459                    } else {
460                        EventEdge::LocalMax
461                    }
462                } else {
463                    EventEdge::Unclear
464                }
465            } else if prev_value > value {
466                EventEdge::Falling
467            } else {
468                EventEdge::Rising
469            }
470        } else if let Some(next_value) = next_value {
471            if next_value > value {
472                EventEdge::Rising
473            } else {
474                EventEdge::Falling
475            }
476        } else {
477            warn!(
478                "could not determine edge of {event} because state could be queried around {}",
479                state.epoch
480            );
481            EventEdge::Unclear
482        };
483
484        Ok(EventDetails {
485            edge,
486            orbit: state,
487            value,
488            prev_value,
489            next_value,
490            pm_duration: event.epoch_precision,
491            repr: event.eval_string(state, almanac)?,
492        })
493    }
494}
