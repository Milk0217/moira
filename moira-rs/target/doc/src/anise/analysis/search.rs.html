21    /// Find the exact state where the request event happens. The event function is expected to be monotone in the provided interval because we find the event using a Brent solver.
22    /// This will only return _one_ event within the provided bracket.
23    #[allow(clippy::identity_op)]
24    pub fn report_event_once(
25        &self,
26        state_spec: &StateSpec,
27        event: &Event,
28        start_epoch: Epoch,
29        end_epoch: Epoch,
30    ) -> Result<EventDetails, AnalysisError> {
31        let max_iter = 50;
32
33        let has_converged =
34            |xa: f64, xb: f64| (xa - xb).abs() <= event.epoch_precision.to_seconds();
35
36        let xa_e = start_epoch;
37        let xb_e = end_epoch;
38
39        // Search in seconds (convert to epoch just in time)
40        let mut xa = 0.0;
41        let mut xb = (xb_e - xa_e).to_seconds();
42        // Evaluate the event at both bounds
43
44        let ya_state = state_spec.evaluate(xa_e, self)?;
45        let yb_state = state_spec.evaluate(xb_e, self)?;
46        let mut ya = event.eval(ya_state, self)?;
47        let mut yb = event.eval(yb_state, self)?;
48
49        // Check if we're already at the root
50        if ya.abs() <= event.value_precision.abs() {
51            debug!(
52                "{event} -- found with |{ya}| < {} @ {xa_e}",
53                event.value_precision.abs()
54            );
55            let prev_state = state_spec.evaluate(xa_e - event.epoch_precision, self).ok();
56            let next_state = state_spec.evaluate(xa_e + event.epoch_precision, self).ok();
57
58            return EventDetails::new(ya_state, ya, event, prev_state, next_state, self);
59        } else if yb.abs() <= event.value_precision.abs() {
60            debug!(
61                "{event} -- found with |{yb}| < {} @ {xb_e}",
62                event.value_precision.abs()
63            );
64            let prev_state = state_spec.evaluate(xb_e - event.epoch_precision, self).ok();
65            let next_state = state_spec.evaluate(xb_e + event.epoch_precision, self).ok();
66
67            return EventDetails::new(ya_state, ya, event, prev_state, next_state, self);
68        }
69
70        // The Brent solver, from the roots crate (sadly could not directly integrate it here)
71        // Source: https://docs.rs/roots/0.0.5/src/roots/numerical/brent.rs.html#57-131
72
73        let (mut xc, mut yc, mut xd) = (xa, ya, xa);
74        let mut flag = true;
75
76        for _ in 0..max_iter {
77            if ya.abs() <= event.value_precision.abs() {
78                // Can't fail, we got it earlier
79                let epoch = xa_e + xa * Unit::Second;
80                let state = state_spec.evaluate(epoch, self)?;
81                debug!(
82                    "{event} -- found with |{ya}| < {} @ {}",
83                    event.value_precision.abs(),
84                    state.epoch,
85                );
86                let prev_state = state_spec
87                    .evaluate(epoch - event.epoch_precision, self)
88                    .ok();
89                let next_state = state_spec
90                    .evaluate(epoch + event.epoch_precision, self)
91                    .ok();
92
93                return EventDetails::new(state, ya, event, prev_state, next_state, self);
94            }
95            if yb.abs() <= event.value_precision.abs() {
96                // Can't fail, we got it earlier
97                let epoch = xa_e + xb * Unit::Second;
98                let state = state_spec.evaluate(epoch, self)?;
99                debug!(
100                    "{event} -- found with |{yb}| < {} @ {}",
101                    event.value_precision.abs(),
102                    state.epoch
103                );
104                let prev_state = state_spec
105                    .evaluate(epoch - event.epoch_precision, self)
106                    .ok();
107                let next_state = state_spec
108                    .evaluate(epoch + event.epoch_precision, self)
109                    .ok();
110
111                return EventDetails::new(state, ya, event, prev_state, next_state, self);
112            }
113            if has_converged(xa, xb) {
114                // The event isn't in the bracket
115                return Err(AnalysisError::EventNotFound {
116                    start: start_epoch,
117                    end: end_epoch,
118                    event: Box::new(event.clone()),
119                });
120            }
121            let mut s = if (ya - yc).abs() > f64::EPSILON && (yb - yc).abs() > f64::EPSILON {
122                xa * yb * yc / ((ya - yb) * (ya - yc))
123                    + xb * ya * yc / ((yb - ya) * (yb - yc))
124                    + xc * ya * yb / ((yc - ya) * (yc - yb))
125            } else {
126                xb - yb * (xb - xa) / (yb - ya)
127            };
128
129            let cond1 = (s - xb) * (s - (3.0 * xa + xb) / 4.0) > 0.0;
130            let cond2 = flag && (s - xb).abs() >= (xb - xc).abs() / 2.0;
131            let cond3 = !flag && (s - xb).abs() >= (xc - xd).abs() / 2.0;
132            let cond4 = flag && has_converged(xb, xc);
133            let cond5 = !flag && has_converged(xc, xd);
134            if cond1 || cond2 || cond3 || cond4 || cond5 {
135                s = (xa + xb) / 2.0;
136                flag = true;
137            } else {
138                flag = false;
139            }
140
141            let next_try = state_spec.clone().evaluate(xa_e + s * Unit::Second, self)?;
142            let ys = event.eval(next_try, self)?;
143
144            xd = xc;
145            xc = xb;
146            yc = yb;
147
148            if ya * ys < 0.0 {
149                // Root is bracketed between xa and s
150                xb = s;
151                yb = ys;
152            } else {
153                // Root is bracketed between s and xb
154                xa = s;
155                ya = ys;
156            }
157
158            // The `arrange` part from your code is to ensure that `b` is always the best guess.
159            // This is a common practice in Brent solvers.
160            if ya.abs() < yb.abs() {
161                // Swap a and b
162                std::mem::swap(&mut xa, &mut xb);
163                std::mem::swap(&mut ya, &mut yb);
164            }
165        }
166        error!("Brent solver failed after {max_iter} iterations");
167        Err(AnalysisError::EventNotFound {
168            start: start_epoch,
169            end: end_epoch,
170            event: Box::new(event.clone()),
171        })
172    }

174    /// Report all of the states and event details where the provided event occurs.
175    ///
176    /// # Limitations
177    /// This method uses a Brent solver, provides a superlinearity convergence (Golden ratio rate).
178    /// If the function that defines the event is not unimodal, the event finder may not converge correctly.
179    /// After the Brent solver is used, this function will check the median gap between events. Assuming most events are periodic,
180    /// any gap whose median repetition is greater than 125% will be slow searches. This _typically_ finds all of the events ... but it
181    /// may also add duplicates! To prevent reporting duplicate events, the found events are deduplicated if the same event is found
182    /// within 3 times the epoch precision. For example, if the epoch precision is 100 ms, if three events are "found" within 300 ms of each other
183    /// then only one of these three is preserved.
184    ///
185    /// # Heuristic detail
186    /// The initial search step is 1% of the duration requested, if the heuristic is set to None.
187    /// For example, if the trajectory is 100 days long, then we split the trajectory into 100 chunks of 1 day and see whether
188    /// the event is in there. If the event happens twice or more times within 1% of the trajectory duration, only the _one_ of
189    /// such events will be found.
190    #[allow(clippy::identity_op)]
191    pub fn report_events(
192        &self,
193        state_spec: &StateSpec,
194        event: &Event,
195        start_epoch: Epoch,
196        end_epoch: Epoch,
197        heuristic: Option<Duration>,
198    ) -> Result<Vec<EventDetails>, AnalysisError> {
199        if start_epoch == end_epoch {
200            return Err(AnalysisError::EventNotFound {
201                start: start_epoch,
202                end: end_epoch,
203                event: Box::new(event.clone()),
204            });
205        }
206        let heuristic = heuristic.unwrap_or((end_epoch - start_epoch) / 100);
207        debug!("searching for {event} with initial heuristic of {heuristic}");
208
209        let (sender, receiver) = channel();
210
211        let epochs: Vec<Epoch> = TimeSeries::inclusive(start_epoch, end_epoch, heuristic).collect();
212        epochs.into_par_iter().for_each_with(sender, |s, epoch| {
213            if let Ok(event_state) =
214                self.report_event_once(state_spec, event, epoch, epoch + heuristic)
215            {
216                s.send(event_state).unwrap()
217            };
218        });
219        let mut events: Vec<_> = receiver.iter().collect();
220
221        if events.is_empty() {
222            warn!("Heuristic failed to find any {event} event, using slower approach");
223            // Crap, we didn't find the event.
224            // Let's find the min and max of this event throughout the trajectory, and search around there.
225            return self.report_events_slow(state_spec, event, start_epoch, end_epoch);
226        }
227
228        // Remove duplicates and reorder
229        events.sort_by(|s1, s2| s1.orbit.epoch.partial_cmp(&s2.orbit.epoch).unwrap());
230        events.dedup_by(|e1, e2| {
231            (e1.orbit.epoch - e2.orbit.epoch).abs() <= event.epoch_precision * 3.0
232        });
233
234        let possible_gap_times = if events.len() > 1 {
235            // We found some states, let's roughly check if we could have missed some events by searching for periodicity.
236            let mut dt_bw_events = events
237                .iter()
238                .take(events.len() - 1)
239                .zip(events.iter().skip(1))
240                .map(|(first, second)| {
241                    (
242                        first.orbit.epoch,
243                        second.orbit.epoch,
244                        second.orbit.epoch - first.orbit.epoch,
245                    )
246                })
247                .collect::<Vec<(Epoch, Epoch, Duration)>>();
248
249            dt_bw_events.sort_by(|dt1, dt2| dt1.2.cmp(&dt2.2));
250
251            let median_duration = dt_bw_events[dt_bw_events.len() / 2].2;
252
253            dt_bw_events
254                .iter()
255                .copied()
256                .filter(|(_start, _end, dt)| *dt > median_duration * 1.25)
257                .collect::<Vec<(Epoch, Epoch, Duration)>>()
258        } else {
259            vec![(start_epoch, end_epoch, end_epoch - start_epoch)]
260        };
261
262        let prev_len = events.len();
263        // Search specifically these gaps.
264        for (gap_start, gap_end, _) in possible_gap_times {
265            if let Ok(mut gapped_events) = self.report_events_slow(
266                state_spec,
267                event,
268                gap_start + event.epoch_precision,
269                gap_end - event.epoch_precision,
270            ) {
271                events.append(&mut gapped_events);
272            }
273        }
274
275        if events.len() != prev_len {
276            // Remove duplicates and reorder once more
277            events.sort_by(|s1, s2| s1.orbit.epoch.partial_cmp(&s2.orbit.epoch).unwrap());
278            // Dedupliate by the same event with three times the epoch precision,
279            // because that would be one edge and then the other edge of the precision plus one event in the middle
280            // from the slow search.
281            events.dedup_by(|e1, e2| {
282                (e1.orbit.epoch - e2.orbit.epoch).abs() <= event.epoch_precision * 3.0
283            });
284        }
285
286        match events.len() {
287            0 => debug!("event {event} not found"),
288            1 => debug!("event {event} found once on {}", events[0].orbit.epoch),
289            _ => {
290                debug!(
291                    "event {event} found {} times from {} until {}",
292                    events.len(),
293                    events.first().unwrap().orbit.epoch,
294                    events.last().unwrap().orbit.epoch
295                )
296            }
297        };
298
299        Ok(events)
300    }

302    /// Slow approach to finding **all** of the events between two epochs. This will evaluate ALL epochs in between the two bounds.
303    /// This approach is more robust, but more computationally demanding since it's O(N).
304    #[allow(clippy::identity_op)]
305    pub fn report_events_slow(
306        &self,
307        state_spec: &StateSpec,
308        event: &Event,
309        start_epoch: Epoch,
310        end_epoch: Epoch,
311    ) -> Result<Vec<EventDetails>, AnalysisError> {
312        let step = event.epoch_precision;
313
314        let (sender, receiver) = channel();
315
316        let epochs: Vec<Epoch> = TimeSeries::inclusive(start_epoch, end_epoch, step).collect();
317
318        epochs.into_par_iter().for_each_with(sender, |s, epoch| {
319            if let Ok(state) = state_spec.evaluate(epoch, self) {
320                if let Ok(this_eval) = event.eval(state, self) {
321                    if this_eval.abs() < event.value_precision.abs() {
322                        // This is an event!
323                        let prev_state = state_spec
324                            .evaluate(epoch - event.epoch_precision, self)
325                            .ok();
326                        let next_state = state_spec
327                            .evaluate(epoch + event.epoch_precision, self)
328                            .ok();
329
330                        if let Ok(details) =
331                            EventDetails::new(state, this_eval, event, prev_state, next_state, self)
332                        {
333                            if s.send(details).is_err() {
334                                eprintln!("receiver for event search dropped");
335                            }
336                        }
337                    }
338                }
339            }
340        });
341
342        let mut events: Vec<_> = receiver.iter().collect();
343
344        // If there still isn't any match, report that the event was not found
345        if events.is_empty() {
346            return Err(AnalysisError::EventNotFound {
347                start: start_epoch,
348                end: end_epoch,
349                event: Box::new(event.clone()),
350            });
351        }
352
353        // Remove duplicates and reorder once more
354        events.sort_by(|s1, s2| s1.orbit.epoch.partial_cmp(&s2.orbit.epoch).unwrap());
355        Ok(events)
356    }

358    /// Find all event arcs, i.e. the start and stop time of when a given event occurs. This function
359    /// calls the memory and computationally intensive [report_events_slow] function.
360    pub fn report_event_arcs(
361        &self,
362        state_spec: &StateSpec,
363        event: &Event,
364        start_epoch: Epoch,
365        end_epoch: Epoch,
366    ) -> Result<Vec<EventArc>, AnalysisError> {
367        // Step 1: Get all zero-crossings. We will completely ignore their reported 'edge' status, already sorted by time.
368        let crossings = match self.report_events_slow(state_spec, event, start_epoch, end_epoch) {
369            Ok(events) => events,
370            Err(_) => {
371                // No crossings were found. The only possibility for an arc is if the entire
372                // trajectory is within the event.
373                let start_orbit = state_spec.evaluate(start_epoch, self)?;
374                let start_eval = event.eval(start_orbit, self)?;
375                if start_eval > 0.0 {
376                    let end_orbit = state_spec.evaluate(end_epoch, self)?;
377                    let end_eval = event.eval(end_orbit, self)?;
378                    let rise = EventDetails::new(
379                        start_orbit,
380                        start_eval,
381                        event,
382                        None,
383                        Some(end_orbit),
384                        self,
385                    )?;
386                    let fall = EventDetails::new(
387                        end_orbit,
388                        end_eval,
389                        event,
390                        Some(start_orbit),
391                        None,
392                        self,
393                    )?;
394                    return Ok(vec![EventArc { rise, fall }]);
395                }
396                return Ok(Vec::new()); // The event never happens.
397            }
398        };
399
400        // We have at least one crossing at this point.
401        let init_crossing = crossings.first().unwrap();
402        let mut is_inside_arc = init_crossing.value >= 0.0;
403
404        let mut arcs = Vec::new();
405        let mut rise: Option<EventDetails> = None;
406
407        if is_inside_arc {
408            if let Some(next_value) = init_crossing.next_value {
409                if next_value < 0.0 {
410                    // We start at the _end_ of this event.
411                    // Compute the event details for this next epoch.
412                    let fall = self.report_event_once(
413                        state_spec,
414                        event,
415                        init_crossing.orbit.epoch + event.epoch_precision,
416                        init_crossing.orbit.epoch + event.epoch_precision * 2,
417                    )?;
418                    arcs.push(EventArc {
419                        rise: init_crossing.clone(),
420                        fall,
421                    });
422                }
423            }
424            rise = Some(init_crossing.clone());
425        }
426
427        for crossing in crossings.iter().skip(1) {
428            let event_value = crossing.value;
429            if event_value >= 0.0 {
430                // We're in an arc.
431                if let Some(next_value) = crossing.next_value {
432                    if next_value < 0.0 && rise.is_some() {
433                        // At the next immediate step, the event ends, so this marks the end of the arc.
434                        arcs.push(EventArc {
435                            rise: rise.clone().unwrap(),
436                            fall: crossing.clone(),
437                        });
438                        is_inside_arc = false;
439                        continue; // Move onto the next crossing.
440                    }
441                    // else we're still in the arc on the next step.
442                }
443                // If we weren't in an arc, store this as the new rise.
444                if !is_inside_arc {
445                    rise = Some(crossing.clone());
446                    is_inside_arc = true;
447                }
448            } else if is_inside_arc {
449                // We aren't in an arc but we were until this event crossing.
450                // Close out this arc.
451                if let Some(rise) = rise.take() {
452                    arcs.push(EventArc {
453                        rise,
454                        fall: crossing.clone(),
455                    });
456                }
457                is_inside_arc = false;
458            }
459        }
460
461        if is_inside_arc {
462            if let Some(rise) = rise {
463                if let Some(fall) = crossings.last().cloned() {
464                    arcs.push(EventArc { rise, fall });
465                }
466            }
467        }
468
469        Ok(arcs)
470    }
