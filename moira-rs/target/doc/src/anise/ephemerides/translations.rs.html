26/// **Limitation:** no translation or rotation may have more than 8 nodes.
27pub const MAX_TREE_DEPTH: usize = 8;

30    /// Returns the Cartesian state of the target frame as seen from the observer frame at the provided epoch, and optionally given the aberration correction.
31    ///
32    /// # SPICE Compatibility
33    /// This function is the SPICE equivalent of spkezr: `spkezr(TARGET_ID, EPOCH_TDB_S, ORIENTATION_ID, ABERRATION, OBSERVER_ID)`
34    /// In ANISE, the TARGET_ID and ORIENTATION are provided in the first argument (TARGET_FRAME), as that frame includes BOTH
35    /// the target ID and the orientation of that target. The EPOCH_TDB_S is the epoch in the TDB time system, which is computed
36    /// in ANISE using Hifitime. THe ABERRATION is computed by providing the optional Aberration flag. Finally, the OBSERVER
37    /// argument is replaced by OBSERVER_FRAME: if the OBSERVER_FRAME argument has the same orientation as the TARGET_FRAME, then this call
38    /// will return exactly the same data as the spkerz SPICE call.
39    ///
40    /// # Warning
41    /// This function only performs the translation and no rotation whatsoever. Use the `transform` function instead to include rotations.
42    ///
43    /// # Note
44    /// This function performs a recursion of no more than twice the [MAX_TREE_DEPTH].
45    ///
46    /// # Algorithm
47    /// 1.  Find the common ancestor of the `target_frame` and `observer_frame` in the ephemeris tree using `common_ephemeris_path`.
48    /// 2.  Initialize the state vectors for both the forward (observer to common ancestor) and backward (target to common ancestor) paths.
49    /// 3.  Iteratively traverse the ephemeris tree from the observer and target frames up to the common ancestor, accumulating the state vectors at each step using `translation_parts_to_parent`.
50    /// 4.  If aberration corrections are requested, calculate the one-way light time and apply the correction to the target's position.
51    /// 5.  The final state is the difference between the backward and forward state vectors.
52    pub fn translate(
53        &self,
54        target_frame: Frame,
55        mut observer_frame: Frame,
56        epoch: Epoch,
57        ab_corr: Option<Aberration>,
58    ) -> Result<CartesianState, EphemerisError> {
59        if observer_frame == target_frame {
60            // Both frames match, return this frame's hash (i.e. no need to go higher up).
61            return Ok(CartesianState::zero(observer_frame));
62        }
63
64        // If there is no frame info, the user hasn't loaded this frame, but might still want to compute a translation.
65        if let Ok(obs_frame_info) = self.frame_info(observer_frame) {
66            // User has loaded the planetary data for this frame, so let's use that as the to_frame.
67            observer_frame = obs_frame_info;
68        }
69
70        match ab_corr {
71            None => {
72                // Geometric case (no aberration correction)
73                let (node_count, _path, common_node) =
74                    self.common_ephemeris_path(observer_frame, target_frame, epoch)?;
75
76                // The `fwrd` variables store the state of the observer frame relative to the common ancestor.
77                let (mut pos_fwrd, mut vel_fwrd, mut frame_fwrd) =
78                    if observer_frame.ephem_origin_id_match(common_node) {
79                        // Observer is the common ancestor, so state is zero.
80                        (Vector3::zeros(), Vector3::zeros(), observer_frame)
81                    } else {
82                        self.translation_parts_to_parent(observer_frame, epoch)?
83                    };
84
85                // The `bwrd` variables store the state of the target frame relative to the common ancestor.
86                let (mut pos_bwrd, mut vel_bwrd, mut frame_bwrd) =
87                    if target_frame.ephem_origin_id_match(common_node) {
88                        // Target is the common ancestor, so state is zero.
89                        (Vector3::zeros(), Vector3::zeros(), target_frame)
90                    } else {
91                        self.translation_parts_to_parent(target_frame, epoch)?
92                    };
93
94                // Traverse the ephemeris tree from both the observer and target up to the common ancestor.
95                for _ in 0..node_count {
96                    if !frame_fwrd.ephem_origin_id_match(common_node) {
97                        // Accumulate the state from the current forward frame to its parent.
98                        let (cur_pos_fwrd, cur_vel_fwrd, cur_frame_fwrd) =
99                            self.translation_parts_to_parent(frame_fwrd, epoch)?;
100
101                        pos_fwrd += cur_pos_fwrd;
102                        vel_fwrd += cur_vel_fwrd;
103                        frame_fwrd = cur_frame_fwrd;
104                    }
105
106                    if !frame_bwrd.ephem_origin_id_match(common_node) {
107                        // Accumulate the state from the current backward frame to its parent.
108                        let (cur_pos_bwrd, cur_vel_bwrd, cur_frame_bwrd) =
109                            self.translation_parts_to_parent(frame_bwrd, epoch)?;
110
111                        pos_bwrd += cur_pos_bwrd;
112                        vel_bwrd += cur_vel_bwrd;
113                        frame_bwrd = cur_frame_bwrd;
114                    }
115                }
116
117                // The final state is the difference between the state of the target and the observer, both relative to the common ancestor.
118                Ok(CartesianState {
119                    radius_km: pos_bwrd - pos_fwrd,
120                    velocity_km_s: vel_bwrd - vel_fwrd,
121                    epoch,
122                    frame: observer_frame.with_orient(target_frame.orientation_id),
123                })
124            }
125            Some(ab_corr) => {
126                // Aberration correction case. This is a rewrite of NAIF SPICE's `spkapo`.
127
128                // Find the geometric position of the observer body with respect to the solar system barycenter (SSB).
129                let obs_ssb = self.translate(observer_frame, SSB_J2000, epoch, None)?;
130                let obs_ssb_pos_km = obs_ssb.radius_km;
131                let obs_ssb_vel_km_s = obs_ssb.velocity_km_s;
132
133                // Find the geometric position of the target body with respect to the SSB at the same epoch.
134                let tgt_ssb = self.translate(target_frame, SSB_J2000, epoch, None)?;
135                let tgt_ssb_pos_km = tgt_ssb.radius_km;
136                let tgt_ssb_vel_km_s = tgt_ssb.velocity_km_s;
137
138                // Calculate the initial relative position and velocity.
139                let mut rel_pos_km = tgt_ssb_pos_km - obs_ssb_pos_km;
140                let mut rel_vel_km_s = tgt_ssb_vel_km_s - obs_ssb_vel_km_s;
141
142                // Compute the initial one-way light time.
143                let mut one_way_lt_s = rel_pos_km.norm() / SPEED_OF_LIGHT_KM_S;
144
145                // Iteratively correct for the one-way light time.
146                // The number of iterations depends on whether a converged solution is requested.
147                let num_it = if ab_corr.converged { 3 } else { 1 };
148                let lt_sign = if ab_corr.transmit_mode { 1.0 } else { -1.0 };
149
150                for _ in 0..num_it {
151                    // Calculate the light-time corrected epoch.
152                    let epoch_lt = epoch + lt_sign * one_way_lt_s * TimeUnit::Second;
153                    // Find the position of the target at the corrected epoch.
154                    let tgt_ssb = self
155                        .translate(target_frame, SSB_J2000, epoch_lt, None)
156                        .map_err(|e| EphemerisError::LightTimeCorrection {
157                            epoch,
158                            epoch_lt,
159                            ab_corr,
160                            source: Box::new(e),
161                        })?;
162                    let tgt_ssb_pos_km = tgt_ssb.radius_km;
163                    let tgt_ssb_vel_km_s = tgt_ssb.velocity_km_s;
164                    // Update the relative position.
165                    rel_pos_km = tgt_ssb_pos_km - obs_ssb_pos_km;
166                    let r_norm = rel_pos_km.norm();
167                    // Update the light-time corrected relative velocity.
168                    let geometric_rel_vel = tgt_ssb_vel_km_s - obs_ssb_vel_km_s;
169                    if r_norm > 0.0 {
170                        let inv_c_r = 1.0 / (SPEED_OF_LIGHT_KM_S * r_norm);
171                        let r_dot_v_rel = rel_pos_km.dot(&geometric_rel_vel);
172                        let r_dot_v_tgt = rel_pos_km.dot(&tgt_ssb_vel_km_s);
173                        // The rate of change of light time.
174                        let dlt = (inv_c_r * r_dot_v_rel) / (1.0 - lt_sign * r_dot_v_tgt * inv_c_r);
175                        rel_vel_km_s = tgt_ssb_vel_km_s * (1.0 + lt_sign * dlt) - obs_ssb_vel_km_s;
176                    } else {
177                        rel_vel_km_s = geometric_rel_vel;
178                    }
179                    // Update the one-way light time for the next iteration.
180                    one_way_lt_s = r_norm / SPEED_OF_LIGHT_KM_S;
181                }
182
183                // If stellar aberration correction is requested, apply it now.
184                if ab_corr.stellar {
185                    rel_pos_km = stellar_aberration(rel_pos_km, obs_ssb_vel_km_s, ab_corr)
186                        .context(EphemerisPhysicsSnafu {
187                            action: "computing stellar aberration",
188                        })?;
189                }
190
191                Ok(CartesianState {
192                    radius_km: rel_pos_km,
193                    velocity_km_s: rel_vel_km_s,
194                    epoch,
195                    frame: observer_frame.with_orient(target_frame.orientation_id),
196                })
197            }
198        }
199    }

201    /// Returns the geometric position vector, velocity vector, and acceleration vector needed to translate the `from_frame` to the `to_frame`, where the distance is in km, the velocity in km/s, and the acceleration in km/s^2.
202    pub fn translate_geometric(
203        &self,
204        target_frame: Frame,
205        observer_frame: Frame,
206        epoch: Epoch,
207    ) -> Result<CartesianState, EphemerisError> {
208        self.translate(target_frame, observer_frame, epoch, Aberration::NONE)
209    }

211    /// Translates the provided Cartesian state into the requested observer frame
212    ///
213    /// **WARNING:** This function only performs the translation and no rotation _whatsoever_. Use the `transform_to` function instead to include rotations.
214    pub fn translate_to(
215        &self,
216        state: CartesianState,
217        mut observer_frame: Frame,
218        ab_corr: Option<Aberration>,
219    ) -> Result<CartesianState, EphemerisError> {
220        let frame_state = self.translate(state.frame, observer_frame, state.epoch, ab_corr)?;
221        let mut new_state = state.add_unchecked(&frame_state);
222
223        // If there is no frame info, the user hasn't loaded this frame, but might still want to compute a translation.
224        if let Ok(obs_frame_info) = self.frame_info(observer_frame) {
225            // User has loaded the planetary data for this frame, so let's use that as the to_frame.
226            observer_frame = obs_frame_info;
227        }
228        new_state.frame = observer_frame.with_orient(state.frame.orientation_id);
229        Ok(new_state)
230    }

232    /// Translates a state with its origin (`to_frame`) and given its units (distance_unit, time_unit), returns that state with respect to the requested frame
233    ///
234    /// **WARNING:** This function only performs the translation and no rotation _whatsoever_. Use the `transform_state_to` function instead to include rotations.
235    #[allow(clippy::too_many_arguments)]
236    pub fn translate_state_to(
237        &self,
238        position: Vector3,
239        velocity: Vector3,
240        from_frame: Frame,
241        observer_frame: Frame,
242        epoch: Epoch,
243        ab_corr: Option<Aberration>,
244        distance_unit: LengthUnit,
245        time_unit: TimeUnit,
246    ) -> Result<CartesianState, EphemerisError> {
247        let dist_unit_factor = LengthUnit::Kilometer.from_meters() * distance_unit.to_meters();
248        let time_unit_factor = time_unit.in_seconds();
249
250        let state = CartesianState {
251            radius_km: position * dist_unit_factor,
252            velocity_km_s: velocity * dist_unit_factor / time_unit_factor,
253            epoch,
254            frame: from_frame,
255        };
256
257        self.translate_to(state, observer_frame, ab_corr)
258    }
