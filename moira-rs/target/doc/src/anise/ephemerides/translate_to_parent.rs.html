30impl Almanac {
31    /// Returns the position vector and velocity vector of the `source` with respect to its parent in the ephemeris at the provided epoch,
32    /// Units are those used in the SPK, typically distances are in kilometers and velocities in kilometers per second.
33    ///
34    /// # Errors
35    /// + As of now, some interpolation types are not supported, and if that were to happen, this would return an error.
36    ///
37    /// # Warning
38    /// This function only performs the translation and no rotation whatsoever. Use the `transform_to_parent_from` function instead to include rotations.
39    pub(crate) fn translation_parts_to_parent(
40        &self,
41        source: Frame,
42        epoch: Epoch,
43    ) -> Result<(Vector3, Vector3, Frame), EphemerisError> {
44        // First, let's find the SPK summary for this frame.
45        let (summary, spk_no, idx_in_spk) =
46            self.spk_summary_at_epoch(source.ephemeris_id, epoch)?;
47
48        let new_frame = source.with_ephem(summary.center_id);
49
50        trace!("translate {source} wrt to {new_frame} @ {epoch:E}");
51
52        // This should not fail because we've fetched the spk_no from above with the spk_summary_at_epoch call.
53        let (_, spk_data) = self
54            .spk_data
55            .get_index(spk_no)
56            .ok_or(EphemerisError::Unreachable)?;
57
58        // Now let's simply evaluate the data
59
60        let (pos_km, vel_km_s) = match summary.data_type()? {
61            DafDataType::Type1ModifiedDifferenceArray => {
62                let data =
63                    spk_data
64                        .nth_data::<ModifiedDiffType1>(idx_in_spk)
65                        .context(SPKSnafu {
66                            action: "fetching data for interpolation",
67                        })?;
68                data.evaluate(epoch, summary)
69                    .context(EphemInterpolationSnafu)?
70            }
71            DafDataType::Type2ChebyshevTriplet => {
72                let data =
73                    spk_data
74                        .nth_data::<Type2ChebyshevSet>(idx_in_spk)
75                        .context(SPKSnafu {
76                            action: "fetching data for interpolation",
77                        })?;
78                data.evaluate(epoch, summary)
79                    .context(EphemInterpolationSnafu)?
80            }
81            DafDataType::Type3ChebyshevSextuplet => {
82                let data =
83                    spk_data
84                        .nth_data::<Type3ChebyshevSet>(idx_in_spk)
85                        .context(SPKSnafu {
86                            action: "fetching data for interpolation",
87                        })?;
88                data.evaluate(epoch, summary)
89                    .context(EphemInterpolationSnafu)?
90            }
91            DafDataType::Type9LagrangeUnequalStep => {
92                let data = spk_data
93                    .nth_data::<LagrangeSetType9>(idx_in_spk)
94                    .context(SPKSnafu {
95                        action: "fetching data for interpolation",
96                    })?;
97                data.evaluate(epoch, summary)
98                    .context(EphemInterpolationSnafu)?
99            }
100            DafDataType::Type13HermiteUnequalStep => {
101                let data = spk_data
102                    .nth_data::<HermiteSetType13>(idx_in_spk)
103                    .context(SPKSnafu {
104                        action: "fetching data for interpolation",
105                    })?;
106                data.evaluate(epoch, summary)
107                    .context(EphemInterpolationSnafu)?
108            }
109            dtype => {
110                return Err(EphemerisError::SPK {
111                    action: "translation to parent",
112                    source: DAFError::UnsupportedDatatype {
113                        dtype,
114                        kind: "SPK computations",
115                    },
116                })
117            }
118        };
119
120        Ok((pos_km, vel_km_s, new_frame))
121    }
122}

124#[cfg_attr(feature = "python", pymethods)]
125impl Almanac {
126    /// Performs the GEOMETRIC translation to the parent. Use translate_from_to for aberration.
127    ///
128    /// :type source: Frame
129    /// :type epoch: Epoch
130    /// :rtype: Orbit
131    pub fn translate_to_parent(
132        &self,
133        source: Frame,
134        epoch: Epoch,
135    ) -> Result<CartesianState, EphemerisError> {
136        let (radius_km, velocity_km_s, frame) = self.translation_parts_to_parent(source, epoch)?;
137
138        Ok(CartesianState {
139            radius_km,
140            velocity_km_s,
141            epoch,
142            frame,
143        })
144    }
